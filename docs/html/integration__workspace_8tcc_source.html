103  /**
104  * Increase the heap start point until the current segment has a smaller
105  * depth than the current maximum depth. After each increment rebuild
106  * the heap from the new start point so the new start point is the largest
107  * error.
108  *
109  * Usage:
110  * In the caller the smallest interval (at max depth) has the largest error.
111  * Before bisecting decrease the sum of the errors over the larger intervals
112  * (error_over_large_intervals) and perform extrapolation.
113  */
114  template<typename Tp, typename RetTp>
115  bool
116  integration_workspace<Tp, RetTp>::increment_curr_index()
117  {
118  size_t limit = this->max_size();
119  size_t last = this->size() - 1 ;
120  size_t jupbnd = last > 1 + limit / 2
121  ? limit + 1 - last
122  : last;
123 
124  const auto i_max = this->curr_index();
125  for (auto k = i_max; k <= jupbnd; ++k)
126  {
127  if (this->m_ival[k].depth < this->max_depth())
128  return true;
129  else if (this->curr_index() + 1 < this->size())
130  {
131  ++this->m_curr_index;
132  this->sort_error();
133  }
134  }
135  return false;
136  }

138  /**
