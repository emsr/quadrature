105  /**
106  * Increase the heap start point until the current segment has a smaller
107  * depth than the current maximum depth. After each increment rebuild
108  * the heap from the new start point so the new start point is the largest
109  * error.
110  *
111  * Usage:
112  * In the caller the smallest interval (at max depth) has the largest error.
113  * Before bisecting decrease the sum of the errors over the larger intervals
114  * (error_over_large_intervals) and perform extrapolation.
115  */
116  template<typename _Tp, typename _RetTp>
117  bool
118  integration_workspace<_Tp, _RetTp>::increment_curr_index()
119  {
120  size_t __limit = this->max_size();
121  size_t __last = this->size() - 1 ;
122  size_t __jupbnd = __last > 1 + __limit / 2
123  ? __limit + 1 - __last
124  : __last;
125 
126  const auto __i_max = this->curr_index();
127  for (auto __k = __i_max; __k <= __jupbnd; ++__k)
128  {
129  if (this->_M_ival[__k].__depth < this->max_depth())
130  return true;
131  else if (this->curr_index() + 1 < this->size())
132  {
133  ++this->_M_curr_index;
134  this->sort_error();
135  }
136  }
137  return false;
138  }
