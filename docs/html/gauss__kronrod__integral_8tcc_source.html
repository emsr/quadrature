41  template<typename _Tp>
42  gauss_kronrod_integral<_Tp>::gauss_kronrod_integral(unsigned __gk_rule)
43  : _M_rule{__gk_rule},
44  _M_x_kronrod{}, _M_w_gauss{}, _M_w_kronrod{}
45  {
46  const int __n = (this->_M_rule - 1) / 2;
47  const auto __eps = 4 * std::numeric_limits<_Tp>::epsilon();
48  __build_gauss_kronrod(__n, __eps,
49  this->_M_x_kronrod, this->_M_w_gauss,
50  this->_M_w_kronrod);
51  }

53  template<typename _Tp>
54  inline bool
55  __test_positivity(_Tp __result, _Tp __resabs)
56  {
57  return(std::abs(__result)
58  >= (1 - 50 * std::numeric_limits<_Tp>::epsilon()) * __resabs);
59  }

505  template<typename _FuncTp,
506  typename _KronrodIter, typename _GaussIter>
507  auto
508  gauss_kronrod_integral<_Tp>::
509  _S_integrate(const _KronrodIter& __x_kronrod,
510  const _GaussIter& __w_gauss,
511  const _KronrodIter& __w_kronrod,
512  _FuncTp __func, _Tp __lower, _Tp __upper)
513  -> gauss_kronrod_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
514  {
515  using _RetTp = std::invoke_result_t<_FuncTp, _Tp>;
516  using _AreaTp = decltype(_RetTp{} * _Tp{});
517 
518  const auto _KronrodSz = std::size(__x_kronrod);
519  //const auto _GaussSz = std::size(__w_gauss);
520  //static_assert(_KronrodSz == 2 * _GaussSz + (_KronrodSz & 1));
521  //std::array<_AreaTp, _KronrodSz> __fv1;
522  //std::array<_AreaTp, _KronrodSz> __fv2;
523  std::vector<_AreaTp> __fv1(_KronrodSz);
524  std::vector<_AreaTp> __fv2(_KronrodSz);
525 
526  const auto __center = (__lower + __upper) / _Tp{2};
527  const auto __half_length = (__upper - __lower) / _Tp{2};
528  const auto __abs_half_length = std::abs(__half_length);
529  const auto __f_center = __func(__center);
530 
531  auto __result_gauss = _AreaTp{0};
532  auto __result_kronrod = __f_center * __w_kronrod[_KronrodSz - 1];
533  auto __result_abs = std::abs(__result_kronrod);
534 
535  if (_KronrodSz % 2 == 0)
536  __result_gauss = __f_center * __w_gauss[_KronrodSz / 2 - 1];
537 
538  for (std::size_t __jj = 0; __jj < (_KronrodSz - 1) / 2; ++__jj)
539  {
540  const std::size_t __jtw = __jj * 2 + 1;
541  const auto __abscissa = __half_length * __x_kronrod[__jtw];
542  const auto __fval1 = __func(__center - __abscissa);
543  const auto __fval2 = __func(__center + __abscissa);
544  const auto __fsum = __fval1 + __fval2;
545  __fv1[__jtw] = __fval1;
546  __fv2[__jtw] = __fval2;
547 
548  __result_gauss += __w_gauss[__jj] * __fsum;
549  __result_kronrod += __w_kronrod[__jtw] * __fsum;
550  __result_abs += __w_kronrod[__jtw]
551  * (std::abs(__fval1) + std::abs(__fval2));
552  }
553 
554  for (std::size_t __jj = 0; __jj < _KronrodSz / 2; ++__jj)
555  {
556  std::size_t __jtwm1 = __jj * 2;
557  const auto __abscissa = __half_length * __x_kronrod[__jtwm1];
558  const auto __fval1 = __func(__center - __abscissa);
559  const auto __fval2 = __func(__center + __abscissa);
560  __fv1[__jtwm1] = __fval1;
561  __fv2[__jtwm1] = __fval2;
562 
563  __result_kronrod += __w_kronrod[__jtwm1] * (__fval1 + __fval2);
564  __result_abs += __w_kronrod[__jtwm1]
565  * (std::abs(__fval1) + std::abs(__fval2));
566  }
567 
568  auto __mean = __result_kronrod / _Tp{2};
569  auto __result_asc = __w_kronrod[_KronrodSz - 1]
570  * std::abs(__f_center - __mean);
571 
572  for (std::size_t __jj = 0; __jj < _KronrodSz - 1; ++__jj)
573  __result_asc += __w_kronrod[__jj]
574  * (std::abs(__fv1[__jj] - __mean)
575  + std::abs(__fv2[__jj] - __mean));
576 
577  auto __err = (__result_kronrod - __result_gauss) * __half_length;
578 
579  __result_kronrod *= __half_length;
580  __result_abs *= __abs_half_length;
581  __result_asc *= __abs_half_length;
582 
583  return {__result_kronrod,
584  __rescale_error(__err, __result_abs, __result_asc),
585  __result_abs, __result_asc};
586  }
