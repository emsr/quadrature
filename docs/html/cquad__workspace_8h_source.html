54  /**
55  * Comparison of cquad intervals.
56  */

58  bool
59  operator<(const cquad_interval<Tp, RetTp>& ivl,
60  const cquad_interval<Tp, RetTp>& ivr)

72  /**
73  * The workspace is a collection of intervals.
74  * Actually, it is a priority queue where the priority
75  * is the absolute error of the interval integral.
76  */
77  template<typename Tp, typename RetTp>
78  struct cquad_workspace
79  {
80  using AreaTp = decltype(RetTp{} * Tp{});
81  using AbsAreaTp = decltype(std::abs(AreaTp{}));
82 
83  std::vector<cquad_interval<Tp, RetTp>> m_ival;
84 
85  cquad_workspace(std::size_t len = 200)
86  : m_ival()
87  { this->m_ival.reserve(len); }
88 
89  std::size_t size() const
90  { return this->m_ival.size(); }
91 
92  std::size_t
93  capacity() const
94  { return this->m_ival.capacity(); }
95 
96  typename std::vector<cquad_interval<Tp, RetTp>>::iterator
97  begin()

101  end()
102  { return this->m_ival.end(); }
103 
104  const cquad_interval<Tp, RetTp>&
105  top() const
106  { return this->m_ival[0]; }
107 
108  cquad_interval<Tp, RetTp>&
109  top()
110  { return this->m_ival[0]; }
111 
112  void
113  clear()
114  { this->m_ival.clear(); }
115 
116  void
117  push(const cquad_interval<Tp, RetTp>& iv)
118  {
119  this->m_ival.push_back(iv);
120  std::push_heap(this->begin(), this->end(),
121  cquad_interval_comp<Tp, RetTp>{});
122  }
123 
124  void
125  pop()
126  {
127  std::pop_heap(this->begin(), this->end());
128  this->m_ival.pop_back();
129  }
130 
131  void
132  update()
133  {
134  std::make_heap(this->begin(), this->end(),
135  cquad_interval_comp<Tp, RetTp>{});
136  }

138  AreaTp
139  total_integral() const
140  {
141  auto tot_igral = AreaTp{0};
142  for (auto& iv : m_ival)
143  tot_igral += iv.m_result;
144  return tot_igral;
145  }
146 
147  AbsAreaTp
148  total_error() const
149  {
150  auto tot_error = AbsAreaTp{0};
151  for (auto& iv : m_ival)
152  tot_error += iv.m_abs_error;
153  return tot_error;
154  }
