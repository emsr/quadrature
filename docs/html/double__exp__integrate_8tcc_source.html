28  /**
29  * @f[
30  * \int_{-1}^{+1}f(x)dx
31  * @f]
32  * Making the change of variables:
33  * @f[
34  * x = tanh\left[\frac{\pi}{2}sinh(u)\right],
35  * dx = \frac{\pi}{2}
36  * \frac{cosh(u)}{cosh^2\left[\frac{\pi}{2}sinh(u)\right]}du
37  * @f]
38  * gives the following integral:
39  * @f[
40  * \int_{-\infty}^{+\infty}f(tanh\left[\frac{\pi}{2}sinh(u)\right])
41  * = \sum_{k=-n}^{+n}
42  * @f]
43  */
44  template<typename _Tp, typename _FuncTp>
45  adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
46  integrate_tanh_sinh(_FuncTp __func, _Tp __a, _Tp __b,
47  _Tp __max_abs_err, _Tp __max_rel_err,
48  int __max_iter)
49  {
50  const auto _S_pi_4 = __gnu_cxx::math::__pi_quarter_v<_Tp>;
51 
52  int __n = 16;
53  __n /= 2;
54 
55  // Find K = ln(ln(max_number))
56  const auto __k_max = std::log(std::log(std::numeric_limits<_Tp>::max()))
57  - _Tp{1};
58  auto __h = __k_max / __n;
59 
60  auto __sum = __func((__a + __b) / _Tp{2}) / _Tp{2};
61  decltype(__sum) __sum1{}, __sum2{};
62  for (int __k = -__n; __k < 0; ++__k)
63  {
64  const auto __u = __h * _Tp(__k);
65  const auto __eu = std::exp(__u);
66  const auto __cosh = __eu + _Tp{1} / __eu;
67  const auto __sinh = __eu - _Tp{1} / __eu;
68  const auto __esh = std::exp(_S_pi_4 * __sinh);
69  const auto __w = __esh + _Tp{1} / __esh;
70  const auto __dxdu = __cosh / (__w * __w);
71  const auto __x1 = (__b * __esh + __a / __esh) / __w;
72  if (__x1 != __a && __x1 != __b)
73  __sum1 += __dxdu * __func(__x1);
74  const auto __x2 = (__a * __esh + __b / __esh) / __w;
75  if (__x2 != __a && __x2 != __b)
76  __sum2 += __dxdu * __func(__x2);
77  }
78 
79  // Interlace values; don't go past the rightmost point.
80  decltype(__sum) __prev_sum{};
81  for (int __iter = 0; __iter < __max_iter; ++__iter)
82  {
83  __prev_sum = __sum + __sum1 + __sum2;
84  for (int __k = -__n; __k < 0; ++__k)
85  {
86  const auto __u = __h * _Tp(__k + 0.5);
87  const auto __eu = std::exp(__u);
88  // A standard sinhcosh would be a nice idea along with sincos.
89  const auto __cosh = __eu + _Tp{1} / __eu;
90  const auto __sinh = __eu - _Tp{1} / __eu;
91  const auto __esh = std::exp(_S_pi_4 * __sinh);
92  const auto __w = __esh + _Tp{1} / __esh;
93  const auto __dxdu = __cosh / (__w * __w);
94  // natural: x1 = (s - 1/s) / (s + 1/s)
95  const auto __x1 = (__b * __esh + __a / __esh) / __w;
96  if (__x1 != __a && __x1 != __b)
97  __sum1 += __dxdu * __func(__x1);
98  // natural: x2 = (-s + 1/s) / (s + 1/s)
99  const auto __x2 = (__a * __esh + __b / __esh) / __w;
100  if (__x2 != __a && __x2 != __b)
101  __sum2 += __dxdu * __func(__x2);
102  }
103 
104  const auto __curr_sum = __sum + __sum1 + __sum2;
105  if (auto __abs_del = std::abs(__curr_sum - __prev_sum);
106  __abs_del < __max_abs_err
107  || __abs_del < std::abs(__max_rel_err * __curr_sum))
108  break;
109 
110  __n *= 2;
111  __h /= _Tp{2};
112  }
113 
114  const auto __fact = _Tp{2} * (__b - __a) * _S_pi_4 * __h;
115  const auto __tot_sum = __sum + __sum1 + __sum2;
116  return {__fact * __tot_sum, __fact * std::abs(__tot_sum - __prev_sum)};
117  }
