30  /**
31  * Discrete Fourier transform on complex data.
32  */
33  template<typename Tp>
34  void
35  discrete_fourier_transform(bool do_forward,
36  std::vector<std::complex<Tp>>& z)
37  {
38  const auto sign(do_forward ? Tp{+1} : Tp{-1});
39  const auto len = z.size();
40 
41  std::vector<std::complex<Tp>> result;
42  result.reserve(len);
43 
44  // Do matrix multiplication.
45  for (std::size_t i = 0; i < len; ++i)
46  {
47  phase_iterator coefficient_iter(sign, i, len);
48  result.push_back(std::inner_product(z.begin(), z.end(),
49  coefficient_iter, std::complex<Tp>{0}));
50  }
51 
52  // Rescale if forward.
53  if (do_forward)
54  {
55  const auto norm = Tp{1} / Tp(len);
56  for (std::size_t i = 0; i < len; ++i)
57  result[i] *= norm;
58  }
59 
60  // Copy data back (swap is fast!).
61  z.swap(result);
62  }

64  /**
65  * Fast Fourier Transform on complex data.
66  */
67  template<typename Tp>
68  void
69  fast_fourier_transform(std::vector<std::complex<Tp>>& z)
70  {
71  const auto len = z.size();
72  if (len % 2 == 1) // Too bad, we're odd.
73  discrete_fourier_transform(true, z);
74  else // Good, we're even.
75  {
76  // Let's divide and conquer!
77  std::vector<std::complex<Tp>> odd;
78  std::vector<std::complex<Tp>> even;
79  const auto halflen = len / 2;
80  odd.reserve(halflen);
81  even.reserve(halflen);
82  for (auto run = z.cbegin(); run != z.cend(); ++run)
83  {
84  even.push_back(*run);
85  ++run;
86  odd.push_back(*run);
87  }
88  fast_fourier_transform(even);
89  fast_fourier_transform(odd);
90  phase_iterator omega_iter(Tp{1}, 1, len);
91  for (std::size_t i = 0, j = halflen; i < halflen;
92  ++i, ++j, ++omega_iter)
93  {
94  z[i] = (even[i] + *omega_iter * odd[i]) / Tp{2};
95  // The next line works because omega^(length/2) = -1.
96  z[j] = (even[i] - *omega_iter * odd[i]) / Tp{2};
97  }
98  }
99  }

101  /**
102  * Inverse Fast Fourier Transform on complex data.
103  */
104  template<typename Tp>
105  void
106  inv_fast_fourier_transform(std::vector<std::complex<Tp>>& z)
107  {
108  const std::size_t len = z.size();
109  if (len % 2 == 1) // Too bad, we're odd.
110  discrete_fourier_transform(false, z);
111  else // Good, we are even.
112  {
113  // Let's divide and conquer!
114  std::vector<std::complex<Tp>> odd;
115  std::vector<std::complex<Tp>> even;
116  const auto halflen = len / 2;
117  odd.reserve(halflen);
118  even.reserve(halflen);
119  for (auto run = z.cbegin(); run != z.cend(); ++run)
120  {
121  even.push_back(*run);
122  ++run;
123  odd.push_back(*run);
124  }
125  inv_fast_fourier_transform(even);
126  inv_fast_fourier_transform(odd);
127  phase_iterator omega_iter(Tp{-1}, 1, len);
128  for (std::size_t i = 0, j = halflen; i < halflen;
129  ++i, ++j, ++omega_iter)
130  {
131  z[i] = even[i] + *omega_iter * odd[i];
132  // The next line works because omega^(length/2) = -1.
133  z[j] = even[i] - *omega_iter * odd[i];
134  }
135  }

138  /**
139  * Fast Fourier Transform on real data.
140  */
141  template<typename Tp>
142  void
143  fast_fourier_transform(std::vector<Tp>& x)
144  {
145  const auto len = x.size();
146  if (len % 2 == 1) // Too bad, we're odd.
147  throw std::domain_error("fast_fourier_transform: "
148  "Real data must have even length.");
149  else // Good, we're even.
150  {
151  // @todo I really want to just make a view to a complex vector.
152  const auto halflen = len / 2;
153  std::vector<std::complex<Tp>> z;
154  z.reserve(halflen + 1);
155  for (std::size_t i = 0; i < halflen; ++i)
156  z.emplace_back(x[2 * i], x[2 * i + 1]);
157  fast_fourier_transform(z);
158  z.emplace_back(z[0]); // Use symmetry. We need N/2 transform.
159  const auto s_i2 = std::complex<Tp>{0, 2};
160  phase_iterator omega_iter(Tp{+1}, 1, halflen);
161  for (std::size_t i = 0; i < halflen; ++i)
162  {
163  const auto z1 = z[i];
164  const auto z2 = std::conj(z[halflen - i]);
165  const auto f = (z1 + z2) / Tp{2}
166  + *omega_iter * (z1 - z2) / s_i2;
167  x[2 * i] = f.real();
168  x[2 * i + 1] = f.imag();
169  }
170  }
171  }

173  /**
174  * Inverse Fast Fourier Transform on real data.
175  */
176  template<typename Tp>
177  void
178  inv_fast_fourier_transform(std::vector<Tp>& x)
179  {
180  const auto len = x.size();
181  if (len % 2 == 1) // Too bad, we're odd.
182  throw std::domain_error("inv_fast_fourier_transform: "
183  "Real data must have even length.");
184  else // Good, we're even.
185  {
186  // @todo I really want to just make a view to a complex vector.
187  const auto halflen = len / 2;
188  std::vector<std::complex<Tp>> z;
189  z.reserve(halflen);
190  const auto s_i2 = std::complex<Tp>{0, 2};
191  phase_iterator omega_iter(Tp{-1}, 1, halflen);
192  for (std::size_t i = 0; i < halflen; ++i)
193  {
194  const auto z1 = std::complex<Tp>(x[2 * i], x[2 * i + 1]);
195  const auto z2 = std::complex<Tp>(x[len - 2 * i - 2], -x[len - 2 * i - 1]);
196  const auto ze = (z1 + z2) / Tp{2};
197  const auto zo = -*omega_iter * (z1 - z2) / s_i2;
198  z.emplace_back(ze + zo);
199  }
200  inv_fast_fourier_transform(z);
201  for (std::size_t i = 0; i < halflen; ++i)
202  {
203  x[2 * i] = z[i].real();
204  x[2 * i + 1] = z[i].imag();
205  }
206  }
207  }

209  /**
210  * Fast Sine Transform on real data.
211  */
212  template <typename Tp>
213  void
214  fast_sine_transform(std::vector<Tp>& x)
215  {
216  const auto len = x.size();
217  const auto halflen = len / 2;
218  const auto n2 = len - 1;
219  phase_iterator omega_iter(Tp{+1}, 1, len);
220  x[0] = Tp{0};
221  for (std::size_t k = 1; k <= halflen; ++k, ++omega_iter)
222  {
223  const auto y1 = omega_iter.sin()
224  * (x[k] + x[n2 - k]);
225  const auto y2 = (x[k] - x[n2 - k]) / Tp{2};
226  x[k] = y1 + y2;
227  x[n2 - k] = y1 - y2;
228  }
229  fast_fourier_transform(x);
230  x[0] *= Tp{0.5L};
231  x[1] = Tp{0};
232  auto sum = Tp{0};
233  for (std::size_t i = 2; i < halflen; i += 2)
234  {
235  sum += std::exchange(x[i], x[i + 1]);
236  x[i + 1] = sum;
237  }
238  }
