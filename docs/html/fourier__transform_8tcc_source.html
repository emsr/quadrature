31  /**
32  * Discrete Fourier transform on complex data.
33  */
34  template<typename _Tp>
35  void
36  __discrete_fourier_transform(bool __do_forward,
37  std::vector<std::complex<_Tp>>& __z)
38  {
39  const auto __sign(__do_forward ? _Tp{+1} : _Tp{-1});
40  const auto __len = __z.size();
41 
42  std::vector<std::complex<_Tp>> __result;
43  __result.reserve(__len);
44 
45  // Do matrix multiplication.
46  for (std::size_t __i = 0; __i < __len; ++__i)
47  {
48  __phase_iterator __coefficient_iter(__sign, __i, __len);
49  __result.push_back(std::inner_product(__z.begin(), __z.end(),
50  __coefficient_iter, std::complex<_Tp>{0}));
51  }
52 
53  // Rescale if forward.
54  if (__do_forward)
55  {
56  const auto __norm = _Tp{1} / _Tp(__len);
57  for (std::size_t __i = 0; __i < __len; ++__i)
58  __result[__i] *= __norm;
59  }
60 
61  // Copy data back (swap is fast!).
62  __z.swap(__result);
63  }

65  /**
66  * Fast Fourier Transform on complex data.
67  */
68  template<typename _Tp>
69  void
70  fast_fourier_transform(std::vector<std::complex<_Tp>>& __z)
71  {
72  const auto __len = __z.size();
73  if (__len % 2 == 1) // Too bad, we're odd.
74  __discrete_fourier_transform(true, __z);
75  else // Good, we're even.
76  {
77  // Let's divide and conquer!
78  std::vector<std::complex<_Tp>> __odd;
79  std::vector<std::complex<_Tp>> __even;
80  const auto __halflen = __len / 2;
81  __odd.reserve(__halflen);
82  __even.reserve(__halflen);
83  for (auto __run = __z.cbegin(); __run != __z.cend(); ++__run)
84  {
85  __even.push_back(*__run);
86  ++__run;
87  __odd.push_back(*__run);
88  }
89  fast_fourier_transform(__even);
90  fast_fourier_transform(__odd);
91  __phase_iterator __omega_iter(_Tp{1}, 1, __len);
92  for (std::size_t __i = 0, __j = __halflen; __i < __halflen;
93  ++__i, ++__j, ++__omega_iter)
94  {
95  __z[__i] = (__even[__i] + *__omega_iter * __odd[__i]) / _Tp{2};
96  // The next line works because omega^(length/2) = -1.
97  __z[__j] = (__even[__i] - *__omega_iter * __odd[__i]) / _Tp{2};
98  }
99  }
100  }

140  * Fast Fourier Transform on real data.
141  */
142  template<typename _Tp>
143  void
144  fast_fourier_transform(std::vector<_Tp>& __x)
145  {
146  const auto __len = __x.size();
147  if (__len % 2 == 1) // Too bad, we're odd.
148  std::__throw_domain_error("fast_fourier_transform: "
149  "Real data must have even length.");
150  else // Good, we're even.
151  {
152  // @todo I really want to just make a view to a complex vector.
153  const auto __halflen = __len / 2;
154  std::vector<std::complex<_Tp>> __z;
155  __z.reserve(__halflen + 1);
156  for (std::size_t __i = 0; __i < __halflen; ++__i)
157  __z.emplace_back(__x[2 * __i], __x[2 * __i + 1]);
158  fast_fourier_transform(__z);
159  __z.emplace_back(__z[0]); // Use symmetry. We need N/2 transform.
160  const auto _S_i2 = std::complex<_Tp>{0, 2};
161  __phase_iterator __omega_iter(_Tp{+1}, 1, __halflen);
162  for (std::size_t __i = 0; __i < __halflen; ++__i)
163  {
164  const auto __z1 = __z[__i];
165  const auto __z2 = std::conj(__z[__halflen - __i]);
166  const auto __f = (__z1 + __z2) / _Tp{2}
167  + *__omega_iter * (__z1 - __z2) / _S_i2;
168  __x[2 * __i] = __f.real();
169  __x[2 * __i + 1] = __f.imag();
170  }
171  }
172  }

211  * Fast Sine Transform on real data.
212  */
213  template <typename _Tp>
214  void
215  fast_sine_transform(std::vector<_Tp>& __x)
216  {
217  const auto __len = __x.size();
218  const auto __halflen = __len / 2;
219  const auto __n2 = __len - 1;
220  __phase_iterator __omega_iter(_Tp{+1}, 1, __len);
221  __x[0] = _Tp{0};
222  for (std::size_t __k = 1; __k <= __halflen; ++__k, ++__omega_iter)
223  {
224  const auto __y1 = __omega_iter.sin()
225  * (__x[__k] + __x[__n2 - __k]);
226  const auto __y2 = (__x[__k] - __x[__n2 - __k]) / _Tp{2};
227  __x[__k] = __y1 + __y2;
228  __x[__n2 - __k] = __y1 - __y2;
229  }
230  fast_fourier_transform(__x);
231  __x[0] *= _Tp{0.5L};
232  __x[1] = _Tp{0};
233  auto __sum = _Tp{0};
234  for (std::size_t __i = 2; __i < __halflen; __i += 2)
235  {
236  __sum += std::exchange(__x[__i], __x[__i + 1]);
237  __x[__i + 1] = __sum;
238  }
239  }
