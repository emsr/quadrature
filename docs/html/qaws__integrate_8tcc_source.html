62  /**
63  * The singular weight function is defined by:
64  * @f[
65  * W(x) = (x-a)^\alpha (b-x)^\beta log^\mu (x-a) log^\nu (b-x)
66  * @f]
67  * where @f$ \alpha > -1 @f$, @f$ \beta > -1 @f$,
68  * and @f$ \mu = 0 @f$, 1, @f$ \nu = 0, 1 @f$.
69  *
70  * The weight function can take four different forms depending
71  * on the values of \mu and \nu,
72  * @f[
73  * W(x) = (x-a)^\alpha (b-x)^\beta (\mu = 0, \nu = 0)
74  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) (\mu = 1, \nu = 0)
75  * W(x) = (x-a)^\alpha (b-x)^\beta log(b-x) (\mu = 0, \nu = 1)
76  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) log(b-x) (\mu = 1, \nu = 1)
77  * @f]
78  *
79  * The QAWS algorithm is designed for integrands with algebraic-logarithmic
80  * singularities at the end-points of an integration region.
81  *
82  * In order to work efficiently the algorithm requires a precomputed table
83  * of Chebyshev moments.
84  */
85  template<typename Tp, typename FuncTp,
86  typename Integrator = gauss_kronrod_integral<Tp>>
87  auto
88  qaws_integrate(integration_workspace<Tp,
89  std::invoke_result_t<FuncTp, Tp>>& workspace,
90  qaws_integration_table<Tp>& table,
91  FuncTp func,
92  Tp lower, Tp upper,
93  Tp max_abs_err, Tp max_rel_err,
94  Integrator quad = gauss_kronrod_integral<Tp>(Kronrod_15))
95  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
96  {
97  // Try to adjust tests for varing precision.
98  const auto m_rel_err = std::pow(Tp{10},
99  -std::numeric_limits<Tp>::digits / Tp{10});
100 
101  if (upper <= lower)
102  throw std::runtime_error("qaws_integrate: "
103  "Limits must form an ascending sequence");
104  if (!valid_tolerances(max_abs_err, max_rel_err))
105  {
106  std::ostringstream msg;
107  msg << "qaws_integrate: Tolerance cannot be achieved with given "
108  "absolute (" << max_abs_err << ") and relative ("
109  << max_rel_err << ") error limits.";
110  throw std::runtime_error(msg.str().c_str());
111  }
112 
113  const auto limit = workspace.capacity();
114  workspace.clear();
115 
116  // Perform the first integration.
117  Tp result0, abserr0;
118  {
119  const auto a1 = lower;
120  const auto mid = (lower + upper) / Tp{2};
121  const auto b2 = upper;
122 
123  auto [area1, error1, err_reliable1]
124  = qc25s(table, func, lower, upper, a1, mid, quad);
125  workspace.append(a1, mid, area1, error1);
126 
127  auto [area2, error2, err_reliable2]
128  = qc25s(table, func, lower, upper, mid, b2, quad);
129  workspace.append(mid, b2, area2, error2);
130 
131  result0 = area1 + area2;
132  abserr0 = error1 + error2;
133  }
134 
135  // Test on accuracy; Use 0.01 relative error as an extra safety
136  // margin on the first iteration (ignored for subsequent iterations).
137  auto tolerance = std::max(max_abs_err, max_rel_err * std::abs(result0));
138  if (abserr0 < tolerance && abserr0
139  < Tp{0.01} * std::abs(result0))
140  return {result0, abserr0};
141  else if (limit == 1)
142  throw integration_error("qaws_integrate: "
143  "A maximum of one iteration was insufficient",
144  MAX_ITER_ERROR, result0, abserr0);
145 
146  auto area = result0;
147  auto errsum = abserr0;
148  auto iteration = 2u;
149  int error_type = NO_ERROR;
150  int roundoff_type1 = 0, roundoff_type2 = 0;
151  do
152  {
153  // Bisect the subinterval with the largest error estimate.
154  const auto& curr = workspace.retrieve();
155 
156  const auto a1 = curr.lower_lim;
157  const auto mid = (curr.lower_lim + curr.upper_lim) / Tp{2};
158  const auto b2 = curr.upper_lim;
159 
160  auto [area1, error1, err_reliable1]
161  = qc25s(table, func, lower, upper, a1, mid, quad);
162 
163  auto [area2, error2, err_reliable2]
164  = qc25s(table, func, lower, upper, mid, b2, quad);
165 
166  const auto area12 = area1 + area2;
167  const auto error12 = error1 + error2;
168 
169  errsum += error12 - curr.abs_error;
170  area += area12 - curr.result;
171 
172  if (err_reliable1 && err_reliable2)
173  {
174  const auto delta = curr.result - area12;
175 
176  if (std::abs (delta) <= m_rel_err * std::abs(area12)
177  && error12 >= 0.99 * curr.abs_error)
178  ++roundoff_type1;
179  if (iteration >= 10 && error12 > curr.abs_error)
180  ++roundoff_type2;
181  }
182 
183  tolerance = std::max(max_abs_err, max_rel_err * std::abs(area));
184  if (errsum > tolerance)
185  {
186  if (roundoff_type1 >= 6 || roundoff_type2 >= 20)
187  error_type = ROUNDOFF_ERROR;
188 
189  // set error flag in the case of bad integrand behaviour at
190  // a point of the integration range
191  if (workspace.subinterval_too_small(a1, mid, b2))
192  error_type = SINGULAR_ERROR;
193  }
194 
195  workspace.split(mid, area1, error1, area2, error2);
196 
197  ++iteration;
198  }
199  while (iteration < limit && !error_type && errsum > tolerance);
200 
201  const auto result = workspace.total_integral();
202  const auto abserr = errsum;
203 
204  if (iteration == limit)
205  error_type = MAX_SUBDIV_ERROR;
206 
207  if (errsum <= tolerance)
208  return {result, abserr};
209 
210  if (error_type == NO_ERROR)
211  return {result, abserr};
212 
213  check_error(__func__, error_type, result, abserr);
214  throw integration_error("qaws_integrate: Unknown error.",
215  UNKNOWN_ERROR, result, abserr);
216  }

221  template<typename Tp, typename FuncTp,
222  typename Integrator = gauss_kronrod_integral<Tp>>
223  std::tuple<Tp, Tp, bool>
224  qc25s(qaws_integration_table<Tp>& t,
225  FuncTp func, Tp lower, Tp upper, Tp a1, Tp b1,
226  Integrator quad)
227  {
228  fn_qaws<Tp, FuncTp> fqaws(&t, func, lower, upper);
229 
230  if (a1 == lower && (t.alpha != Tp{0} || t.mu != 0))
231  {
232  const auto factor
233  = std::pow(0.5 * (b1 - a1), t.alpha + Tp{1});
234 
235  auto f = [fqaws](Tp x)
236  -> Tp { return fqaws.eval_right(x); };
237  auto chout = qcheb_integrate(f, a1, b1);
238  const auto& cheb12 = chout.cheb12;
239  const auto& cheb24 = chout.cheb24;
240 
241  if (t.mu == 0)
242  {
243  const auto u = factor;
244 
245  auto [res12, res24]
246  = compute_result(t.ri, cheb12, cheb24);
247 
248  const auto result = u * res24;
249  const auto abserr = std::abs(u * (res24 - res12));
250  return std::make_tuple(result, abserr, false);
251  }
252  else
253  {
254  const auto u = factor * std::log(b1 - a1);
255  const auto v = factor;
256 
257  auto [res12a, res24a]
258  = compute_result(t.ri, cheb12, cheb24);
259  auto [res12b, res24b]
260  = compute_result(t.rg, cheb12, cheb24);
261 
262  const auto result = u * res24a + v * res24b;
263  const auto abserr = std::abs(u * (res24a - res12a))
264  + std::abs(v * (res24b - res12b));
265  return std::make_tuple(result, abserr, false);
266  }
267  }
268  else if (b1 == upper && (t.beta != Tp{0} || t.nu != 0))
269  {
270  auto factor = std::pow(0.5 * (b1 - a1), t.beta + Tp{1});
271 
272  auto f = [fqaws](Tp x)
273  -> Tp { return fqaws.eval_left(x); };
274  auto chout = qcheb_integrate(f, a1, b1);
275  const auto& cheb12 = chout.cheb12;
276  const auto& cheb24 = chout.cheb24;
277 
278  if (t.nu == 0)
279  {
280  const auto u = factor;
281 
282  auto [res12, res24]
283  = compute_result(t.rj, cheb12, cheb24);
284 
285  const auto result = u * res24;
286  const auto abserr = std::abs(u * (res24 - res12));
287  return std::make_tuple(result, abserr, false);
288  }
289  else
290  {
291  const auto u = factor * std::log(b1 - a1);
292  const auto v = factor;
293 
294  auto [res12a, res24a]
295  = compute_result(t.rj, cheb12, cheb24);
296  auto [res12b, res24b]
297  = compute_result(t.rh, cheb12, cheb24);
298 
299  const auto result = u * res24a + v * res24b;
300  const auto abserr = std::abs(u * (res24a - res12a))
301  + std::abs(v * (res24b - res12b));
302  return std::make_tuple(result, abserr, false);
303  }
304  }
305  else
306  {
307  auto f = [fqaws](Tp x)
308  ->Tp
309  { return fqaws.eval_middle(x); };
310 
311  auto [result, abserr, resabs, resasc]
312  = quad(f, a1, b1);
313 
314  bool err_reliable;
315  if (abserr == resasc)
316  err_reliable = false;
317  else
318  err_reliable = true;
319 
320  return std::make_tuple(result, abserr, err_reliable);
321  }
322  }

349  template<typename Tp, typename FuncTp>
350  std::invoke_result_t<FuncTp, Tp>
351  fn_qaws<Tp, FuncTp>::eval_middle(Tp x) const
352  {
353  auto factor = Tp{1};
354 
355  if (this->table->alpha != Tp{0})
356  factor *= std::pow(x - this->a, this->table->alpha);
357 
358  if (table->mu == 1)
359  factor *= std::log(x - this->a);
360 
361  if (this->table->beta != Tp{0})
362  factor *= std::pow(this->b - x, this->table->beta);
363 
364  if (table->nu == 1)
365  factor *= std::log(this->b - x);
366 
367  return factor * this->func(x);
368  }

389  auto factor = Tp{1};

392  factor *= std::pow(this->b - x, this->table->beta);

417  return compute_result_t<AreaTp>{res12, res24};
