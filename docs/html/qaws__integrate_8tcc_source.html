62  /**
63  * The singular weight function is defined by:
64  * @f[
65  * W(x) = (x-a)^\alpha (b-x)^\beta log^\mu (x-a) log^\nu (b-x)
66  * @f]
67  * where @f$ \alpha > -1 @f$, @f$ \beta > -1 @f$,
68  * and @f$ \mu = 0 @f$, 1, @f$ \nu = 0, 1 @f$.
69  *
70  * The weight function can take four different forms depending
71  * on the values of \mu and \nu,
72  * @f[
73  * W(x) = (x-a)^\alpha (b-x)^\beta (\mu = 0, \nu = 0)
74  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) (\mu = 1, \nu = 0)
75  * W(x) = (x-a)^\alpha (b-x)^\beta log(b-x) (\mu = 0, \nu = 1)
76  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) log(b-x) (\mu = 1, \nu = 1)
77  * @f]
78  *
79  * The QAWS algorithm is designed for integrands with algebraic-logarithmic
80  * singularities at the end-points of an integration region.
81  *
82  * In order to work efficiently the algorithm requires a precomputed table
83  * of Chebyshev moments.
84  */
85  template<typename _Tp, typename _FuncTp,
86  typename _Integrator = gauss_kronrod_integral<_Tp>>
87  auto
88  qaws_integrate(integration_workspace<_Tp,
89  std::invoke_result_t<_FuncTp, _Tp>>& __workspace,
90  qaws_integration_table<_Tp>& __table,
91  _FuncTp __func,
92  _Tp __lower, _Tp __upper,
93  _Tp __max_abs_err, _Tp __max_rel_err,
94  _Integrator __quad = gauss_kronrod_integral<_Tp>(Kronrod_15))
95  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
96  {
97  // Try to adjust tests for varing precision.
98  const auto _M_rel_err = std::pow(_Tp{10},
99  -std::numeric_limits<_Tp>::digits / _Tp{10});
100 
101  if (__upper <= __lower)
102  std::__throw_runtime_error("qaws_integrate: "
103  "Limits must form an ascending sequence");
104  if (!valid_tolerances(__max_abs_err, __max_rel_err))
105  {
106  std::ostringstream __msg;
107  __msg << "qaws_integrate: Tolerance cannot be achieved with given "
108  "absolute (" << __max_abs_err << ") and relative ("
109  << __max_rel_err << ") error limits.";
110  std::__throw_runtime_error(__msg.str().c_str());
111  }
112 
113  const auto __limit = __workspace.capacity();
114  __workspace.clear();
115 
116  // Perform the first integration.
117  _Tp __result0, __abserr0;
118  {
119  const auto __a1 = __lower;
120  const auto __mid = (__lower + __upper) / _Tp{2};
121  const auto __b2 = __upper;
122 
123  auto [__area1, __error1, __err_reliable1]
124  = qc25s(__table, __func, __lower, __upper, __a1, __mid, __quad);
125  __workspace.append(__a1, __mid, __area1, __error1);
126 
127  auto [__area2, __error2, __err_reliable2]
128  = qc25s(__table, __func, __lower, __upper, __mid, __b2, __quad);
129  __workspace.append(__mid, __b2, __area2, __error2);
130 
131  __result0 = __area1 + __area2;
132  __abserr0 = __error1 + __error2;
133  }
134 
135  // Test on accuracy; Use 0.01 relative error as an extra safety
136  // margin on the first iteration (ignored for subsequent iterations).
137  auto __tolerance = std::max(__max_abs_err, __max_rel_err * std::abs(__result0));
138  if (__abserr0 < __tolerance && __abserr0
139  < _Tp{0.01} * std::abs(__result0))
140  return {__result0, __abserr0};
141  else if (__limit == 1)
142  __throw_integration_error("qaws_integrate: "
143  "A maximum of one iteration was insufficient",
144  MAX_ITER_ERROR, __result0, __abserr0);
145 
146  auto __area = __result0;
147  auto __errsum = __abserr0;
148  auto __iteration = 2u;
149  int __error_type = NO_ERROR;
150  int __roundoff_type1 = 0, __roundoff_type2 = 0;
151  do
152  {
153  // Bisect the subinterval with the largest error estimate.
154  const auto& __curr = __workspace.retrieve();
155 
156  const auto __a1 = __curr.__lower_lim;
157  const auto __mid = (__curr.__lower_lim + __curr.__upper_lim) / _Tp{2};
158  const auto __b2 = __curr.__upper_lim;
159 
160  auto [__area1, __error1, __err_reliable1]
161  = qc25s(__table, __func, __lower, __upper, __a1, __mid, __quad);
162 
163  auto [__area2, __error2, __err_reliable2]
164  = qc25s(__table, __func, __lower, __upper, __mid, __b2, __quad);
165 
166  const auto __area12 = __area1 + __area2;
167  const auto __error12 = __error1 + __error2;
168 
169  __errsum += __error12 - __curr.__abs_error;
170  __area += __area12 - __curr.__result;
171 
172  if (__err_reliable1 && __err_reliable2)
173  {
174  const auto __delta = __curr.__result - __area12;
175 
176  if (std::abs (__delta) <= _M_rel_err * std::abs(__area12)
177  && __error12 >= 0.99 * __curr.__abs_error)
178  ++__roundoff_type1;
179  if (__iteration >= 10 && __error12 > __curr.__abs_error)
180  ++__roundoff_type2;
181  }
182 
183  __tolerance = std::max(__max_abs_err, __max_rel_err * std::abs(__area));
184  if (__errsum > __tolerance)
185  {
186  if (__roundoff_type1 >= 6 || __roundoff_type2 >= 20)
187  __error_type = ROUNDOFF_ERROR;
188 
189  // set error flag in the case of bad integrand behaviour at
190  // a point of the integration range
191  if (__workspace.subinterval_too_small(__a1, __mid, __b2))
192  __error_type = SINGULAR_ERROR;
193  }
194 
195  __workspace.split(__mid, __area1, __error1, __area2, __error2);
196 
197  ++__iteration;
198  }
199  while (__iteration < __limit && !__error_type && __errsum > __tolerance);
200 
201  const auto __result = __workspace.total_integral();
202  const auto __abserr = __errsum;
203 
204  if (__iteration == __limit)
205  __error_type = MAX_SUBDIV_ERROR;
206 
207  if (__errsum <= __tolerance)
208  return {__result, __abserr};
209 
210  if (__error_type == NO_ERROR)
211  return {__result, __abserr};
212 
213  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
214  __throw_integration_error("qaws_integrate: Unknown error.",
215  UNKNOWN_ERROR, __result, __abserr);
216  }

350  std::invoke_result_t<_FuncTp, _Tp>
351  fn_qaws<_Tp, _FuncTp>::eval_middle(_Tp __x) const
352  {
353  auto __factor = _Tp{1};
354 
355  if (this->table->alpha != _Tp{0})
356  __factor *= std::pow(__x - this->a, this->table->alpha);
357 
358  if (table->mu == 1)
359  __factor *= std::log(__x - this->a);
360 
361  if (this->table->beta != _Tp{0})
362  __factor *= std::pow(this->b - __x, this->table->beta);
363 
364  if (table->nu == 1)
365  __factor *= std::log(this->b - __x);
366 
367  return __factor * this->func(__x);
368  }
