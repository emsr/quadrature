38  /**
39  * This function attempts to compute a Fourier integral of the function f
40  * over the semi-infinite interval [a,+\infty)
41  */
42  template<typename Tp, typename FuncTp>
43  auto
44  qawf_integrate(integration_workspace<Tp,
45  std::invoke_result_t<FuncTp, Tp>>& workspace,
46  integration_workspace<Tp,
47  std::invoke_result_t<FuncTp, Tp>>& cycle_workspace,
48  oscillatory_integration_table<Tp>& wf,
49  FuncTp func,
50  Tp lower, Tp max_abs_err)
51  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
52  {
53  using AreaTp = std::invoke_result_t<FuncTp, Tp>;
54  using AbsAreaTp = decltype(std::abs(AreaTp{}));
55 
56  std::size_t ktmin = 0;
57  std::size_t iteration = 0;
58 
59  extrapolation_table<AreaTp, AbsAreaTp> table;
60 
61  Tp cycle;
62  auto omega = wf.omega;
63 
64  const Tp p = 0.9;
65  Tp factor = 1;
66  Tp initial_eps, eps;
67  int error_type = NO_ERROR;
68 
69  int status = 0;
70  auto result = Tp{0};
71  auto abserr = Tp{0};
72 
73  const auto s_max = std::numeric_limits<Tp>::max();
74  const auto limit = workspace.capacity();
75 
76  workspace.clear();
77  cycle_workspace.clear();
78  //wf.clear();
79 
80  // Test on accuracy.
81  if (max_abs_err <= Tp{0})
82  throw std::domain_error("absolute tolerance epsabs must be positive") ;
83 
84  if (omega == Tp{0})
85  {
86  if (wf.circfun == oscillatory_integration_table<Tp>::INTEG_SINE)
87  // The function sin(w x) f(x) is always zero for w = 0.
88  return {Tp{0}, Tp{0}};
89  else
90  // The function cos(w x) f(x) is always f(x) for w = 0.
91  return qagiu_integrate(cycle_workspace, func, lower, max_abs_err,
92  Tp{0});
93  }
94 
95  if (max_abs_err * (Tp{1} - p) > std::numeric_limits<Tp>::min())
96  eps = max_abs_err * (Tp{1} - p);
97  else
98  eps = max_abs_err;
99 
100  initial_eps = eps;
101 
102  auto area = Tp{0};
103  auto errsum = Tp{0};
104  auto res_ext = Tp{0};
105  auto err_ext = s_max;
106  auto correc = Tp{0};
107  auto total_error = Tp{0};
108  auto truncation_error = Tp{0};
109 
110  cycle = (2 * std::floor(std::abs(omega)) + 1)
111  * M_PI / std::abs(omega);
112 
113  wf.set_length(cycle);
114 
115  for (iteration = 0; iteration < limit; ++iteration)
116  {
117  const auto a1 = lower + iteration * cycle;
118  const auto b1 = a1 + cycle;
119 
120  const auto max_abs_err1 = eps * factor;
121 
122  auto out1 = qawo_integrate(cycle_workspace, wf, func, a1,
123  max_abs_err1, Tp{0});
124  auto area1 = out1.result;
125  auto error1 = out1.abserr;
126 
127  workspace.append(a1, b1, area1, error1);
128 
129  factor *= p;
130 
131  area += area1;
132  errsum += error1;
133 
134  // Estimate the truncation error as 50 times the final term.
135  truncation_error = 50 * std::abs(area1);
136  total_error = errsum + truncation_error;
137  if (total_error < max_abs_err && iteration > 4)
138  goto compute_result;
139 
140  if (error1 > correc)
141  correc = error1;
142 
143  if (status)
144  eps = std::max(initial_eps, correc * (Tp{1} - p));
145 
146  if (status && total_error < 10 * correc && iteration > 3)
147  goto compute_result;
148 
149  table.append(area);
150 
151  if (table.get_nn() < 2)
152  continue;
153 
154  Tp reseps, erreps;
155  std::tie(reseps, erreps) = table.qelg();
156 
157  ++ktmin;
158  if (ktmin >= 15 && err_ext < Tp{0.001L} * total_error)
159  error_type = EXTRAP_ROUNDOFF_ERROR;
160 
161  if (erreps < err_ext)
162  {
163  ktmin = 0;
164  err_ext = erreps;
165  res_ext = reseps;
166 
167  if (err_ext + 10 * correc <= max_abs_err)
168  break;
169  if (err_ext <= max_abs_err && 10 * correc >= max_abs_err)
170  break;
171  }
172  }
173 
174  if (iteration == limit)
175  error_type = MAX_ITER_ERROR;
176 
177  if (err_ext == s_max)
178  goto compute_result;
179 
180  err_ext += 10 * correc;
181 
182  result = res_ext;
183  abserr = err_ext;
184 
185  if (error_type == NO_ERROR)
186  return {result, abserr};
187 
188  if (res_ext != Tp{0} && area != Tp{0})
189  {
190  if (err_ext / std::abs(res_ext) > errsum / std::abs(area))
191  goto compute_result;
192  }
193  else if (err_ext > errsum)
194  goto compute_result;
195  else if (area == Tp{0})
196  goto return_error;
197 
198  if (error_type == EXTRAP_ROUNDOFF_ERROR)
199  err_ext += truncation_error;
200 
201  goto return_error;
202 
203  compute_result:
204 
205  result = area;
206  abserr = total_error;
207 
208  if (error_type == NO_ERROR)
209  return {result, abserr};
210 
211  return_error:
212 
213  check_error(__func__, error_type, result, abserr);
214  throw integration_error("qawf_integrate: Unknown error.", UNKNOWN_ERROR,
215  result, abserr);
216  }
