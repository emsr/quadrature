38  /**
39  * This function attempts to compute a Fourier integral of the function f
40  * over the semi-infinite interval [a,+\infty)
41  */
42  template<typename _Tp, typename _FuncTp>
43  auto
44  qawf_integrate(integration_workspace<_Tp,
45  std::invoke_result_t<_FuncTp, _Tp>>& __workspace,
46  integration_workspace<_Tp,
47  std::invoke_result_t<_FuncTp, _Tp>>& __cycle_workspace,
48  oscillatory_integration_table<_Tp>& __wf,
49  _FuncTp __func,
50  _Tp __lower, _Tp __max_abs_err)
51  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
52  {
53  std::size_t __ktmin = 0;
54  std::size_t __iteration = 0;
55 
56  extrapolation_table<_Tp> __table;
57 
58  _Tp __cycle;
59  auto __omega = __wf.omega;
60 
61  const _Tp __p = 0.9;
62  _Tp __factor = 1;
63  _Tp __initial_eps, __eps;
64  int __error_type = NO_ERROR;
65 
66  int __status = 0;
67  auto __result = _Tp{0};
68  auto __abserr = _Tp{0};
69 
70  const auto _S_max = std::numeric_limits<_Tp>::max();
71  const auto __limit = __workspace.capacity();
72 
73  __workspace.clear();
74  __cycle_workspace.clear();
75  //__wf.clear();
76 
77  // Test on accuracy.
78  if (__max_abs_err <= _Tp{0})
79  std::__throw_domain_error("absolute tolerance epsabs must be positive") ;
80 
81  if (__omega == _Tp{0})
82  {
83  if (__wf.circfun == oscillatory_integration_table<_Tp>::INTEG_SINE)
84  // The function sin(w x) f(x) is always zero for w = 0.
85  return {_Tp{0}, _Tp{0}};
86  else
87  // The function cos(w x) f(x) is always f(x) for w = 0.
88  return qagiu_integrate(__cycle_workspace, __func, __lower, __max_abs_err,
89  _Tp{0});
90  }
91 
92  if (__max_abs_err * (_Tp{1} - __p) > std::numeric_limits<_Tp>::min())
93  __eps = __max_abs_err * (_Tp{1} - __p);
94  else
95  __eps = __max_abs_err;
96 
97  __initial_eps = __eps;
98 
99  auto __area = _Tp{0};
100  auto __errsum = _Tp{0};
101  auto __res_ext = _Tp{0};
102  auto __err_ext = _S_max;
103  auto __correc = _Tp{0};
104  auto __total_error = _Tp{0};
105  auto __truncation_error = _Tp{0};
106 
107  __cycle = (2 * std::floor(std::abs(__omega)) + 1)
108  * M_PI / std::abs(__omega);
109 
110  __wf.set_length(__cycle);
111 
112  for (__iteration = 0; __iteration < __limit; ++__iteration)
113  {
114  const auto __a1 = __lower + __iteration * __cycle;
115  const auto __b1 = __a1 + __cycle;
116 
117  const auto __max_abs_err1 = __eps * __factor;
118 
119  auto __out1 = qawo_integrate(__cycle_workspace, __wf, __func, __a1,
120  __max_abs_err1, _Tp{0});
121  auto __area1 = __out1.__result;
122  auto __error1 = __out1.__abserr;
123 
124  __workspace.append(__a1, __b1, __area1, __error1);
125 
126  __factor *= __p;
127 
128  __area += __area1;
129  __errsum += __error1;
130 
131  // Estimate the truncation error as 50 times the final term.
132  __truncation_error = 50 * std::abs(__area1);
133  __total_error = __errsum + __truncation_error;
134  if (__total_error < __max_abs_err && __iteration > 4)
135  goto compute_result;
136 
137  if (__error1 > __correc)
138  __correc = __error1;
139 
140  if (__status)
141  __eps = std::max(__initial_eps, __correc * (_Tp{1} - __p));
142 
143  if (__status && __total_error < 10 * __correc && __iteration > 3)
144  goto compute_result;
145 
146  __table.append(__area);
147 
148  if (__table.get_nn() < 2)
149  continue;
150 
151  _Tp __reseps, __erreps;
152  std::tie(__reseps, __erreps) = __table.qelg();
153 
154  ++__ktmin;
155  if (__ktmin >= 15 && __err_ext < _Tp{0.001L} * __total_error)
156  __error_type = EXTRAP_ROUNDOFF_ERROR;
157 
158  if (__erreps < __err_ext)
159  {
160  __ktmin = 0;
161  __err_ext = __erreps;
162  __res_ext = __reseps;
163 
164  if (__err_ext + 10 * __correc <= __max_abs_err)
165  break;
166  if (__err_ext <= __max_abs_err && 10 * __correc >= __max_abs_err)
167  break;
168  }
169  }
170 
171  if (__iteration == __limit)
172  __error_type = MAX_ITER_ERROR;
173 
174  if (__err_ext == _S_max)
175  goto compute_result;
176 
177  __err_ext += 10 * __correc;
178 
179  __result = __res_ext;
180  __abserr = __err_ext;
181 
182  if (__error_type == NO_ERROR)
183  return {__result, __abserr};
184 
185  if (__res_ext != _Tp{0} && __area != _Tp{0})
186  {
187  if (__err_ext / std::abs(__res_ext) > __errsum / std::abs(__area))
188  goto compute_result;
189  }
190  else if (__err_ext > __errsum)
191  goto compute_result;
192  else if (__area == _Tp{0})
193  goto return_error;
194 
195  if (__error_type == EXTRAP_ROUNDOFF_ERROR)
196  __err_ext += __truncation_error;
197 
198  goto return_error;
199 
200  compute_result:
201 
202  __result = __area;
203  __abserr = __total_error;
204 
205  if (__error_type == NO_ERROR)
206  return {__result, __abserr};
207 
208  return_error:
209 
210  __check_error(__func__, __error_type, __result, __abserr);
211  __throw_integration_error("qawf_integrate: Unknown error.", UNKNOWN_ERROR,
212  __result, __abserr);
213  }
