228  template<typename Tp, typename FuncTp>
229  auto
230  qng_integrate(FuncTp func,
231  Tp lower, Tp upper,
232  Tp max_abs_err, Tp max_rel_err)
233  -> gauss_kronrod_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
234  {
235  using RetTp = std::invoke_result_t<FuncTp, Tp>;
236  using AreaTp = decltype(RetTp{} * Tp{});
237 
238  RetTp fv1[5], fv2[5], fv3[5], fv4[5];
239  RetTp savfun[21];
240 
241  const auto half_length = (upper - lower) / Tp{2};
242  const auto abs_half_length = std::abs(half_length);
243  const auto center = (upper + lower) / Tp{2};
244  const auto f_center = func(center);
245 
246  if (!valid_tolerances(max_abs_err, max_rel_err))
247  {
248  std::ostringstream msg;
249  msg << "qng_integrate: Tolerance cannot be achieved with given "
250  "absolute (" << max_abs_err << ") and relative ("
251  << max_rel_err << ") error limits.";
252  throw std::runtime_error(msg.str().c_str());
253  }
254 
255  // Compute the integral using the 10- and 21-point formula.
256 
257  auto res10 = AreaTp{0};
258  auto res21 = Tp(qng_w21b[5]) * f_center;
259  // Approximation to the integral of abs(f)
260  auto resabs = Tp(qng_w21b[5]) * std::abs(f_center);
261 
262  for (int k = 0; k < 5; ++k)
263  {
264  const auto abscissa = half_length * Tp(qng_x1[k]);
265  const auto fval1 = func(center + abscissa);
266  const auto fval2 = func(center - abscissa);
267  const auto fval = fval1 + fval2;
268  res10 += Tp(qng_w10[k]) * fval;
269  res21 += Tp(qng_w21a[k]) * fval;
270  resabs += Tp(qng_w21a[k])
271  * (std::abs(fval1) + std::abs(fval2));
272  savfun[k] = fval;
273  fv1[k] = fval1;
274  fv2[k] = fval2;
275  }
276 
277  for (int k = 0; k < 5; ++k)
278  {
279  const auto abscissa = half_length * Tp(qng_x2[k]);
280  const auto fval1 = func(center + abscissa);
281  const auto fval2 = func(center - abscissa);
282  const auto fval = fval1 + fval2;
283  res21 += Tp(qng_w21b[k]) * fval;
284  resabs += Tp(qng_w21b[k])
285  * (std::abs(fval1) + std::abs(fval2));
286  savfun[k + 5] = fval;
287  fv3[k] = fval1;
288  fv4[k] = fval2;
289  }
290 
291  resabs *= abs_half_length;
292 
293  // Approximation to the integral of abs(f-I/(b-a)).
294  const auto mean = Tp{0.5L} * res21;
295  auto resasc = Tp(qng_w21b[5]) * std::abs(f_center - mean);
296  for (int k = 0; k < 5; ++k)
297  {
298  resasc +=
299  (Tp(qng_w21a[k]) * (std::abs(fv1[k] - mean)
300  + std::abs(fv2[k] - mean))
301  + Tp(qng_w21b[k]) * (std::abs(fv3[k] - mean)
302  + std::abs(fv4[k] - mean)));
303  }
304  resasc *= abs_half_length;
305 
306  // Test for convergence.
307  auto result_kronrod = res21 * half_length;
308  auto err = rescale_error((res21 - res10) * half_length,
309  resabs, resasc);
310  if (err < max_abs_err
311  || err < max_rel_err * std::abs(result_kronrod))
312  return {result_kronrod, err, resabs, resasc};
313 
314  // Compute the integral using the 43-point formula.
315  auto res43 = Tp(qng_w43b[11]) * f_center;
316  for (int k = 0; k < 10; ++k)
317  res43 += savfun[k] * Tp(qng_w43a[k]);
318  for (int k = 0; k < 11; ++k)
319  {
320  const auto abscissa = half_length * Tp(qng_x3[k]);
321  const auto fval = func(center + abscissa)
322  + func(center - abscissa);
323  res43 += fval * Tp(qng_w43b[k]);
324  savfun[k + 10] = fval;
325  }
326 
327  // Test for convergence.
328  result_kronrod = res43 * half_length;
329  err = rescale_error((res43 - res21) * half_length,
330  resabs, resasc);
331  if (err < max_abs_err
332  || err < max_rel_err * std::abs(result_kronrod))
333  return {result_kronrod, err, resabs, resasc};
334 
335  // Compute the integral using the 87-point formula.
336  auto res87 = Tp(qng_w87b[22]) * f_center;
337  for (int k = 0; k < 21; ++k)
338  res87 += savfun[k] * Tp(qng_w87a[k]);
339  for (int k = 0; k < 22; ++k)
340  {
341  const auto abscissa = half_length * Tp(qng_x4[k]);
342  res87 += Tp(qng_w87b[k]) * (func(center + abscissa)
343  + func(center - abscissa));
344  }
345 
346  // Test for convergence.
347  result_kronrod = res87 * half_length;
348  err = rescale_error((res87 - res43) * half_length,
349  resabs, resasc);
350  if (err < max_abs_err
351  || err < max_rel_err * std::abs(result_kronrod))
352  return {result_kronrod, err, resabs, resasc};
353 
354  // Failed to converge.
355  throw integration_error("qng_integrate: "
356  "Failed to reach tolerance with highest-order rule",
357  TOLERANCE_ERROR, result_kronrod, err);
358  }
