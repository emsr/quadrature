35  /**
36  * Compute the product of the fx with one of the inverse
37  * Vandermonde-like matrices.
38  */
39  template<typename _RetTp>
40  void
41  _Vinvfx(const std::array<_RetTp, 33>& __fx,
42  _RetTp* __coeff, const int __depth)
43  {
44  switch (__depth)
45  {
46  case 0:
47  for (int __i = 0; __i <= 4; ++__i)
48  {
49  __coeff[__i] = _RetTp{0};
50  for (int __j = 0; __j <= 4; ++__j)
51  __coeff[__i] += _RetTp(V1inv[__i * 5 + __j]) * __fx[__j * 8];
52  }
53  break;
54 
55  case 1:
56  for (int __i = 0; __i <= 8; ++__i)
57  {
58  __coeff[__i] = _RetTp{0};
59  for (int __j = 0; __j <= 8; ++__j)
60  __coeff[__i] += _RetTp(V2inv[__i * 9 + __j]) * __fx[__j * 4];
61  }
62  break;
63 
64  case 2:
65  for (int __i = 0; __i <= 16; ++__i)
66  {
67  __coeff[__i] = _RetTp{0};
68  for (int __j = 0; __j <= 16; ++__j)
69  __coeff[__i] += _RetTp(V3inv[__i * 17 + __j]) * __fx[__j * 2];
70  }
71  break;
72 
73  case 3:
74  for (int __i = 0; __i <= 32; ++__i)
75  {
76  __coeff[__i] = _RetTp{0};
77  for (int __j = 0; __j <= 32; ++__j)
78  __coeff[__i] += _RetTp(V4inv[__i * 33 + __j]) * __fx[__j];
79  }
80  break;
81  }
82  }

84  /**
85  * Downdate the interpolation given by the n coefficients c
86  * by removing the nodes with indices in NaNs.
87  */
88  template<typename _Tp>
89  void
90  downdate(_Tp* __coeff, std::ptrdiff_t __n, std::ptrdiff_t __depth,
91  std::ptrdiff_t* __NaN, std::ptrdiff_t __num_NaNs)
92  {
93  constexpr std::ptrdiff_t __bidx[4] = { 0, 6, 16, 34 };
94  _Tp __b_new[34], __alpha;
95 
96  for (std::ptrdiff_t __i = 0; __i <= __n + 1; ++__i)
97  __b_new[__i] = bee[__bidx[__depth] + __i];
98  for (std::ptrdiff_t __i = 0; __i < __num_NaNs; ++__i)
99  {
100  __b_new[__n + 1] = __b_new[__n + 1] / _Tp(Lalpha[__n]);
101  __b_new[__n] = (__b_new[__n] + _Tp(xi[__NaN[__i]]) * __b_new[__n + 1])
102  / _Tp(Lalpha[__n - 1]);
103  for (std::ptrdiff_t __j = __n - 1; __j > 0; --__j)
104  __b_new[__j] = (__b_new[__j] + _Tp(xi[__NaN[__i]]) * __b_new[__j + 1]
105  - _Tp(Lgamma[__j + 1]) * __b_new[__j + 2])
106  / _Tp(Lalpha[__j - 1]);
107  for (std::ptrdiff_t __j = 0; __j <= __n; ++__j)
108  __b_new[__j] = __b_new[__j + 1];
109  __alpha = __coeff[__n] / __b_new[__n];
110  for (std::ptrdiff_t __j = 0; __j < __n; ++__j)
111  __coeff[__j] -= __alpha * __b_new[__j];
112  __coeff[__n] = _Tp{0};
113  --__n;
114  }
115  }

117  /**
118  * CQUAD is a new doubly-adaptive general-purpose quadrature routine
119  * which can handle most types of singularities, non-numerical function
120  * values such as Inf or NaN, as well as some divergent integrals.
121  * It generally requires more function evaluations than
122  * the other integration routines in this library, yet fails
123  * less often for difficult integrands.
124  *
125  * The underlying algorithm uses a doubly-adaptive scheme
126  * in which Clenshaw-Curtis quadrature rules of increasing degree
127  * are used to compute the integral in each interval.
128  * The L_2-norm of the difference between the underlying interpolatory
129  * polynomials of two successive rules is used as an error estimate.
130  * The interval is subdivided if the difference between two successive
131  * rules is too large or a rule of maximum degree has been reached.
132  *
133  * The CQUAD algorithm divides the integration region into subintervals,
134  * and in each iteration, the subinterval with the largest estimated error
135  * is processed. The algorithm uses Clenshaw-Curtis quadrature rules
136  * of degree 4, 8, 16 and 32 over 5, 9, 17 and 33 nodes respectively.
137  * Each interval is initialized with the lowest-degree rule.
138  * When an interval is processed, the next-higher degree rule is evaluated
139  * and an error estimate is computed based on the L_2-norm of the difference
140  * between the underlying interpolating polynomials of both rules.
141  * If the highest-degree rule has already been used, or the interpolatory
142  * polynomials differ significantly, the interval is bisected.
143  */
144  template<typename _Tp, typename _FuncTp>
145  auto
146  cquad_integrate(cquad_workspace<_Tp, std::invoke_result_t<_FuncTp, _Tp>>& __ws,
147  _FuncTp __func,
148  _Tp __a, _Tp __b,
149  _Tp __epsabs, _Tp __epsrel)
150  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
151  {
152  using _RetTp = std::invoke_result_t<_FuncTp, _Tp>;
153  using _AreaTp = decltype(_RetTp{} * _Tp{});
154  //using _AbsAreaTp = decltype(std::abs(_AreaTp{}));
155  using _AbsRetTp = decltype(std::abs(_RetTp{}));
156 
157  // Some constants that we will need.
158  constexpr std::ptrdiff_t __n[4] = { 4, 8, 16, 32 };
159  constexpr std::ptrdiff_t __skip[4] = { 8, 4, 2, 1 };
160  constexpr std::ptrdiff_t __idx[4] = { 0, 5, 14, 31 };
161  constexpr std::ptrdiff_t __ndiv_max = 20;
162  constexpr auto _S_eps = std::numeric_limits<_Tp>::epsilon();
163  constexpr auto _S_NaN = std::numeric_limits<_Tp>::quiet_NaN();
164  constexpr auto _S_inf = std::numeric_limits<_Tp>::infinity();
165  constexpr _Tp _S_sqrt2 = M_SQRT2;
166  constexpr auto __w = _S_sqrt2 / _Tp{2};
167 
168  // Actual variables (as opposed to constants above).
169  bool __split;
170  std::ptrdiff_t __num_NaNs, __NaN[32];
171  _AbsRetTp __nc, __ncdiff;
172 
173  // Check for unreasonable accuracy demands.
174  if (__epsabs < _Tp{0} || __epsrel < _Tp{0})
175  std::__throw_domain_error("tolerances may not be negative");
176  if (__epsabs <= _Tp{0} && __epsrel < _S_eps)
177  std::__throw_domain_error("unreasonable accuracy requirement");
178 
179  // Create the first interval.
180  __ws.clear();
181  cquad_interval<_Tp, _RetTp> __iv;
182  auto __m = (__a + __b) / _Tp{2};
183  auto __h = (__b - __a) / _Tp{2};
184  __num_NaNs = 0;
185  for (std::ptrdiff_t __i = 0; __i <= __n[3]; ++__i)
186  {
187  __iv.fx[__i] = __func(__m + _Tp(xi[__i]) * __h);
188  if (std::isinf(__iv.fx[__i]) || std::isnan(__iv.fx[__i]))
189  {
190  __NaN[__num_NaNs++] = __i;
191  __iv.fx[__i] = _RetTp{0};
192  }
193  }
194  _Vinvfx(__iv.fx, &(__iv._M_coeff[__idx[0]]), 0);
195  _Vinvfx(__iv.fx, &(__iv._M_coeff[__idx[3]]), 3);
196  _Vinvfx(__iv.fx, &(__iv._M_coeff[__idx[2]]), 2);
197  for (std::ptrdiff_t __i = 0; __i < __num_NaNs; ++__i)
198  __iv.fx[__NaN[__i]] = _S_NaN;
199  __iv._M_lower_lim = __a;
200  __iv._M_upper_lim = __b;
201  __iv.depth = 3;
202  __iv.rdepth = 1;
203  __iv.ndiv = 0;
204  __iv._M_result = _Tp{2} * __h * __iv._M_coeff[__idx[3]] * __w;
205  __nc = _AbsRetTp{0};
206  for (std::ptrdiff_t __i = __n[2] + 1; __i <= __n[3]; ++__i)
207  {
208  const auto __temp = std::abs(__iv._M_coeff[__idx[3] + __i]);
209  __nc += __temp * __temp;
210  }
211  __ncdiff = __nc;
212  for (std::ptrdiff_t __i = 0; __i <= __n[2]; ++__i)
213  {
214  const auto __temp = std::abs(__iv._M_coeff[__idx[2] + __i]
215  - __iv._M_coeff[__idx[3] + __i]);
216  __ncdiff += __temp * __temp;
217  __nc += std::abs(__iv._M_coeff[__idx[3] + __i])
218  * std::abs(__iv._M_coeff[__idx[3] + __i]);
219  }
220  __ncdiff = std::sqrt(__ncdiff);
221  __nc = std::sqrt(__nc);
222  __iv._M_abs_error = __ncdiff * _Tp{2} * __h;
223  if (__ncdiff / __nc > _Tp{0.1} && __iv._M_abs_error < _Tp{2} * __h * __nc)
224  __iv._M_abs_error = _Tp{2} * __h * __nc;
225  __ws.push(__iv);
226 
227  // Main loop...
228  auto __igral = __iv._M_result;
229  auto __igral_final = _AreaTp{0};
230  auto __err = __iv._M_abs_error;
231  auto __err_final = _Tp{0};
232  while (__ws.size() > 0 && __err > _Tp{0} &&
233  !(__err <= std::abs(__igral) * __epsrel || __err <= __epsabs)
234  && !(__err_final > std::abs(__igral) * __epsrel
235  && __err - __err_final < std::abs(__igral) * __epsrel)
236  && !(__err_final > __epsabs && __err - __err_final < __epsabs))
237  {
238  // Put our finger on the interval with the largest error.
239  auto& __iv = __ws.top();
240  __m = (__iv._M_lower_lim + __iv._M_upper_lim) / _Tp{2};
241  __h = (__iv._M_upper_lim - __iv._M_lower_lim) / _Tp{2};
242 
243  // Should we try to increase the degree?
244  if (__iv.depth < 3)
245  {
246  // Keep tabs on some variables.
247  auto __depth = ++__iv.depth;
248 
249  // Get the new (missing) function values.
250  for (std::ptrdiff_t __i = __skip[__depth];
251  __i <= 32; __i += 2 * __skip[__depth])
252  __iv.fx[__i] = __func(__m + _Tp(xi[__i]) * __h);
253  __num_NaNs = 0;
254  for (std::ptrdiff_t __i = 0; __i <= 32; __i += __skip[__depth])
255  if (std::isinf(__iv.fx[__i]) || std::isnan(__iv.fx[__i]))
256  {
257  __NaN[__num_NaNs++] = __i;
258  __iv.fx[__i] = _RetTp{0};
259  }
260 
261  // Compute the new coefficients.
262  _Vinvfx(__iv.fx, &(__iv._M_coeff[__idx[__depth]]), __depth);
263 
264  // Downdate any NaNs.
265  if (__num_NaNs > 0)
266  {
267  downdate(&(__iv._M_coeff[__idx[__depth]]),
268  __n[__depth], __depth, __NaN, __num_NaNs);
269  for (std::ptrdiff_t __i = 0; __i < __num_NaNs; ++__i)
270  __iv.fx[__NaN[__i]] = _S_NaN;
271  }
272 
273  // Compute the error estimate.
274  __nc = _AbsRetTp{0};
275  for (std::ptrdiff_t __i = __n[__depth - 1] + 1;
276  __i <= __n[__depth]; ++__i)
277  {
278  const auto __temp = std::abs(__iv._M_coeff[__idx[__depth] + __i]);
279  __nc += __temp * __temp;
280  }
281  __ncdiff = __nc;
282  for (std::ptrdiff_t __i = 0; __i <= __n[__depth - 1]; ++__i)
283  {
284  const auto __temp
285  = std::abs(__iv._M_coeff[__idx[__depth - 1] + __i]
286  - __iv._M_coeff[__idx[__depth] + __i]);
287  __ncdiff += __temp * __temp;
288  __nc += abs(__iv._M_coeff[__idx[__depth] + __i])
289  * abs(__iv._M_coeff[__idx[__depth] + __i]);
290  }
291  __ncdiff = std::sqrt(__ncdiff);
292  __nc = std::sqrt(__nc);
293  __iv._M_abs_error = __ncdiff * _Tp{2} * __h;
294 
295  // Compute the local integral.
296  __iv._M_result = _Tp{2} * __h * __w
297  * __iv._M_coeff[__idx[__depth]];
298 
299  // Split the interval prematurely?
300  __split = (__nc > _Tp{0} && __ncdiff / __nc > _Tp{0.1});
301  }
302  else // Maximum degree reached, just split.
303  __split = true;
304 
305  // Should we drop this interval?
306  if ((__m + __h * _Tp(xi[0])) >= (__m + __h * _Tp(xi[1]))
307  || (__m + __h * _Tp(xi[31])) >= (__m + __h * _Tp(xi[32]))
308  || __iv._M_abs_error < std::abs(__iv._M_result) * _S_eps * 10)
309  {
310  // Keep this interval's contribution.
311  __err_final += __iv._M_abs_error;
312  __igral_final += __iv._M_result;
313 
314  __ws.pop();
315  }
316  else if (__split) // Do we need to split this interval?
317  {
318  // Some values we will need often...
319  auto __depth = __iv.depth;
320 
321  // Generate the interval on the left.
322  cquad_interval<_Tp, _RetTp> __ivl;
323  __ivl._M_lower_lim = __iv._M_lower_lim;
324  __ivl._M_upper_lim = __m;
325  __ivl.depth = 0;
326  __ivl.rdepth = __iv.rdepth + 1;
327  __ivl.fx[0] = __iv.fx[0];
328  __ivl.fx[32] = __iv.fx[16];
329  for (std::ptrdiff_t __i = __skip[0]; __i < 32; __i += __skip[0])
330  __ivl.fx[__i] = __func((__ivl._M_lower_lim + __ivl._M_upper_lim)
331  / _Tp{2} + _Tp(xi[__i]) * __h / _Tp{2});
332  __num_NaNs = 0;
333  for (std::ptrdiff_t __i = 0; __i <= 32; __i += __skip[0])
334  {
335  if (std::isinf(__ivl.fx[__i]) || std::isnan(__ivl.fx[__i]))
336  {
337  __NaN[__num_NaNs++] = __i;
338  __ivl.fx[__i] = _RetTp{0};
339  }
340  }
341  _Vinvfx(__ivl.fx, __ivl._M_coeff, 0);
342  if (__num_NaNs > 0)
343  {
344  downdate(__ivl._M_coeff, __n[0], 0, __NaN, __num_NaNs);
345  for (std::ptrdiff_t __i = 0; __i < __num_NaNs; ++__i)
346  __ivl.fx[__NaN[__i]] = _S_NaN;
347  }
348  for (std::ptrdiff_t __i = 0; __i <= __n[__depth]; ++__i)
349  {
350  __ivl._M_coeff[__idx[__depth] + __i] = _Tp{0};
351  for (std::ptrdiff_t __j = __i; __j <= __n[__depth]; ++__j)
352  __ivl._M_coeff[__idx[__depth] + __i]
353  += _Tp(Tleft[__i * 33 + __j])
354  * __iv._M_coeff[__idx[__depth] + __j];
355  }
356  __ncdiff = _AbsRetTp{0};
357  for (std::ptrdiff_t __i = 0; __i <= __n[0]; ++__i)
358  {
359  const auto __temp
360  = std::abs(__ivl._M_coeff[__i]

428  }

433  __ncdiff += __temp * __temp;
434  }

436  __ivr._M_abs_error = __ncdiff * __h;

440  + (std::abs(__iv._M_coeff[0]) > 0
