39  /**
40  * Compute the product of the fx with one of the inverse
41  * Vandermonde-like matrices.
42  */
43  template<typename RetTp>
44  void
45  Vinvfx(const std::array<RetTp, 33>& fx,
46  RetTp* coeff, const int depth)
47  {
48  switch (depth)
49  {
50  case 0:
51  for (int i = 0; i <= 4; ++i)
52  {
53  coeff[i] = RetTp{0};
54  for (int j = 0; j <= 4; ++j)
55  coeff[i] += RetTp(V1inv[i * 5 + j]) * fx[j * 8];
56  }
57  break;
58 
59  case 1:
60  for (int i = 0; i <= 8; ++i)
61  {
62  coeff[i] = RetTp{0};
63  for (int j = 0; j <= 8; ++j)
64  coeff[i] += RetTp(V2inv[i * 9 + j]) * fx[j * 4];
65  }
66  break;
67 
68  case 2:
69  for (int i = 0; i <= 16; ++i)
70  {
71  coeff[i] = RetTp{0};
72  for (int j = 0; j <= 16; ++j)
73  coeff[i] += RetTp(V3inv[i * 17 + j]) * fx[j * 2];
74  }
75  break;
76 
77  case 3:
78  for (int i = 0; i <= 32; ++i)
79  {
80  coeff[i] = RetTp{0};
81  for (int j = 0; j <= 32; ++j)
82  coeff[i] += RetTp(V4inv[i * 33 + j]) * fx[j];
83  }
84  break;
85  }
86  }

88  /**
89  * Downdate the interpolation given by the n coefficients c
90  * by removing the nodes with indices in NaNs.
91  */
92  template<typename Tp>
93  void
94  downdate(Tp* coeff, std::ptrdiff_t n, std::ptrdiff_t depth,
95  std::ptrdiff_t* NaN, std::ptrdiff_t num_NaNs)
96  {
97  constexpr std::ptrdiff_t bidx[4] = { 0, 6, 16, 34 };
98  Tp b_new[34], alpha;
99 
100  for (std::ptrdiff_t i = 0; i <= n + 1; ++i)
101  b_new[i] = bee[bidx[depth] + i];
102  for (std::ptrdiff_t i = 0; i < num_NaNs; ++i)
103  {
104  b_new[n + 1] = b_new[n + 1] / Tp(Lalpha[n]);
105  b_new[n] = (b_new[n] + Tp(xi[NaN[i]]) * b_new[n + 1])
106  / Tp(Lalpha[n - 1]);
107  for (std::ptrdiff_t j = n - 1; j > 0; --j)
108  b_new[j] = (b_new[j] + Tp(xi[NaN[i]]) * b_new[j + 1]
109  - Tp(Lgamma[j + 1]) * b_new[j + 2])
110  / Tp(Lalpha[j - 1]);
111  for (std::ptrdiff_t j = 0; j <= n; ++j)
112  b_new[j] = b_new[j + 1];
113  alpha = coeff[n] / b_new[n];
114  for (std::ptrdiff_t j = 0; j < n; ++j)
115  coeff[j] -= alpha * b_new[j];
116  coeff[n] = Tp{0};
117  --n;
118  }
119  }

121  /**
122  * CQUAD is a new doubly-adaptive general-purpose quadrature routine
123  * which can handle most types of singularities, non-numerical function
124  * values such as Inf or NaN, as well as some divergent integrals.
125  * It generally requires more function evaluations than
126  * the other integration routines in this library, yet fails
127  * less often for difficult integrands.
128  *
129  * The underlying algorithm uses a doubly-adaptive scheme
130  * in which Clenshaw-Curtis quadrature rules of increasing degree
131  * are used to compute the integral in each interval.
132  * The L_2-norm of the difference between the underlying interpolatory
133  * polynomials of two successive rules is used as an error estimate.
134  * The interval is subdivided if the difference between two successive
135  * rules is too large or a rule of maximum degree has been reached.
136  *
137  * The CQUAD algorithm divides the integration region into subintervals,
138  * and in each iteration, the subinterval with the largest estimated error
139  * is processed. The algorithm uses Clenshaw-Curtis quadrature rules
140  * of degree 4, 8, 16 and 32 over 5, 9, 17 and 33 nodes respectively.
141  * Each interval is initialized with the lowest-degree rule.
142  * When an interval is processed, the next-higher degree rule is evaluated
143  * and an error estimate is computed based on the L_2-norm of the difference
144  * between the underlying interpolating polynomials of both rules.
145  * If the highest-degree rule has already been used, or the interpolatory
146  * polynomials differ significantly, the interval is bisected.
147  */
148  template<typename Tp, typename FuncTp>
149  auto
150  cquad_integrate(cquad_workspace<Tp, std::invoke_result_t<FuncTp, Tp>>& ws,
151  FuncTp func,
152  Tp a, Tp b,
153  Tp epsabs, Tp epsrel)
154  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
155  {
156  using RetTp = std::invoke_result_t<FuncTp, Tp>;
157  using AreaTp = decltype(RetTp{} * Tp{});
158  //using AbsAreaTp = decltype(std::abs(AreaTp{}));
159  using AbsRetTp = decltype(std::abs(RetTp{}));
160 
161  // Some constants that we will need.
162  constexpr std::ptrdiff_t n[4] = { 4, 8, 16, 32 };
163  constexpr std::ptrdiff_t skip[4] = { 8, 4, 2, 1 };
164  constexpr std::ptrdiff_t idx[4] = { 0, 5, 14, 31 };
165  constexpr std::ptrdiff_t ndiv_max = 20;
166  constexpr auto s_eps = std::numeric_limits<Tp>::epsilon();
167  constexpr auto s_NaN = std::numeric_limits<Tp>::quiet_NaN();
168  constexpr auto s_inf = std::numeric_limits<Tp>::infinity();
169  constexpr Tp s_sqrt2 = M_SQRT2;
170  constexpr auto w = s_sqrt2 / Tp{2};
171 
172  // Actual variables (as opposed to constants above).
173  bool split;
174  std::ptrdiff_t num_NaNs, NaN[32];
175  AbsRetTp nc, ncdiff;
176 
177  // Check for unreasonable accuracy demands.
178  if (epsabs < Tp{0} || epsrel < Tp{0})
179  throw std::domain_error("tolerances may not be negative");
180  if (epsabs <= Tp{0} && epsrel < s_eps)
181  throw std::domain_error("unreasonable accuracy requirement");
182 
183  // Create the first interval.
184  ws.clear();
185  cquad_interval<Tp, RetTp> iv;
186  auto m = (a + b) / Tp{2};
187  auto h = (b - a) / Tp{2};
188  num_NaNs = 0;
189  for (std::ptrdiff_t i = 0; i <= n[3]; ++i)
190  {
191  iv.fx[i] = func(m + Tp(xi[i]) * h);
192  if (std::isinf(iv.fx[i]) || std::isnan(iv.fx[i]))
193  {
194  NaN[num_NaNs++] = i;
195  iv.fx[i] = RetTp{0};
196  }
197  }
198  Vinvfx(iv.fx, &(iv.m_coeff[idx[0]]), 0);
199  Vinvfx(iv.fx, &(iv.m_coeff[idx[3]]), 3);
200  Vinvfx(iv.fx, &(iv.m_coeff[idx[2]]), 2);
201  for (std::ptrdiff_t i = 0; i < num_NaNs; ++i)
202  iv.fx[NaN[i]] = s_NaN;
203  iv.m_lower_lim = a;
204  iv.m_upper_lim = b;
205  iv.depth = 3;
206  iv.rdepth = 1;
207  iv.ndiv = 0;
208  iv.m_result = Tp{2} * h * iv.m_coeff[idx[3]] * w;
209  nc = AbsRetTp{0};
210  for (std::ptrdiff_t i = n[2] + 1; i <= n[3]; ++i)
211  {
212  const auto temp = std::abs(iv.m_coeff[idx[3] + i]);
213  nc += temp * temp;
214  }
215  ncdiff = nc;
216  for (std::ptrdiff_t i = 0; i <= n[2]; ++i)
217  {
218  const auto temp = std::abs(iv.m_coeff[idx[2] + i]
219  - iv.m_coeff[idx[3] + i]);
220  ncdiff += temp * temp;
221  nc += std::abs(iv.m_coeff[idx[3] + i])
222  * std::abs(iv.m_coeff[idx[3] + i]);
223  }
224  ncdiff = std::sqrt(ncdiff);
225  nc = std::sqrt(nc);
226  iv.m_abs_error = ncdiff * Tp{2} * h;
227  if (ncdiff / nc > Tp{0.1} && iv.m_abs_error < Tp{2} * h * nc)
228  iv.m_abs_error = Tp{2} * h * nc;
229  ws.push(iv);
230 
231  // Main loop...
232  auto igral = iv.m_result;
233  auto igral_final = AreaTp{0};
234  auto err = iv.m_abs_error;
235  auto err_final = Tp{0};
236  while (ws.size() > 0 && err > Tp{0} &&
237  !(err <= std::abs(igral) * epsrel || err <= epsabs)
238  && !(err_final > std::abs(igral) * epsrel
239  && err - err_final < std::abs(igral) * epsrel)
240  && !(err_final > epsabs && err - err_final < epsabs))
241  {
242  // Put our finger on the interval with the largest error.
243  auto& iv = ws.top();
244  m = (iv.m_lower_lim + iv.m_upper_lim) / Tp{2};
245  h = (iv.m_upper_lim - iv.m_lower_lim) / Tp{2};
246 
247  // Should we try to increase the degree?
248  if (iv.depth < 3)
249  {
250  // Keep tabs on some variables.
251  auto depth = ++iv.depth;
252 
253  // Get the new (missing) function values.
254  for (std::ptrdiff_t i = skip[depth];
255  i <= 32; i += 2 * skip[depth])
256  iv.fx[i] = func(m + Tp(xi[i]) * h);
257  num_NaNs = 0;
258  for (std::ptrdiff_t i = 0; i <= 32; i += skip[depth])
259  if (std::isinf(iv.fx[i]) || std::isnan(iv.fx[i]))
260  {
261  NaN[num_NaNs++] = i;
262  iv.fx[i] = RetTp{0};
263  }
264 
265  // Compute the new coefficients.
266  Vinvfx(iv.fx, &(iv.m_coeff[idx[depth]]), depth);
267 
268  // Downdate any NaNs.
269  if (num_NaNs > 0)
270  {
271  downdate(&(iv.m_coeff[idx[depth]]),
272  n[depth], depth, NaN, num_NaNs);
273  for (std::ptrdiff_t i = 0; i < num_NaNs; ++i)
274  iv.fx[NaN[i]] = s_NaN;
275  }
276 
277  // Compute the error estimate.
278  nc = AbsRetTp{0};
279  for (std::ptrdiff_t i = n[depth - 1] + 1;
280  i <= n[depth]; ++i)
281  {
282  const auto temp = std::abs(iv.m_coeff[idx[depth] + i]);
283  nc += temp * temp;
284  }
285  ncdiff = nc;
286  for (std::ptrdiff_t i = 0; i <= n[depth - 1]; ++i)
287  {
288  const auto temp
289  = std::abs(iv.m_coeff[idx[depth - 1] + i]
290  - iv.m_coeff[idx[depth] + i]);
291  ncdiff += temp * temp;
292  nc += abs(iv.m_coeff[idx[depth] + i])
293  * abs(iv.m_coeff[idx[depth] + i]);
294  }
295  ncdiff = std::sqrt(ncdiff);
296  nc = std::sqrt(nc);
297  iv.m_abs_error = ncdiff * Tp{2} * h;
298 
299  // Compute the local integral.
300  iv.m_result = Tp{2} * h * w
301  * iv.m_coeff[idx[depth]];
302 
303  // Split the interval prematurely?
304  split = (nc > Tp{0} && ncdiff / nc > Tp{0.1});
305  }
306  else // Maximum degree reached, just split.
307  split = true;
308 
309  // Should we drop this interval?
310  if ((m + h * Tp(xi[0])) >= (m + h * Tp(xi[1]))
311  || (m + h * Tp(xi[31])) >= (m + h * Tp(xi[32]))
312  || iv.m_abs_error < std::abs(iv.m_result) * s_eps * 10)
313  {
314  // Keep this interval's contribution.
315  err_final += iv.m_abs_error;
316  igral_final += iv.m_result;
317 
318  ws.pop();
319  }
320  else if (split) // Do we need to split this interval?
321  {
322  // Some values we will need often...
323  auto depth = iv.depth;
324 
325  // Generate the interval on the left.
326  cquad_interval<Tp, RetTp> ivl;
327  ivl.m_lower_lim = iv.m_lower_lim;
328  ivl.m_upper_lim = m;
329  ivl.depth = 0;
330  ivl.rdepth = iv.rdepth + 1;
331  ivl.fx[0] = iv.fx[0];
332  ivl.fx[32] = iv.fx[16];
333  for (std::ptrdiff_t i = skip[0]; i < 32; i += skip[0])
334  ivl.fx[i] = func((ivl.m_lower_lim + ivl.m_upper_lim)
335  / Tp{2} + Tp(xi[i]) * h / Tp{2});
336  num_NaNs = 0;
337  for (std::ptrdiff_t i = 0; i <= 32; i += skip[0])
338  {
339  if (std::isinf(ivl.fx[i]) || std::isnan(ivl.fx[i]))
340  {
341  NaN[num_NaNs++] = i;
342  ivl.fx[i] = RetTp{0};
343  }
344  }
345  Vinvfx(ivl.fx, ivl.m_coeff, 0);
346  if (num_NaNs > 0)
347  {
348  downdate(ivl.m_coeff, n[0], 0, NaN, num_NaNs);
349  for (std::ptrdiff_t i = 0; i < num_NaNs; ++i)
350  ivl.fx[NaN[i]] = s_NaN;
351  }
352  for (std::ptrdiff_t i = 0; i <= n[depth]; ++i)
353  {
354  ivl.m_coeff[idx[depth] + i] = Tp{0};
355  for (std::ptrdiff_t j = i; j <= n[depth]; ++j)
356  ivl.m_coeff[idx[depth] + i]
357  += Tp(Tleft[i * 33 + j])
358  * iv.m_coeff[idx[depth] + j];
359  }
360  ncdiff = AbsRetTp{0};
361  for (std::ptrdiff_t i = 0; i <= n[0]; ++i)
362  {
363  const auto temp
364  = std::abs(ivl.m_coeff[i]
365  - ivl.m_coeff[idx[depth] + i]);
366  ncdiff += temp * temp;
367  }
368  for (std::ptrdiff_t i = n[0] + 1; i <= n[depth]; ++i)
369  {
370  const auto temp
371  = std::abs(ivl.m_coeff[idx[depth] + i]);
372  ncdiff += temp * temp;
373  }
374  ncdiff = std::sqrt(ncdiff);
375  ivl.m_abs_error = ncdiff * h;
376 
377  // Check for divergence.
378  ivl.ndiv = iv.ndiv
379  + (std::abs(iv.m_coeff[0]) > 0
380  && std::abs(ivl.m_coeff[0] / iv.m_coeff[0]) > 2);
381  if (ivl.ndiv > ndiv_max && 2 * ivl.ndiv > ivl.rdepth)
382  {
383  //const auto result = std::copysign(s_inf, igral);
384  return {AreaTp(s_inf), s_inf};
385  }
386 
387  // Compute the local integral.
388  ivl.m_result = h * w * ivl.m_coeff[0];
389 
390  // Generate the interval on the right.
391  cquad_interval<Tp, RetTp> ivr;
392  ivr.m_lower_lim = m;
393  ivr.m_upper_lim = iv.m_upper_lim;
394  ivr.depth = 0;
395  ivr.rdepth = iv.rdepth + 1;
396  ivr.fx[0] = iv.fx[16];
397  ivr.fx[32] = iv.fx[32];
398  for (std::ptrdiff_t i = skip[0]; i < 32; i += skip[0])
399  ivr.fx[i] = func((ivr.m_lower_lim + ivr.m_upper_lim)
400  / Tp{2} + Tp(xi[i]) * h / Tp{2});
401  num_NaNs = 0;
402  for (std::ptrdiff_t i = 0; i <= 32; i += skip[0])
403  {
404  if (std::isinf(ivr.fx[i]) || std::isnan(ivr.fx[i]))
405  {
406  NaN[num_NaNs++] = i;
407  ivr.fx[i] = RetTp{0};
408  }
409  }
410  Vinvfx (ivr.fx, ivr.m_coeff, 0);
411  if (num_NaNs > 0)
412  {
413  downdate(ivr.m_coeff, n[0], 0, NaN, num_NaNs);
414  for (std::ptrdiff_t i = 0; i < num_NaNs; ++i)
415  ivr.fx[NaN[i]] = s_NaN;
416  }
417  for (std::ptrdiff_t i = 0; i <= n[depth]; ++i)
418  {
419  ivr.m_coeff[idx[depth] + i] = Tp{0};
420  for (std::ptrdiff_t j = i; j <= n[depth]; ++j)
421  ivr.m_coeff[idx[depth] + i]
422  += Tp(Tright[i * 33 + j])
423  * iv.m_coeff[idx[depth] + j];
424  }
425  ncdiff = AbsRetTp{0};
426  for (std::ptrdiff_t i = 0; i <= n[0]; ++i)
427  {
428  const auto temp
429  = std::abs(ivr.m_coeff[i]
430  - ivr.m_coeff[idx[depth] + i]);
431  ncdiff += temp * temp;
432  }
433  for (std::ptrdiff_t i = n[0] + 1; i <= n[depth]; ++i)
434  {
435  const auto temp
436  = std::abs(ivr.m_coeff[idx[depth] + i]);
437  ncdiff += temp * temp;
438  }
439  ncdiff = std::sqrt(ncdiff);
440  ivr.m_abs_error = ncdiff * h;
441 
442  // Check for divergence.
443  ivr.ndiv = iv.ndiv
444  + (std::abs(iv.m_coeff[0]) > 0
445  && std::abs(ivr.m_coeff[0] / iv.m_coeff[0]) > 2);
446  if (ivr.ndiv > ndiv_max && 2 * ivr.ndiv > ivr.rdepth)
447  {
448  //const auto result = std::copysign(s_inf, igral);
449  return {AreaTp(s_inf), s_inf};
450  }
451 
452  // Compute the local integral.
453  ivr.m_result = h * w * ivr.m_coeff[0];
454 
455  ws.pop();
456  ws.push(ivl);
457  ws.push(ivr);
458  }
459  else // Otherwise, just fix-up the heap.
460  ws.update();
461 
462  // Collect the value of the integral and error.
463  igral = igral_final + ws.total_integral();
464  err = err_final + ws.total_error();
465  }
466 
467  return {igral, err};
468  }
