35  extrapolation_table<_Tp>::qelg()
36  {
37  const auto __cur_n = this->_M_nn - 1;
38  const auto __current = this->_M_rlist2[__cur_n];
39 
40  const auto _S_eps = std::numeric_limits<_Tp>::epsilon();
41  // Less than max to prevent overflow.
42  const auto _S_max = std::numeric_limits<_Tp>::max() / _Tp{100};
43 
44  auto __absolute = _S_max;
45  auto __relative = 5 * _S_eps * std::abs(__current);
46 
47  const auto __newelm = __cur_n / 2;
48  const auto __n_orig = __cur_n;
49  auto __n_final = __cur_n;
50 
51  const auto __nres_orig = this->_M_nres;
52 
53  auto __result = __current;
54  auto __abserr = _S_max;
55 
56  if (__cur_n < 2)
57  {
58  __result = __current;
59  __abserr = std::max(__absolute, __relative);
60  return std::make_tuple(__result, __abserr);
61  }
62 
63  this->_M_rlist2[__cur_n + 2] = this->_M_rlist2[__cur_n];
64  this->_M_rlist2[__cur_n] = _S_max;
65 
66  for (size_t __ii = 0; __ii < __newelm; ++__ii)
67  {
68  auto __res = this->_M_rlist2[__cur_n - 2 * __ii + 2];
69  const auto __e0 = this->_M_rlist2[__cur_n - 2 * __ii - 2];
70  const auto __e1 = this->_M_rlist2[__cur_n - 2 * __ii - 1];
71  const auto __e2 = __res;
72 
73  const auto __e1abs = std::abs(__e1);
74  const auto __delta2 = __e2 - __e1;
75  const auto __err2 = std::abs(__delta2);
76  const auto __tol2 = std::max(std::abs(__e2), __e1abs) * _S_eps;
77  const auto __delta3 = __e1 - __e0;
78  const auto __err3 = std::abs(__delta3);
79  const auto __tol3 = std::max(__e1abs, std::abs(__e0)) * _S_eps;
80 
81  if (__err2 <= __tol2 && __err3 <= __tol3)
82  {
83  // If e0, e1 and e2 are equal to within machine accuracy,
84  // convergence is assumed.
85  __result = __res;
86  __absolute = __err2 + __err3;
87  __relative = 5 * _S_eps * std::abs(__res);
88  __abserr = std::max(__absolute, __relative);
89  return std::make_tuple(__result, __abserr);
90  }
91 
92  const auto __e3 = this->_M_rlist2[__cur_n - 2 * __ii];
93  this->_M_rlist2[__cur_n - 2 * __ii] = __e1;
94  const auto __delta1 = __e1 - __e3;
95  const auto __err1 = std::abs(__delta1);
96  const auto __tol1 = std::max(__e1abs, std::abs(__e3)) * _S_eps;
97 
98  // If two elements are very close to each other, omit a part of
99  // the table by adjusting the value of n.
100  if (__err1 <= __tol1 || __err2 <= __tol2 || __err3 <= __tol3)
101  {
102  __n_final = 2 * __ii;
103  break;
104  }
105 
106  const auto __ss = _Tp{1} / __delta1 + _Tp{1} / __delta2
107  - _Tp{1} / __delta3;
108 
109  // Test to detect irregular behaviour in the table,
110  // and eventually omit a part of the table by adjusting
111  // the value of n.
112  if (std::abs(__ss * __e1) <= this->_M_irreg_test)
113  {
114  __n_final = 2 * __ii;
115  break;
116  }
117 
118  // Compute a new element and eventually adjust the value of result.
119  __res = __e1 + _Tp{1} / __ss;
120  this->_M_rlist2[__cur_n - 2 * __ii] = __res;
121  {
122  const auto __error = __err2 + std::abs(__res - __e2) + __err3;
123 
124  if (__error <= __abserr)
125  {
126  __abserr = __error;
127  __result = __res;
128  }
129  }
130  }
131 
132  // Shift the table.
133 
134  {
135  const size_t __limexp = 50 - 1;
136 
137  if (__n_final == __limexp)
138  __n_final = 2 * (__limexp / 2);
139  }
140 
141  if (__n_orig % 2 == 1)
142  {
143  for (size_t __ii = 0; __ii <= __newelm; ++__ii)
144  this->_M_rlist2[__ii * 2 + 1] = this->_M_rlist2[__ii * 2 + 3];
145  }
146  else
147  {
148  for (size_t __ii = 0; __ii <= __newelm; ++__ii)
149  this->_M_rlist2[__ii * 2] = this->_M_rlist2[__ii * 2 + 2];
150  }
151 
152  if (__n_orig != __n_final)
153  {
154  for (size_t __ii = 0; __ii <= __n_final; ++__ii)
155  this->_M_rlist2[__ii] = this->_M_rlist2[__ii + __n_orig - __n_final];
156  }
157 
158  this->_M_nn = __n_final + 1;
159 
160  if (__nres_orig < 3)
161  {
162  this->_M_res3la[__nres_orig] = __result;
163  __abserr = _S_max;
164  }
165  else
166  { // Compute error estimate.
167  __abserr = (std::abs(__result - this->_M_res3la[2])
168  + std::abs(__result - this->_M_res3la[1])
169  + std::abs(__result - this->_M_res3la[0]));
170 
171  this->_M_res3la[0] = this->_M_res3la[1];
172  this->_M_res3la[1] = this->_M_res3la[2];
173  this->_M_res3la[2] = __result;
174  }
175 
176  /* In QUADPACK the variable table->nres is incremented at the top of
177  qelg, so it increases on every call. This leads to the array
178  res3la being accessed when its elements are still undefined, so I
179  have moved the update to this point so that its value more
180  useful. */
181 
182  this->_M_nres = __nres_orig + 1;
183 
184  __abserr = std::max(__abserr, 5 * _S_eps * std::abs(__result));
185 
186  return std::make_tuple(__result, __abserr);
187  }
