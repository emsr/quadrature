48  oscillatory_integration_table(Tp omega_in, Tp length_in,
49  circular_function circfun_in,
50  std::size_t n_in)
51  : n(n_in),
52  omega(omega_in),
53  length(length_in),
54  par(0.5 * omega_in * length_in),
55  circfun(circfun_in),
56  chebmo(25 * n_in)

58  auto scale = Tp{1};
59  for (auto i = 0u; i < this->n; ++i)
60  {
61  this->compute_moments(this->par * scale, i);
62  scale *= 0.5;
63  // Prevent divide by zero.
64  if (const auto scale2 = scale * scale;
65  scale2 * scale2 == Tp{0})
66  {
67  this->n = i;
68  break;
69  }
70  }
71  }
