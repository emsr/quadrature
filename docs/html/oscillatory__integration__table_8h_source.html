50  oscillatory_integration_table(_Tp omega_in, _Tp length_in,
51  circular_function circfun_in,
52  std::size_t n_in)
53  : n(n_in),
54  omega(omega_in),
55  length(length_in),
56  par(0.5 * omega_in * length_in),
57  circfun(circfun_in),
58  chebmo(25 * n_in)
59  {
60  auto __scale = _Tp{1};
61  for (auto __i = 0u; __i < this->n; ++__i)
62  {
63  this->compute_moments(this->par * __scale, __i);
64  __scale *= 0.5;
65  // Prevent divide by zero.
66  if (const auto __scale2 = __scale * __scale;
67  __scale2 * __scale2 == _Tp{0})
68  {
69  this->n = __i;
70  break;
71  }
72  }
73  }

79  oscillatory_integration_table<_Tp>&
80  set_length(_Tp length_in)
81  {
82  this->length = length_in;

88  void
89  reset(_Tp omega_in, _Tp length_in,
90  circular_function circfun_in)
91  {
92  this->omega = omega_in;
93  this->length = length_in;
94  this->par = 0.5 * omega_in * length_in;
95  this->circfun = circfun_in;
96  auto __scale = _Tp{1};

98  {
99  this->compute_moments(this->par * __scale, __i);
100  __scale *= 0.5;
101  }
102  }
