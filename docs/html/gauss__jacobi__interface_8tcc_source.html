205 /** Aproximates the integral of a function using a a quadrature.
206  *
207  * @f[
208  * \int_{-1}^1 f(x) dx \approx \sum_{i=0}^{Q-1} w_i f(x_i)
209  * @f]
210  *
211  * Where @f$x_i@f$ and @f$w_i@f$ are given by the quadrature method emplyed.
212  *
213  * @param quad A structure containing quadrature information
214  * @param f Value of function at quadrature points
215  * @return @f$\int_{-1}^{1} f(x) dx \approx \sum_{i=0}^{Q-1} w_i f(x_i)@f$
216  */
217 template<typename _Tp>
218  template<typename _Func>
219  _Tp
220  jac_quadrature<_Tp>::integrate(_Func fun)
221  {
222  std::vector<_Tp> f(this->Q);
223  for (int i = 0; i < this->Q; ++i)
224  f[i] = fun(this->x[i]);
225  return std::inner_product(std::begin(this->w), std::end(this->w),
226  std::begin(f), _Tp{0});
227  }
