46  /**
47  * Adaptive integration for Cauchy principal values:
48  * @f[
49  * I = \int_a^b dx f(x) / (x - c)
50  * @f]
51  * The adaptive bisection algorithm of QAG is used, with modifications
52  * to ensure that subdivisions do not occur at the singular point x = c.
53  * When a subinterval contains the point x = c or is close to it then
54  * a special 25-point modified Clenshaw-Curtis rule is used to control
55  * the singularity. Further away from the singularity the algorithm uses
56  * a user-supplied integration rule (default 15-point Gauss-Kronrod).
57  */
58  template<typename _Tp, typename _FuncTp,
59  typename _Integrator = gauss_kronrod_integral<_Tp>>
60  auto
61  qawc_integrate(integration_workspace<_Tp,
62  std::invoke_result_t<_FuncTp, _Tp>>& __workspace,
63  _FuncTp __func,
64  _Tp __lower, _Tp __upper, _Tp __center,
65  _Tp __max_abs_err, _Tp __max_rel_err,
66  _Integrator __quad = gauss_kronrod_integral<_Tp>(Kronrod_15))
67  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
68  {
69  using _RetTp = std::invoke_result_t<_FuncTp, _Tp>;
70  using _AreaTp = decltype(_RetTp{} * _Tp{});
71  using _AbsAreaTp = decltype(std::abs(_AreaTp{}));
72 
73  auto __result = _AreaTp{};
74  auto __abserr = _AbsAreaTp{};
75 
76  const auto __limit = __workspace.capacity();
77  // Try to adjust tests for varing precision.
78  const auto _M_rel_err = std::pow(_Tp{10.0},
79  -std::numeric_limits<_Tp>::digits / _Tp{10.0});
80 
81  int __sign = 1;
82  if (__upper < __lower)
83  {
84  std::swap(__lower, __upper);
85  __sign = -1;
86  }
87 
88  if (!valid_tolerances(__max_abs_err, __max_rel_err))
89  {
90  std::ostringstream __msg;
91  __msg << "qawc_integrate: Tolerance cannot be achieved with given "
92  "absolute (" << __max_abs_err << ") and relative ("
93  << __max_rel_err << ") error limits.";
94  std::__throw_runtime_error(__msg.str().c_str());
95  }
96 
97  if (__center == __lower || __center == __upper)
98  std::__throw_runtime_error ("qawc_integrate: "
99  "Cannot integrate with singularity "
100  "on endpoint.");
101 
102  __workspace.clear();
103 
104  // Perform the first integration.
105  auto [__result0, __abserr0, __err_reliable]
106  = qc25c(__func, __lower, __upper, __center, __quad);
107 
108  __workspace.append(__lower, __upper, __result0, __abserr0);
109 
110  // Test on accuracy; Use 0.01 relative error as an extra safety
111  // margin on the first iteration (ignored for subsequent iterations).
112  auto __tolerance = std::max(__max_abs_err,
113  __max_rel_err * std::abs( __result0));
114  if (__abserr0 < __tolerance && __abserr0
115  < _Tp{0.01} * std::abs(__result0))
116  return {__sign * __result0, __abserr0};
117  else if (__limit == 1)
118  __throw_integration_error("qawc_integrate: "
119  "A maximum of one iteration was insufficient",
120  MAX_ITER_ERROR, __sign * __result0, __abserr0);
121 
122  auto __area = __result0;
123  auto __errsum = __abserr0;
124  auto __iteration = 1u;
125  int __error_type = NO_ERROR;
126  int __roundoff_type1 = 0, __roundoff_type2 = 0;
127  do
128  {
129  // Bisect the subinterval with the largest error estimate.
130  const auto& __curr = __workspace.retrieve();
131 
132  const auto __a1 = __curr.__lower_lim;
133  const auto __b2 = __curr.__upper_lim;
134  auto __mid = (__curr.__lower_lim + __curr.__upper_lim) / _Tp{2};
135  if (__center > __a1 && __center <= __mid)
136  __mid = (__center + __b2) / _Tp{2};
137  else if (__center > __mid && __center < __b2)
138  __mid = (__a1 + __center) / _Tp{2};
139  const auto __a2 = __mid;
140 
141  auto [__area1, __error1, __err_reliable1]
142  = qc25c(__func, __a1, __mid, __center, __quad);
143 
144  auto [__area2, __error2, __err_reliable2]
145  = qc25c(__func, __a2, __b2, __center, __quad);
146 
147  const auto __area12 = __area1 + __area2;
148  const auto __error12 = __error1 + __error2;
149 
150  __errsum += __error12 - __curr.__abs_error;
151  __area += __area12 - __curr.__result;
152 
153  if (__err_reliable1 && __err_reliable2)
154  {
155  const auto __delta = __curr.__result - __area12;
156 
157  if (std::abs (__delta) <= _M_rel_err * std::abs (__area12)
158  && __error12 >= 0.99 * __curr.__abs_error)
159  ++__roundoff_type1;
160  if (__iteration >= 10 && __error12 > __curr.__abs_error)
161  ++__roundoff_type2;
162  }
163 
164  __tolerance = std::max(__max_abs_err, __max_rel_err * std::abs(__area));
165  if (__errsum > __tolerance)
166  {
167  if (__roundoff_type1 >= 6 || __roundoff_type2 >= 20)
168  __error_type = ROUNDOFF_ERROR;
169 
170  // set error flag in the case of bad integrand behaviour at
171  // a point of the integration range
172  if (__workspace.subinterval_too_small(__a1, __a2, __b2))
173  __error_type = SINGULAR_ERROR;
174  }
175 
176  __workspace.split(__mid, __area1, __error1, __area2, __error2);
177 
178  ++__iteration;
179  }
180  while (__iteration < __limit && !__error_type && __errsum > __tolerance);
181 
182  __result = __sign * __workspace.total_integral();
183  __abserr = __errsum;
184 
185  if (__iteration == __limit)
186  __error_type = MAX_SUBDIV_ERROR;
187 
188  if (__errsum <= __tolerance)
189  return {__result, __abserr};
190 
191  if (__error_type == NO_ERROR)
192  return {__result, __abserr};
193 
194  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
195  __throw_integration_error("qawc_integrate: Unknown error.",
196  UNKNOWN_ERROR, __result, __abserr);
197  }

199  /**
200  * Adaptive integration for Cauchy principal values:
201  * @f[
202  * I = \int_a^b dx f(x) / (x - c)
203  * @f]
204  * The adaptive bisection algorithm of QAG is used, with modifications
205  * to ensure that subdivisions do not occur at the singular point x = c.
206  * When a subinterval contains the point x = c or is close to it then
207  * a special 25-point modified Clenshaw-Curtis rule is used to control
208  * the singularity. Further away from the singularity the algorithm uses
209  * a user-supplied integration rule.
210  */
211  template<typename _Tp, typename _FuncTp,
212  typename _Integrator = gauss_kronrod_integral<_Tp>>
213  auto
214  qc25c(_FuncTp __func, _Tp __lower, _Tp __upper, _Tp __center,
215  _Integrator __quad)
216  -> std::tuple<decltype(_Tp{} * __func(_Tp{})), _Tp, bool>
217  {
218  using _RetTp = std::invoke_result_t<_FuncTp, _Tp>;
219  using _AreaTp = decltype(_RetTp{} * _Tp{});
220  using _AbsAreaTp = decltype(std::abs(_AreaTp{}));
221  bool __err_reliable;
222 
223  auto __result = _AreaTp{};
224  auto __abserr = _AbsAreaTp{};
225 
226  const auto __cc = (_Tp{2} * __center - __upper - __lower)
227  / (__upper - __lower);
228 
229  if (std::abs(__cc) > _Tp{1.1})
230  {
231  auto __func_cauchy = [__func, __center](_Tp __x)
232  -> _Tp
233  { return __func(__x) / (__x - __center); };
234 
235  auto [__result, __abserr, __resabs, __resasc]
236  = __quad(__func_cauchy, __lower, __upper);
237 
238  if (__abserr == __resasc)
239  __err_reliable = false;
240  else
241  __err_reliable = true;
242 
243  return std::make_tuple(__result, __abserr, __err_reliable);
244  }
245  else
246  {
247  auto [__cheb12, __cheb24] = qcheb_integrate(__func, __lower, __upper);
248  const auto __moment = compute_moments(__cheb24.size(), __cc);
249 
250  auto __res12 = _AreaTp{0};
251  for (size_t __i = 0u; __i < __cheb12.size(); ++__i)
252  __res12 += __cheb12[__i] * __moment[__i];
253 
254  auto __res24 = _AreaTp{0};
255  for (size_t __i = 0u; __i < __cheb24.size(); ++__i)
256  __res24 += __cheb24[__i] * __moment[__i];
257 
258  __result = __res24;
259  __abserr = std::abs(__res24 - __res12);
260  __err_reliable = false;
261 
262  return std::make_tuple(__result, __abserr, __err_reliable);
263  }
264  }

266  /**
267  * Compute Clenshaw-Curtis moments.
268  * An iterator range version would be nicer I think.
269  */
270  template<typename _Tp>
271  std::vector<_Tp>
272  compute_moments(std::size_t __N, _Tp __cc)
273  {
274  std::vector<_Tp> __moment(__N);
275 
276  auto __a0 = std::log(std::abs((_Tp{1} - __cc) / (_Tp{1} + __cc)));
277  auto __a1 = _Tp{2} + __a0 * __cc;
278 
279  __moment[0] = __a0;
280  __moment[1] = __a1;
281 
282  for (size_t __k = 2; __k < __N; ++__k)
283  {
284  _Tp __a2;
285 
286  if ((__k % 2) == 0)
287  __a2 = _Tp{2} * __cc * __a1 - __a0;
288  else
289  {
290  const auto __km1 = _Tp(__k - 1);
291  __a2 = _Tp{2} * __cc * __a1
292  - __a0
293  - _Tp{4} / (__km1 * __km1 - _Tp{1});
294  }
295 
296  __moment[__k] = __a2;
297 
298  __a0 = __a1;
299  __a1 = __a2;
300  }
301 
302  return __moment;
303  }
