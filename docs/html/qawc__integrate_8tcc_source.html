47  /**
48  * Adaptive integration for Cauchy principal values:
49  * @f[
50  * I = \int_a^b dx f(x) / (x - c)
51  * @f]
52  * The adaptive bisection algorithm of QAG is used, with modifications
53  * to ensure that subdivisions do not occur at the singular point x = c.
54  * When a subinterval contains the point x = c or is close to it then
55  * a special 25-point modified Clenshaw-Curtis rule is used to control
56  * the singularity. Further away from the singularity the algorithm uses
57  * a user-supplied integration rule (default 15-point Gauss-Kronrod).
58  */
59  template<typename Tp, typename FuncTp,
60  typename Integrator = gauss_kronrod_integral<Tp>>
61  auto
62  qawc_integrate(integration_workspace<Tp,
63  std::invoke_result_t<FuncTp, Tp>>& workspace,
64  FuncTp func,
65  Tp lower, Tp upper, Tp center,
66  Tp max_abs_err, Tp max_rel_err,
67  Integrator quad = gauss_kronrod_integral<Tp>(Kronrod_15))
68  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
69  {
70  using RetTp = std::invoke_result_t<FuncTp, Tp>;
71  using AreaTp = decltype(RetTp{} * Tp{});
72  using AbsAreaTp = decltype(std::abs(AreaTp{}));
73 
74  auto result = AreaTp{};
75  auto abserr = AbsAreaTp{};
76 
77  const auto limit = workspace.capacity();
78  // Try to adjust tests for varing precision.
79  const auto m_rel_err = std::pow(Tp{10.0},
80  -std::numeric_limits<Tp>::digits / Tp{10.0});
81 
82  int sign = 1;
83  if (upper < lower)
84  {
85  std::swap(lower, upper);
86  sign = -1;
87  }
88 
89  if (!valid_tolerances(max_abs_err, max_rel_err))
90  {
91  std::ostringstream msg;
92  msg << "qawc_integrate: Tolerance cannot be achieved with given "
93  "absolute (" << max_abs_err << ") and relative ("
94  << max_rel_err << ") error limits.";
95  throw std::runtime_error(msg.str().c_str());
96  }
97 
98  if (center == lower || center == upper)
99  throw std::runtime_error ("qawc_integrate: "
100  "Cannot integrate with singularity "
101  "on endpoint.");
102 
103  workspace.clear();
104 
105  // Perform the first integration.
106  auto [result0, abserr0, err_reliable]
107  = qc25c(func, lower, upper, center, quad);
108 
109  workspace.append(lower, upper, result0, abserr0);
110 
111  // Test on accuracy; Use 0.01 relative error as an extra safety
112  // margin on the first iteration (ignored for subsequent iterations).
113  auto tolerance = std::max(max_abs_err,
114  max_rel_err * std::abs( result0));
115  if (abserr0 < tolerance && abserr0
116  < Tp{0.01} * std::abs(result0))
117  return {sign * result0, abserr0};
118  else if (limit == 1)
119  throw integration_error("qawc_integrate: "
120  "A maximum of one iteration was insufficient",
121  MAX_ITER_ERROR, sign * result0, abserr0);
122 
123  auto area = result0;
124  auto errsum = abserr0;
125  auto iteration = 1u;
126  int error_type = NO_ERROR;
127  int roundoff_type1 = 0, roundoff_type2 = 0;
128  do
129  {
130  // Bisect the subinterval with the largest error estimate.
131  const auto& curr = workspace.retrieve();
132 
133  const auto a1 = curr.lower_lim;
134  const auto b2 = curr.upper_lim;
135  auto mid = (curr.lower_lim + curr.upper_lim) / Tp{2};
136  if (center > a1 && center <= mid)
137  mid = (center + b2) / Tp{2};
138  else if (center > mid && center < b2)
139  mid = (a1 + center) / Tp{2};
140  const auto a2 = mid;
141 
142  auto [area1, error1, err_reliable1]
143  = qc25c(func, a1, mid, center, quad);
144 
145  auto [area2, error2, err_reliable2]
146  = qc25c(func, a2, b2, center, quad);
147 
148  const auto area12 = area1 + area2;
149  const auto error12 = error1 + error2;
150 
151  errsum += error12 - curr.abs_error;
152  area += area12 - curr.result;
153 
154  if (err_reliable1 && err_reliable2)
155  {
156  const auto delta = curr.result - area12;
157 
158  if (std::abs (delta) <= m_rel_err * std::abs (area12)
159  && error12 >= 0.99 * curr.abs_error)
160  ++roundoff_type1;
161  if (iteration >= 10 && error12 > curr.abs_error)
162  ++roundoff_type2;
163  }
164 
165  tolerance = std::max(max_abs_err, max_rel_err * std::abs(area));
166  if (errsum > tolerance)
167  {
168  if (roundoff_type1 >= 6 || roundoff_type2 >= 20)
169  error_type = ROUNDOFF_ERROR;
170 
171  // set error flag in the case of bad integrand behaviour at
172  // a point of the integration range
173  if (workspace.subinterval_too_small(a1, a2, b2))
174  error_type = SINGULAR_ERROR;
175  }
176 
177  workspace.split(mid, area1, error1, area2, error2);
178 
179  ++iteration;
180  }
181  while (iteration < limit && !error_type && errsum > tolerance);
182 
183  result = sign * workspace.total_integral();
184  abserr = errsum;
185 
186  if (iteration == limit)
187  error_type = MAX_SUBDIV_ERROR;
188 
189  if (errsum <= tolerance)
190  return {result, abserr};
191 
192  if (error_type == NO_ERROR)
193  return {result, abserr};
194 
195  check_error(__func__, error_type, result, abserr);
196  throw integration_error("qawc_integrate: Unknown error.",
197  UNKNOWN_ERROR, result, abserr);
198  }

200  /**
201  * Adaptive integration for Cauchy principal values:
202  * @f[
203  * I = \int_a^b dx f(x) / (x - c)
204  * @f]
205  * The adaptive bisection algorithm of QAG is used, with modifications
206  * to ensure that subdivisions do not occur at the singular point x = c.
207  * When a subinterval contains the point x = c or is close to it then
208  * a special 25-point modified Clenshaw-Curtis rule is used to control
209  * the singularity. Further away from the singularity the algorithm uses
210  * a user-supplied integration rule.
211  */
212  template<typename Tp, typename FuncTp,
213  typename Integrator = gauss_kronrod_integral<Tp>>
214  auto
215  qc25c(FuncTp func, Tp lower, Tp upper, Tp center,
216  Integrator quad)
217  -> std::tuple<decltype(Tp{} * func(Tp{})), Tp, bool>
218  {
219  using RetTp = std::invoke_result_t<FuncTp, Tp>;
220  using AreaTp = decltype(RetTp{} * Tp{});
221  using AbsAreaTp = decltype(std::abs(AreaTp{}));
222  bool err_reliable;
223 
224  auto result = AreaTp{};
225  auto abserr = AbsAreaTp{};
226 
227  const auto cc = (Tp{2} * center - upper - lower)
228  / (upper - lower);
229 
230  if (std::abs(cc) > Tp{1.1})
231  {
232  auto func_cauchy = [func, center](Tp x)
233  -> Tp
234  { return func(x) / (x - center); };
235 
236  auto [result, abserr, resabs, resasc]
237  = quad(func_cauchy, lower, upper);
238 
239  if (abserr == resasc)
240  err_reliable = false;
241  else
242  err_reliable = true;
243 
244  return std::make_tuple(result, abserr, err_reliable);
245  }
246  else
247  {
248  auto [cheb12, cheb24] = qcheb_integrate(func, lower, upper);
249  const auto moment = compute_moments(cheb24.size(), cc);
250 
251  auto res12 = AreaTp{0};
252  for (size_t i = 0u; i < cheb12.size(); ++i)
253  res12 += cheb12[i] * moment[i];
254 
255  auto res24 = AreaTp{0};
256  for (size_t i = 0u; i < cheb24.size(); ++i)
257  res24 += cheb24[i] * moment[i];
258 
259  result = res24;
260  abserr = std::abs(res24 - res12);
261  err_reliable = false;
262 
263  return std::make_tuple(result, abserr, err_reliable);
264  }
265  }

267  /**
268  * Compute Clenshaw-Curtis moments.
269  * An iterator range version would be nicer I think.
270  */
271  template<typename Tp>
272  std::vector<Tp>
273  compute_moments(std::size_t N, Tp cc)
274  {
275  std::vector<Tp> moment(N);
276 
277  auto a0 = std::log(std::abs((Tp{1} - cc) / (Tp{1} + cc)));
278  auto a1 = Tp{2} + a0 * cc;
279 
280  moment[0] = a0;
281  moment[1] = a1;
282 
283  for (size_t k = 2; k < N; ++k)
284  {
285  Tp a2;
286 
287  if ((k % 2) == 0)
288  a2 = Tp{2} * cc * a1 - a0;
289  else
290  {
291  const auto km1 = Tp(k - 1);
292  a2 = Tp{2} * cc * a1
293  - a0
294  - Tp{4} / (km1 * km1 - Tp{1});
295  }
296 
297  moment[k] = a2;
298 
299  a0 = a1;
300  a1 = a2;
301  }
302 
303  return moment;
304  }
