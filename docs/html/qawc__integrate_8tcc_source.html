46  /**
47  * Adaptive integration for Cauchy principal values:
48  * @f[
49  * I = \int_a^b dx f(x) / (x - c)
50  * @f]
51  * The adaptive bisection algorithm of QAG is used, with modifications
52  * to ensure that subdivisions do not occur at the singular point x = c.
53  * When a subinterval contains the point x = c or is close to it then
54  * a special 25-point modified Clenshaw-Curtis rule is used to control
55  * the singularity. Further away from the singularity the algorithm uses
56  * a user-supplied integration rule (default 15-point Gauss-Kronrod).
57  */
58  template<typename Tp, typename FuncTp,
59  typename Integrator = gauss_kronrod_integral<Tp>>
60  auto
61  qawc_integrate(integration_workspace<Tp,
62  std::invoke_result_t<FuncTp, Tp>>& workspace,
63  FuncTp func,
64  Tp lower, Tp upper, Tp center,
65  Tp max_abs_err, Tp max_rel_err,
66  Integrator quad = gauss_kronrod_integral<Tp>(Kronrod_15))
67  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
68  {
69  using RetTp = std::invoke_result_t<FuncTp, Tp>;
70  using AreaTp = decltype(RetTp{} * Tp{});
71  using AbsAreaTp = decltype(std::abs(AreaTp{}));
72 
73  auto result = AreaTp{};
74  auto abserr = AbsAreaTp{};
75 
76  const auto limit = workspace.capacity();
77  // Try to adjust tests for varing precision.
78  const auto m_rel_err = std::pow(Tp{10.0},
79  -std::numeric_limits<Tp>::digits / Tp{10.0});
80 
81  int sign = 1;
82  if (upper < lower)
83  {
84  std::swap(lower, upper);
85  sign = -1;
86  }
87 
88  if (!valid_tolerances(max_abs_err, max_rel_err))
89  {
90  std::ostringstream msg;
91  msg << "qawc_integrate: Tolerance cannot be achieved with given "
92  "absolute (" << max_abs_err << ") and relative ("
93  << max_rel_err << ") error limits.";
94  throw std::runtime_error(msg.str().c_str());
95  }
96 
97  if (center == lower || center == upper)
98  throw std::runtime_error ("qawc_integrate: "
99  "Cannot integrate with singularity "
100  "on endpoint.");
101 
102  workspace.clear();
103 
104  // Perform the first integration.
105  auto [result0, abserr0, err_reliable]
106  = qc25c(func, lower, upper, center, quad);
107 
108  workspace.append(lower, upper, result0, abserr0);
109 
110  // Test on accuracy; Use 0.01 relative error as an extra safety
111  // margin on the first iteration (ignored for subsequent iterations).
112  auto tolerance = std::max(max_abs_err,
113  max_rel_err * std::abs( result0));
114  if (abserr0 < tolerance && abserr0
115  < Tp{0.01} * std::abs(result0))
116  return {sign * result0, abserr0};
117  else if (limit == 1)
118  throw integration_error("qawc_integrate: "
119  "A maximum of one iteration was insufficient",
120  MAX_ITER_ERROR, sign * result0, abserr0);
121 
122  auto area = result0;
123  auto errsum = abserr0;
124  auto iteration = 1u;
125  int error_type = NO_ERROR;
126  int roundoff_type1 = 0, roundoff_type2 = 0;
127  do
128  {
129  // Bisect the subinterval with the largest error estimate.
130  const auto& curr = workspace.retrieve();
131 
132  const auto a1 = curr.lower_lim;
133  const auto b2 = curr.upper_lim;
134  auto mid = (curr.lower_lim + curr.upper_lim) / Tp{2};
135  if (center > a1 && center <= mid)
136  mid = (center + b2) / Tp{2};
137  else if (center > mid && center < b2)
138  mid = (a1 + center) / Tp{2};
139  const auto a2 = mid;
140 
141  auto [area1, error1, err_reliable1]
142  = qc25c(func, a1, mid, center, quad);
143 
144  auto [area2, error2, err_reliable2]
145  = qc25c(func, a2, b2, center, quad);
146 
147  const auto area12 = area1 + area2;
148  const auto error12 = error1 + error2;
149 
150  errsum += error12 - curr.abs_error;
151  area += area12 - curr.result;
152 
153  if (err_reliable1 && err_reliable2)
154  {
155  const auto delta = curr.result - area12;
156 
157  if (std::abs (delta) <= m_rel_err * std::abs (area12)
158  && error12 >= 0.99 * curr.abs_error)
159  ++roundoff_type1;
160  if (iteration >= 10 && error12 > curr.abs_error)
161  ++roundoff_type2;
162  }
163 
164  tolerance = std::max(max_abs_err, max_rel_err * std::abs(area));
165  if (errsum > tolerance)
166  {
167  if (roundoff_type1 >= 6 || roundoff_type2 >= 20)
168  error_type = ROUNDOFF_ERROR;
169 
170  // set error flag in the case of bad integrand behaviour at
171  // a point of the integration range
172  if (workspace.subinterval_too_small(a1, a2, b2))
173  error_type = SINGULAR_ERROR;
174  }
175 
176  workspace.split(mid, area1, error1, area2, error2);
177 
178  ++iteration;
179  }
180  while (iteration < limit && !error_type && errsum > tolerance);
181 
182  result = sign * workspace.total_integral();
183  abserr = errsum;
184 
185  if (iteration == limit)
186  error_type = MAX_SUBDIV_ERROR;
187 
188  if (errsum <= tolerance)
189  return {result, abserr};
190 
191  if (error_type == NO_ERROR)
192  return {result, abserr};
193 
194  check_error(__func__, error_type, result, abserr);
195  throw integration_error("qawc_integrate: Unknown error.",
196  UNKNOWN_ERROR, result, abserr);
197  }

199  /**
200  * Adaptive integration for Cauchy principal values:
201  * @f[
202  * I = \int_a^b dx f(x) / (x - c)
203  * @f]
204  * The adaptive bisection algorithm of QAG is used, with modifications
205  * to ensure that subdivisions do not occur at the singular point x = c.
206  * When a subinterval contains the point x = c or is close to it then
207  * a special 25-point modified Clenshaw-Curtis rule is used to control
208  * the singularity. Further away from the singularity the algorithm uses
209  * a user-supplied integration rule.
210  */
211  template<typename Tp, typename FuncTp,
212  typename Integrator = gauss_kronrod_integral<Tp>>
213  auto
214  qc25c(FuncTp func, Tp lower, Tp upper, Tp center,
215  Integrator quad)
216  -> std::tuple<decltype(Tp{} * func(Tp{})), Tp, bool>
217  {
218  using RetTp = std::invoke_result_t<FuncTp, Tp>;
219  using AreaTp = decltype(RetTp{} * Tp{});
220  using AbsAreaTp = decltype(std::abs(AreaTp{}));
221  bool err_reliable;
222 
223  auto result = AreaTp{};
224  auto abserr = AbsAreaTp{};
225 
226  const auto cc = (Tp{2} * center - upper - lower)
227  / (upper - lower);
228 
229  if (std::abs(cc) > Tp{1.1})
230  {
231  auto func_cauchy = [func, center](Tp x)
232  -> Tp
233  { return func(x) / (x - center); };
234 
235  auto [result, abserr, resabs, resasc]
236  = quad(func_cauchy, lower, upper);
237 
238  if (abserr == resasc)
239  err_reliable = false;
240  else
241  err_reliable = true;
242 
243  return std::make_tuple(result, abserr, err_reliable);
244  }
245  else
246  {
247  auto [cheb12, cheb24] = qcheb_integrate(func, lower, upper);
248  const auto moment = compute_moments(cheb24.size(), cc);
249 
250  auto res12 = AreaTp{0};
251  for (size_t i = 0u; i < cheb12.size(); ++i)
252  res12 += cheb12[i] * moment[i];
253 
254  auto res24 = AreaTp{0};
255  for (size_t i = 0u; i < cheb24.size(); ++i)
256  res24 += cheb24[i] * moment[i];
257 
258  result = res24;
259  abserr = std::abs(res24 - res12);
260  err_reliable = false;
261 
262  return std::make_tuple(result, abserr, err_reliable);
263  }
264  }

266  /**
267  * Compute Clenshaw-Curtis moments.
268  * An iterator range version would be nicer I think.
269  */
270  template<typename Tp>
271  std::vector<Tp>
272  compute_moments(std::size_t N, Tp cc)
273  {
274  std::vector<Tp> moment(N);
275 
276  auto a0 = std::log(std::abs((Tp{1} - cc) / (Tp{1} + cc)));
277  auto a1 = Tp{2} + a0 * cc;
278 
279  moment[0] = a0;
280  moment[1] = a1;
281 
282  for (size_t k = 2; k < N; ++k)
283  {
284  Tp a2;
285 
286  if ((k % 2) == 0)
287  a2 = Tp{2} * cc * a1 - a0;
288  else
289  {
290  const auto km1 = Tp(k - 1);
291  a2 = Tp{2} * cc * a1
292  - a0
293  - Tp{4} / (km1 * km1 - Tp{1});
294  }
295 
296  moment[k] = a2;
297 
298  a0 = a1;
299  a1 = a2;
300  }
301 
302  return moment;
303  }
