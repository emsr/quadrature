104  template<typename _Tp>
105  void
106  __check_error(std::string_view __func, int __errcode,
107  _Tp __result = std::numeric_limits<_Tp>::quiet_NaN(),
108  _Tp __abserr = std::numeric_limits<_Tp>::quiet_NaN())
109  {
110  std::ostringstream msg;
111  msg << __func << ": ";
112 
113  if (__errcode > 2)
114  --__errcode;
115  switch(__errcode)
116  {
117  case NO_ERROR:
118  return;
119  case MAX_ITER_ERROR:
120  msg << "Number of iterations was insufficient";
121  break;
122  case ROUNDOFF_ERROR:
123  msg << "Cannot reach tolerance because of roundoff error";
124  break;
125  case SINGULAR_ERROR:
126  msg << "Bad integrand behavior found in the integration interval";
127  break;
128  case EXTRAP_ROUNDOFF_ERROR:
129  msg << "Roundoff error detected in the extrapolation";
130  break;
131  case DIVERGENCE_ERROR:
132  msg << "Integral is divergent, or slowly convergent";
133  break;
134  case MAX_SUBDIV_ERROR:
135  msg << "Maximum number of subdivisions reached";
136  break;
137  case TOLERANCE_ERROR:
138  msg << "Cannot reach tolerance with maximum order rule";
139  break;
140  default:
141  msg << "Could not integrate function";
142  }
143 
144  __throw_integration_error(msg.str().c_str(), __errcode,
145  __result, __abserr);
146  }

151  template<typename _Tp>
152  _Tp
153  __rescale_error(_Tp __err,
154  const _Tp __result_abs, const _Tp __result_asc)
155  {
156  const auto _S_eps = std::numeric_limits<_Tp>::epsilon();
157  const auto _S_min = std::numeric_limits<_Tp>::min();
158 
159  __err = std::abs(__err);
160  if (__result_asc != 0 && __err != _Tp{0})
161  {
162  auto __scale = std::pow((_Tp{200} * __err / __result_asc), _Tp{1.5});
163 
164  if (__scale < _Tp{1})
165  __err = __result_asc * __scale;
166  else
167  __err = __result_asc;
168  }
169  if (__result_abs > _S_min / (_Tp{50} * _S_eps))
170  {
171  auto __min_err = _Tp{50} * _S_eps * __result_abs;
172 
173  if (__min_err > __err)
174  __err = __min_err;
175  }
176 
177  return __err;
178  }
