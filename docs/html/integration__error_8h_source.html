79  template<typename AreaTp, typename AbsAreaTp>
80  void
81  check_error(std::string_view func, int errcode,
82  AreaTp result, AbsAreaTp abserr)
83  {
84  std::ostringstream msg;
85  msg << func << ": ";
86 
87  if (errcode > 2)
88  --errcode;
89  switch(errcode)
90  {
91  case NO_ERROR:
92  return;
93  case MAX_ITER_ERROR:
94  msg << "Number of iterations was insufficient";
95  break;
96  case ROUNDOFF_ERROR:
97  msg << "Cannot reach tolerance because of roundoff error";
98  break;
99  case SINGULAR_ERROR:
100  msg << "Bad integrand behavior found in the integration interval";
101  break;
102  case EXTRAP_ROUNDOFF_ERROR:
103  msg << "Roundoff error detected in the extrapolation";
104  break;
105  case DIVERGENCE_ERROR:
106  msg << "Integral is divergent, or slowly convergent";
107  break;
108  case MAX_SUBDIV_ERROR:
109  msg << "Maximum number of subdivisions reached";
110  break;
111  case TOLERANCE_ERROR:
112  msg << "Cannot reach tolerance with maximum order rule";
113  break;
114  default:
115  msg << "Could not integrate function";
116  }
117 
118  throw integration_error(msg.str().c_str(), errcode, result, abserr);
119  }

124  template<typename AreaTp, typename AbsAreaTp>
125  AbsAreaTp
126  rescale_error(AreaTp err,
127  const AbsAreaTp result_abs, const AbsAreaTp result_asc)
128  {
129  const auto s_eps = std::numeric_limits<AbsAreaTp>::epsilon();
130  const auto s_min = std::numeric_limits<AbsAreaTp>::min();
131 
132  AbsAreaTp abserr = std::abs(err);
133  if (result_asc != AbsAreaTp{} && abserr != AbsAreaTp{})
134  {
135  auto scale = std::pow((AbsAreaTp{200} * abserr / result_asc), AbsAreaTp{1.5});
136 
137  if (scale < AbsAreaTp{1})
138  abserr = result_asc * scale;
139  else
140  abserr = result_asc;
141  }
142  if (result_abs > s_min / (AbsAreaTp{50} * s_eps))
143  {
144  auto min_err = AbsAreaTp{50} * s_eps * result_abs;
145 
146  if (min_err > abserr)
147  abserr = min_err;
148  }
149 
150  return abserr;
151  }
