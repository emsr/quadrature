61  /**
62  * Routine to retrieve the i-th Gauss-Legendre point and weight.
63  * Useful when the caller wishes to access the information stored in
64  * the high-precision gauss_legendre_table struct.
65  * Points are indexed and presented in increasing order to the caller.
66  */
67  template<typename _Tp>
68  std::tuple<_Tp, _Tp>
69  gauss_legendre_table<_Tp>::get_point(_Tp __lower, _Tp __upper, size_t __i) const
70  {
71  const auto __hwidth = (__upper - __lower) / _Tp{2};
72  const auto __midpt = (__lower + __upper) / _Tp{2};
73 
74  if (__i >= this->order)
75  std::__throw_domain_error("gauss_legendre_table: i must be less than n");
76 
77  // See comments above gauss_legendre_table for struct's x, w layout.
78  // Simply unpack that layout into a sorted set of points, weights.
79  _Tp __xi, __wi;
80  if (this->order & 1) // n is odd
81  {
82  const auto __k = int(__i) - int(this->order) / 2;
83  const auto sign = (__k < 0 ? -1 : +1);
84 
85  __xi = __midpt + sign * __hwidth * this->pt(sign * __k);
86  __wi = __hwidth * this->wt(sign * __k);
87  }
88  else if (/* n is even && */ __i < this->order / 2)
89  {
90  __i = int(this->order) / 2 - 1 - int(__i);
91  __xi = __midpt - __hwidth * this->pt(__i);
92  __wi = __hwidth * this->wt(__i);
93  }
94  else // n is even and __i >= n / 2
95  {
96  __i -= this->order / 2;
97  __xi = __midpt + __hwidth * this->pt(__i);
98  __wi = __hwidth * this->wt(__i);
99  }
100 
101  return std::make_tuple(__xi, __wi);
102  }
