57  /**
58  * Routine to retrieve the i-th Gauss-Legendre point and weight.
59  * Useful when the caller wishes to access the information stored in
60  * the high-precision gauss_legendre_table struct.
61  * Points are indexed and presented in increasing order to the caller.
62  */
63  template<typename Tp>
64  std::tuple<Tp, Tp>
65  gauss_legendre_table<Tp>::get_point(Tp lower, Tp upper, size_t i) const
66  {
67  const auto hwidth = (upper - lower) / Tp{2};
68  const auto midpt = (lower + upper) / Tp{2};
69 
70  if (i >= this->order)
71  throw std::domain_error("gauss_legendre_table: i must be less than n");
72 
73  // See comments above gauss_legendre_table for struct's x, w layout.
74  // Simply unpack that layout into a sorted set of points, weights.
75  Tp xi, wi;
76  if (this->order & 1) // n is odd
77  {
78  const auto k = int(i) - int(this->order) / 2;
79  const auto sign = (k < 0 ? -1 : +1);
80 
81  xi = midpt + sign * hwidth * this->pt(sign * k);
82  wi = hwidth * this->wt(sign * k);
83  }
84  else if (/* n is even && */ i < this->order / 2)
85  {
86  i = int(this->order) / 2 - 1 - int(i);
87  xi = midpt - hwidth * this->pt(i);
88  wi = hwidth * this->wt(i);
89  }
90  else // n is even and i >= n / 2
91  {
92  i -= this->order / 2;
93  xi = midpt + hwidth * this->pt(i);
94  wi = hwidth * this->wt(i);
95  }
96 
97  return std::make_tuple(xi, wi);
98  }
