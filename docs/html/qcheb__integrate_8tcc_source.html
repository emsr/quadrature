29  template<typename RetTp>
30  struct chebyshev_integral_t
31  {
32  std::array<RetTp, 13> cheb12;
33  std::array<RetTp, 25> cheb24;
34  };
35 
36  template<typename Tp, typename FuncTp>
37  auto
38  qcheb_integrate(FuncTp func, Tp lower, Tp upper)
39  -> chebyshev_integral_t<std::invoke_result_t<FuncTp, Tp>>
40  {
41  using RetTp = std::invoke_result_t<FuncTp, Tp>;
42 
43  chebyshev_integral_t<RetTp> out;
44  auto& cheb12 = out.cheb12;
45  auto& cheb24 = out.cheb24;
46  RetTp fval[25], v[12];
47 
48  // These are the values of cos(pi*k/24) for k=1..11 needed for the
49  // Chebyshev expansion of f(x). These are the zeros of the Chebyshev
50  // function of the second kind of order 23: U_23(x).
51 
52  constexpr Tp
53  x[11]
54  {
55  9.914448613738104111442846968605486e-01L,
56  9.659258262890682867486612158530536e-01L,
57  9.238795325112867561257834975394469e-01L,
58  8.660254037844386467595427060757126e-01L,
59  7.933533402912351645734146973742314e-01L,
60  7.071067811865475243919762573395221e-01L,
61  6.087614290087206394044894932434070e-01L,
62  4.999999999999999999855184455596035e-01L,
63  3.826834323650897717110798781478690e-01L,
64  2.588190451025207623287087436359508e-01L,
65  1.305261922200515915256103766723547e-01L,
66  };
67 
68  const auto center = (upper + lower) / Tp{2};
69  const auto half_length = (upper - lower) / Tp{2};
70 
71  fval[0] = func(upper) / Tp{2};
72  fval[12] = func(center);
73  fval[24] = func(lower) / Tp{2};
74 
75  for (int i = 1; i < 12; ++i)
76  {
77  const std::size_t j = 24 - i;
78  const auto u = half_length * x[i - 1];
79  fval[i] = func(center + u);
80  fval[j] = func(center - u);
81  }
82 
83  for (int i = 0; i < 12; ++i)
84  {
85  const std::size_t j = 24 - i;
86  v[i] = fval[i] - fval[j];
87  fval[i] = fval[i] + fval[j];
88  }
89 
90  {
91  const auto alam1 = v[0] - v[8];
92  const auto alam2 = x[5] * (v[2] - v[6] - v[10]);
93  cheb12[3] = alam1 + alam2;
94  cheb12[9] = alam1 - alam2;
95  }
96 
97  {
98  const auto alam1 = v[1] - v[7] - v[9];
99  const auto alam2 = v[3] - v[5] - v[11];
100  {
101  const auto alam = x[2] * alam1 + x[8] * alam2;
102  cheb24[3] = cheb12[3] + alam;
103  cheb24[21] = cheb12[3] - alam;
104  }
105 
106  {
107  const auto alam = x[8] * alam1 - x[2] * alam2;
108  cheb24[9] = cheb12[9] + alam;
109  cheb24[15] = cheb12[9] - alam;
110  }
111  }
112 
113  {
114  const auto part1 = x[3] * v[4];
115  const auto part2 = x[7] * v[8];
116  const auto part3 = x[5] * v[6];
117 
118  {
119  const auto alam1 = v[0] + part1 + part2;
120  const auto alam2 = x[1] * v[2] + part3 + x[9] * v[10];
121 
122  cheb12[1] = alam1 + alam2;
123  cheb12[11] = alam1 - alam2;
124  }
125 
126  {
127  const Tp alam1 = v[0] - part1 + part2;
128  const Tp alam2 = x[9] * v[2] - part3 + x[1] * v[10];
129  cheb12[5] = alam1 + alam2;
130  cheb12[7] = alam1 - alam2;
131  }
132  }
133 
134  {
135  const auto alam = (x[0] * v[1] + x[2] * v[3]
136  + x[4] * v[5] + x[6] * v[7]
137  + x[8] * v[9] + x[10] * v[11]);
138  cheb24[1] = cheb12[1] + alam;
139  cheb24[23] = cheb12[1] - alam;
140  }
141 
142  {
143  const auto alam = (x[10] * v[1] - x[8] * v[3]
144  + x[6] * v[5] - x[4] * v[7]
145  + x[2] * v[9] - x[0] * v[11]);
146  cheb24[11] = cheb12[11] + alam;
147  cheb24[13] = cheb12[11] - alam;
148  }
149 
150  {
151  const auto alam = (x[4] * v[1] - x[8] * v[3]
152  - x[0] * v[5] - x[10] * v[7]
153  + x[2] * v[9] + x[6] * v[11]);
154  cheb24[5] = cheb12[5] + alam;
155  cheb24[19] = cheb12[5] - alam;
156  }
157 
158  {
159  const auto alam = (x[6] * v[1] - x[2] * v[3]
160  - x[10] * v[5] + x[0] * v[7]
161  - x[8] * v[9] - x[4] * v[11]);
162  cheb24[7] = cheb12[7] + alam;
163  cheb24[17] = cheb12[7] - alam;
164  }
165 
166  for (int i = 0; i < 6; ++i)
167  {
168  const std::size_t j = 12 - i;
169  v[i] = fval[i] - fval[j];
170  fval[i] = fval[i] + fval[j];
171  }
172 
173  {
174  const auto alam1 = v[0] + x[7] * v[4];
175  const auto alam2 = x[3] * v[2];
176 
177  cheb12[2] = alam1 + alam2;
178  cheb12[10] = alam1 - alam2;
179  }
180 
181  cheb12[6] = v[0] - v[4];
182 
183  {
184  const auto alam = x[1] * v[1] + x[5] * v[3] + x[9] * v[5];
185  cheb24[2] = cheb12[2] + alam;
186  cheb24[22] = cheb12[2] - alam;
187  }
188 
189  {
190  const auto alam = x[5] * (v[1] - v[3] - v[5]);
191  cheb24[6] = cheb12[6] + alam;
192  cheb24[18] = cheb12[6] - alam;
193  }
194 
195  {
196  const auto alam = x[9] * v[1] - x[5] * v[3] + x[1] * v[5];
197  cheb24[10] = cheb12[10] + alam;
198  cheb24[14] = cheb12[10] - alam;
199  }
200 
201  for (int i = 0; i < 3; ++i)
202  {
203  const std::size_t j = 6 - i;
204  v[i] = fval[i] - fval[j];
205  fval[i] = fval[i] + fval[j];
206  }
207 
208  cheb12[4] = v[0] + x[7] * v[2];
209  cheb12[8] = fval[0] - x[7] * fval[2];
210 
211  {
212  const auto alam = x[3] * v[1];
213  cheb24[4] = cheb12[4] + alam;
214  cheb24[20] = cheb12[4] - alam;
215  }
216 
217  {
218  const auto alam = x[7] * fval[1] - fval[3];
219  cheb24[8] = cheb12[8] + alam;
220  cheb24[16] = cheb12[8] - alam;
221  }
222 
223  cheb12[0] = fval[0] + fval[2];
224 
225  {
226  const auto alam = fval[1] + fval[3];
227  cheb24[0] = cheb12[0] + alam;
228  cheb24[24] = cheb12[0] - alam;
229  }
230 
231  cheb12[12] = v[0] - v[2];
232  cheb24[12] = cheb12[12];
233 
234  for (int i = 1; i < 12; ++i)
235  cheb12[i] *= Tp{1} / Tp{6};
236 
237  cheb12[0] *= Tp{1} / Tp{12};
238  cheb12[12] *= Tp{1} / Tp{12};
239 
240  for (int i = 1; i < 24; ++i)
241  cheb24[i] *= Tp{1} / Tp{12};
242 
243  cheb24[0] *= Tp{1} / Tp{24};
244  cheb24[24] *= Tp{1} / Tp{24};
245 
246  return out;
247  }
