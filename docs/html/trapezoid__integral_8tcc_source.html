29  /**
30  * Integrate the function by naive subdivision.
31  */
32  template<typename Tp, typename FuncTp>
33  typename composite_trapezoid_integral< Tp, FuncTp>::AreaTp
34  composite_trapezoid_integral< Tp, FuncTp>::operator()()
35  {
36  const auto delta = (this->m_upper_lim - this->m_lower_lim)
37  / this->m_num_segs;
38 
39  auto sum = this->m_fun(this->m_lower_lim) / Tp{2};
40  for (std::size_t j = 1; j < this->m_num_segs; ++j)
41  sum += this->m_fun(this->m_lower_lim + j * delta);
42  sum += this->m_fun(this->m_upper_lim) / Tp{2};
43 
44  this->m_result = sum * delta;
45 
46  return this->m_result;
47  }

49  /**
50  * Integrate the function by globally adaptive binary subdivision.
51  */
52  template<typename Tp, typename FuncTp>
53  typename trapezoid_integral< Tp, FuncTp>::AreaTp
54  trapezoid_integral< Tp, FuncTp>::operator()()
55  {
56  auto sum_prev = this->m_step();
57  for (std::size_t j = 1; j < s_max_iter; ++j)
58  {
59  const auto sum = this->m_step();
60  this->m_abs_error = std::abs(sum - sum_prev);
61  if (this->m_abs_error < this->m_rel_tol * std::abs(sum))
62  return sum;
63  if (j > 6
64  && std::abs(sum) < this->m_rel_tol
65  && std::abs(sum_prev) < this->m_rel_tol )
66  return sum;
67  sum_prev = sum;
68  }
69  return sum_prev;
70  }

72  /**
73  * Chances are, if the function returns Nan or inf, we stepped on a pole.
74  */
75  template<typename Tp, typename FuncTp>
76  std::invoke_result_t<FuncTp, Tp>
77  wrap_func(FuncTp func, Tp x)
78  {
79  auto y = func(x);
80  if (std::isnan(y) || std::isinf(y))
81  return Tp{0};
82  else
83  return y;
84  }

89  template<typename Tp, typename FuncTp>
90  typename trapezoid_integral< Tp, FuncTp>::AreaTp
91  trapezoid_integral< Tp, FuncTp>::m_step()
92  {
93  if (this->m_iter == 0)
94  {
95  this->m_iter = 1;
96  this->m_result = (this->m_upper_lim - this->m_lower_lim)
97  * (wrap_func(this->m_fun, this->m_lower_lim)
98  + wrap_func(this->m_fun, this->m_upper_lim))
99  / Tp{2};
100  this->m_pow2 = 1;
101  }
102  else
103  {
104  ++this->m_iter;
105  const auto del = (this->m_upper_lim - this->m_lower_lim)
106  / this->m_pow2;
107  if (std::abs(del) < s_min_delta)
108  return this->m_result;
109  auto x = this->m_lower_lim + del / Tp{2};
110  auto sum = AreaTp{};
111  for (std::size_t j = 0; j < this->m_pow2; ++j, x += del)
112  sum += wrap_func(this->m_fun, x);
113  this->m_result = (this->m_result + del * sum) / Tp{2};
114  this->m_pow2 *= 2;
115  }
116  return this->m_result;
117  }
