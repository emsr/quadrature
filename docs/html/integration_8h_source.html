34  /**

74  /**
75  * An error tolerance model for integrals.
76  */
77  template<typename Tp>
78  struct error_tolerance_t
79  {
80  constexpr
81  error_tolerance_t(Tp max_abs_err, Tp max_rel_err,
82  unsigned min_num_passes);
83 
84  /// Maximum absolute error tolerance.
85  Tp m_max_abs_err;
86  /// Maximum relative error tolerance.
87  Tp m_max_rel_err;
88  /// Current tolerance.
89  Tp m_tolerance = this->tolerance(Tp{1});
90  /// Minimum number of consecutive passes before result is OK.
91  unsigned m_min_num_passes;
92  /// Current number of passes.
93  unsigned m_num_passes;
94 
95  /// Set and return the tolerance given an initial result.
96  template<typename ResTp>
97  constexpr Tp
98  tolerance(ResTp result)
99  {
100  this->m_tolerance
101  = std::max(this->m_max_abs_err,
102  Tp(this->m_max_rel_err * std::abs(result)));
103  return this->m_tolerance;
104  }
105 
106  /// Return whether convergence is acceptable tolerance
107  /// given two subsequent results.
108  template<typename ResTp>
109  bool
110  test(ResTp curr_result, ResTp prev_result)
111  {
112  const auto del = std::abs(curr_result - prev_result);
113  if (del < this->m_max_abs_err
114  || del < this->m_max_rel_err * std::abs(curr_result))
115  ++this->m_num_passes;
116  else
117  this->m_num_passes = 0;
118  return this->m_num_passes >= this->m_min_num_passes;
119  }
120 
121  /// Return the current tolerance.
122  constexpr Tp
123  tolerance() const
124  { return this->m_tolerance; }
125 
126  /// Test for valid error tolerances.
127  static constexpr bool
128  s_valid_tolerances(Tp max_abs_err, Tp max_rel_err)
129  {
130  constexpr auto s_eps = std::numeric_limits<Tp>::epsilon();
131  return !(max_abs_err <= Tp{0}
132  && (max_rel_err < Tp{50} * s_eps
133  || max_rel_err < 0.5e-28));
134  // I don't understand the etymology of this last number.
135  }
136  };

139  template<typename Tp>
140  inline bool
141  valid_tolerances(Tp max_abs_err, Tp max_rel_err)
142  {
143  return error_tolerance_t<Tp>::

437  /**
438  * The singular weight function is defined by:
439  * @f[
440  * W(x) = (x-a)^\alpha (b-x)^\beta log^\mu (x-a) log^\nu (b-x)
441  * @f]
442  * where @f$ \alpha > -1 @f$, @f$ \beta > -1 @f$,
443  * and @f$ \mu = 0 @f$, 1, @f$ \nu = 0, 1 @f$.
444  *
445  * The weight function can take four different forms depending
446  * on the values of \mu and \nu,
447  * @f[
448  * W(x) = (x-a)^\alpha (b-x)^\beta (\mu = 0, \nu = 0)
449  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) (\mu = 1, \nu = 0)
450  * W(x) = (x-a)^\alpha (b-x)^\beta log(b-x) (\mu = 0, \nu = 1)
451  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) log(b-x) (\mu = 1, \nu = 1)
452  * @f]
453  *
454  * The QAWS algorithm is designed for integrands with algebraic-logarithmic
455  * singularities at the end-points of an integration region.
456  */
457  template<typename Tp, typename FuncTp>
458  auto
459  integrate_singular_endpoints(FuncTp func,
460  Tp lower, Tp upper,
461  Tp alpha, Tp beta,
462  int mu, int nu,
463  Tp max_abs_error,
464  Tp max_rel_error,
465  std::size_t max_iter = 1024)
466  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>;

480  * @f[
481  * \int_{-1}^{+1}f(x)dx
482  * @f]
483  * Making the change of variables:
484  * @f[
485  * x = tanh\left[\frac{\pi}{2}sinh(u)\right],
486  * dx = \frac{\pi}{2}
487  * \frac{cosh(u)}{cosh^2\left[\frac{\pi}{2}sinh(u)\right]}du
488  * @f]
489  * gives the following integral:
490  * @f[
