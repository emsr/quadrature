78  /**
79  * An error tolerance model for integrals.
80  */
81  template<typename _Tp>
82  struct error_tolerance_t
83  {
84  constexpr
85  error_tolerance_t(_Tp __max_abs_err, _Tp __max_rel_err,
86  unsigned __min_num_passes);
87 
88  /// Maximum absolute error tolerance.
89  _Tp _M_max_abs_err;
90  /// Maximum relative error tolerance.
91  _Tp _M_max_rel_err;
92  /// Current tolerance.
93  _Tp _M_tolerance = this->tolerance(_Tp{1});
94  /// Minimum number of consecutive passes before result is OK.
95  unsigned _M_min_num_passes;
96  /// Current number of passes.
97  unsigned _M_num_passes;
98 
99  /// Set and return the tolerance given an initial result.
100  template<typename _ResTp>
101  constexpr _Tp
102  tolerance(_ResTp __result)
103  {
104  this->_M_tolerance
105  = std::max(this->_M_max_abs_err,
106  _Tp(this->_M_max_rel_err * std::abs(__result)));
107  return this->_M_tolerance;
108  }
109 
110  /// Return whether convergence is acceptable tolerance
111  /// given two subsequent results.
112  template<typename _ResTp>
113  bool
114  test(_ResTp __curr_result, _ResTp __prev_result)
115  {
116  const auto __del = std::abs(__curr_result - __prev_result);
117  if (__del < this->_M_max_abs_err
118  || __del < this->_M_max_rel_err * std::abs(__curr_result))
119  ++this->_M_num_passes;
120  else
121  this->_M_num_passes = 0;
122  return this->_M_num_passes >= this->_M_min_num_passes;
123  }
124 
125  /// Return the current tolerance.
126  constexpr _Tp
127  tolerance() const
128  { return this->_M_tolerance; }
129 
130  /// Test for valid error tolerances.
131  static constexpr bool
132  _S_valid_tolerances(_Tp __max_abs_err, _Tp __max_rel_err)
133  {
134  constexpr auto _S_eps = std::numeric_limits<_Tp>::epsilon();
135  return !(__max_abs_err <= _Tp{0}
136  && (__max_rel_err < _Tp{50} * _S_eps
137  || __max_rel_err < 0.5e-28));
138  // I don't understand the etymology of this last number.
139  }
140  };

146  {
147  return error_tolerance_t<_Tp>::

443  /**
444  * The singular weight function is defined by:
445  * @f[
446  * W(x) = (x-a)^\alpha (b-x)^\beta log^\mu (x-a) log^\nu (b-x)
447  * @f]
448  * where @f$ \alpha > -1 @f$, @f$ \beta > -1 @f$,
449  * and @f$ \mu = 0 @f$, 1, @f$ \nu = 0, 1 @f$.
450  *
451  * The weight function can take four different forms depending
452  * on the values of \mu and \nu,
453  * @f[
454  * W(x) = (x-a)^\alpha (b-x)^\beta (\mu = 0, \nu = 0)
455  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) (\mu = 1, \nu = 0)
456  * W(x) = (x-a)^\alpha (b-x)^\beta log(b-x) (\mu = 0, \nu = 1)
457  * W(x) = (x-a)^\alpha (b-x)^\beta log(x-a) log(b-x) (\mu = 1, \nu = 1)
458  * @f]
459  *
460  * The QAWS algorithm is designed for integrands with algebraic-logarithmic
461  * singularities at the end-points of an integration region.
462  */
463  template<typename _Tp, typename _FuncTp>
464  auto
465  integrate_singular_endpoints(_FuncTp __func,
466  _Tp __lower, _Tp __upper,
467  _Tp __alpha, _Tp __beta,
468  int __mu, int __nu,
469  _Tp __max_abs_error,
470  _Tp __max_rel_error,
471  std::size_t __max_iter = 1024)
472  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>;
