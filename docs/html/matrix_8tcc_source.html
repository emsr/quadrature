28  /**
29  * Solve a tridiagonal system A x = b:
30  *
31  * @param[in] supd The superdiagonal of the matrix in [0, n - 2]
32  * @param[in] diag The diagonal of the matrix in [0, n - 1]
33  * @param[in] subd The subdiagonal of the matrix in [1, n - 1]
34  *
35  * @param[out] rhs The right hand side in [0, n - 1],
36  * replaced by the solution vector x
37  */
38  template<typename RandAccIter, typename RandAccIterRHS>
39  int
40  s_tridiag(size_t n,
41  RandAccIter supd, RandAccIter diag, RandAccIter subd,
42  RandAccIterRHS rhs)
43  {
44  using Tp = std::decay_t<decltype(subd[0])>;
45 
46  subd[0] = diag[0];
47 
48  if (n == 0)
49  return 0;
50 
51  if (n == 1)
52  {
53  rhs[0] /= diag[0];
54  return 0;
55  }
56 
57  diag[0] = supd[0];
58  supd[0] = Tp{0};
59  supd[n - 1] = Tp{0};
60 
61  for (auto k = 0u; k < n - 1; ++k)
62  {
63  const auto k1 = k + 1;
64 
65  if (std::abs(subd[k1]) >= std::abs(subd[k]))
66  {
67  std::swap(subd[k1], subd[k]);
68  std::swap(diag[k1], diag[k]);
69  std::swap(supd[k1], supd[k]);
70  std::swap(rhs[k1], rhs[k]);
71  }
72 
73  if (subd[k] == Tp{0})
74  return 1;
75 
76  {
77  const auto t = -subd[k1] / subd[k];
78 
79  subd[k1] = diag[k1] + t * diag[k];
80  diag[k1] = supd[k1] + t * supd[k];
81  supd[k1] = Tp{0};
82  rhs[k1] = rhs[k1] + t * rhs[k];
83  }
84  }
85 
86  if (subd[n - 1] == Tp{0})
87  return 1;
88 
89  rhs[n - 1] = rhs[n - 1] / subd[n - 1];
90 
91  rhs[n - 2] = (rhs[n - 2] - diag[n - 2] * rhs[n - 1])
92  / subd[n - 2];
93 
94  for (std::ptrdiff_t k = n ; k > 2; --k)
95  {
96  const auto kb = k - 3;
97  rhs[kb] = (rhs[kb] - diag[kb] * rhs[kb + 1]
98  - supd[kb] * rhs[kb + 2]) / subd[kb];
99  }
100 
101  return 0;
102  }

104  /**
105  * @brief Diagonalize a symmetric tridiagonal matrix.
106  *
107  * This routine is a slightly modified version of the EISPACK routine to
108  * perform the implicit QL algorithm on a symmetric tridiagonal matrix.
109  *
110  * It has been modified to produce the product Q' * Z, where Z is an input
111  * vector and Q is the orthogonal matrix diagonalizing the input matrix.
112  * The changes consist (essentialy) of applying the orthogonal
113  * transformations directly to Z as they are generated.
114  *
115  * Original C++ version by John Burkardt (GPL).
116  *
117  * @see Sylvan Elhay, Jaroslav Kautsky,
118  * Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
119  * Interpolatory Quadrature,
120  * ACM Transactions on Mathematical Software,
121  * Volume 13, Number 4, December 1987, pages 399-415.
122  *
123  * @see Roger Martin, James Wilkinson,
124  * The Implicit QL Algorithm,
125  * Numerische Mathematik,
126  * Volume 12, Number 5, December 1968, pages 377-383.
127  *
128  * @param[in] n The order of the matrix.
129  * @param[in,out] diag The diagonal entries [0, n-1] of the matrix.
130  * On output, the information in diag is overwritten.
131  * @param[in,out] subd The subdiagonal entries of the matrix,
132  * in entries [0, n-2]. On output,
133  * the information in subd is overwritten.
134  * @param[in,out] rhs On input, a vector. On output, the value of Q' * Z,
135  * where Q is the matrix that diagonalizes the
136  * input symmetric tridiagonal matrix.
137  */
138  template<typename RandAccIter, typename RandAccIterRHS>
139  int
140  s_tridiag_symm(std::size_t n, RandAccIter& diag, RandAccIter& subd,
141  RandAccIterRHS& rhs)
142  {
143  using Tp = std::decay_t<decltype(subd[0])>;
144 
145  const int max_iter = 50;
146  int upper = 0;
147 
148  const auto prec = std::numeric_limits<Tp>::epsilon();
149 
150  if (n == 1)
151  return 0;
152 
153  subd[n - 1] = Tp{0};
154 
155  for (int lower = 1; lower <= int(n); ++lower)
156  {
157  int j = 0;
158  while (true)
159  {
160  for (upper = lower; upper <= int(n); ++upper)
161  {
162  if (upper == int(n))
163  break;
164  if (std::abs(subd[upper - 1])
165  <= prec * (std::abs(diag[upper - 1])
166  + std::abs(diag[upper])))
167  break;
168  }
169  auto p = diag[lower - 1];
170  if (upper == lower)
171  break;
172  if (max_iter <= j)
173  throw std::runtime_error("s_tridiag_symm:"
174  " Iteration limit exceeded");
175  ++j;
176  auto g = (diag[lower] - p)
177  / (Tp{2} * subd[lower - 1]);
178  auto r = std::sqrt(g * g + Tp{1});
179  g = diag[upper - 1] - p
180  + subd[lower - 1]
181  / (g + std::copysign(std::abs(r), g));
182  auto s = Tp{1};
183  auto c = Tp{1};
184  p = Tp{0};
185 
186  const auto mml = upper - lower;
187  for (int ii = 1; ii <= mml; ii++)
188  {
189  const auto i = upper - ii;
190  auto f = s * subd[i - 1];
191  auto b = c * subd[i - 1];
192 
193  if (std::abs(g) <= std::abs(f))
194  {
195  c = g / f;
196  r = std::sqrt(c * c + Tp{1});
197  subd[i] = f * r;
198  s = Tp{1} / r;
199  c *= s;
200  }
201  else
202  {
203  s = f / g;
204  r = std::sqrt(s * s + Tp{1});
205  subd[i] = g * r;
206  c = Tp{1} / r;
207  s *= c;
208  }
209 
210  g = diag[i] - p;
211  r = (diag[i - 1] - g) * s + Tp{2} * c * b;
212  p = s * r;
213  diag[i] = g + p;
214  g = c * r - b;
215  f = rhs[i];
216  rhs[i] = s * rhs[i - 1] + c * f;
217  rhs[i - 1] = c * rhs[i - 1] - s * f;
218  }
219  diag[lower - 1] = diag[lower - 1] - p;
220  subd[lower - 1] = g;
221  subd[upper - 1] = Tp{0};
222  }
223  }
224 
225  // Sorting.
226  for (int ii = 2; ii <= upper; ++ii)
227  {
228  int i = ii - 1;
229  int k = i;
230  auto p = diag[i - 1];
231 
232  for (int j = ii; j <= int(n); ++j)
233  if (diag[j - 1] < p)
234  {
235  k = j;
236  p = diag[j - 1];
237  }
238 
239  if (k != i)
240  {
241  diag[k - 1] = diag[i - 1];
242  diag[i - 1] = p;
243  std::swap(rhs[i - 1], rhs[k - 1]);
244  }
245  }
246 
247  return 0;
248  }
