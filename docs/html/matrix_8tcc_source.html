29  /**
30  * Solve a tridiagonal system A x = b:
31  *
32  * @param[in] supd The superdiagonal of the matrix in [0, n - 2]
33  * @param[in] diag The diagonal of the matrix in [0, n - 1]
34  * @param[in] subd The subdiagonal of the matrix in [1, n - 1]
35  *
36  * @param[out] rhs The right hand side in [0, n - 1],
37  * replaced by the solution vector x
38  */
39  template<typename _RandAccIter, typename _RandAccIterRHS>
40  int
41  _S_tridiag(size_t __n,
42  _RandAccIter __supd, _RandAccIter __diag, _RandAccIter __subd,
43  _RandAccIterRHS __rhs)
44  {
45  using _Tp = std::decay_t<decltype(__subd[0])>;
46 
47  __subd[0] = __diag[0];
48 
49  if (__n == 0)
50  return 0;
51 
52  if (__n == 1)
53  {
54  __rhs[0] /= __diag[0];
55  return 0;
56  }
57 
58  __diag[0] = __supd[0];
59  __supd[0] = _Tp{0};
60  __supd[__n - 1] = _Tp{0};
61 
62  for (auto __k = 0u; __k < __n - 1; ++__k)
63  {
64  const auto __k1 = __k + 1;
65 
66  if (std::abs(__subd[__k1]) >= std::abs(__subd[__k]))
67  {
68  std::swap(__subd[__k1], __subd[__k]);
69  std::swap(__diag[__k1], __diag[__k]);
70  std::swap(__supd[__k1], __supd[__k]);
71  std::swap(__rhs[__k1], __rhs[__k]);
72  }
73 
74  if (__subd[__k] == _Tp{0})
75  return 1;
76 
77  {
78  const auto __t = -__subd[__k1] / __subd[__k];
79 
80  __subd[__k1] = __diag[__k1] + __t * __diag[__k];
81  __diag[__k1] = __supd[__k1] + __t * __supd[__k];
82  __supd[__k1] = _Tp{0};
83  __rhs[__k1] = __rhs[__k1] + __t * __rhs[__k];
84  }
85  }
86 
87  if (__subd[__n - 1] == _Tp{0})
88  return 1;
89 
90  __rhs[__n - 1] = __rhs[__n - 1] / __subd[__n - 1];
91 
92  __rhs[__n - 2] = (__rhs[__n - 2] - __diag[__n - 2] * __rhs[__n - 1])
93  / __subd[__n - 2];
94 
95  for (std::ptrdiff_t __k = __n ; __k > 2; --__k)
96  {
97  const auto __kb = __k - 3;
98  __rhs[__kb] = (__rhs[__kb] - __diag[__kb] * __rhs[__kb + 1]
99  - __supd[__kb] * __rhs[__kb + 2]) / __subd[__kb];
100  }
101 
102  return 0;
103  }

105  /**
106  * @brief Diagonalize a symmetric tridiagonal matrix.
107  *
108  * This routine is a slightly modified version of the EISPACK routine to
109  * perform the implicit QL algorithm on a symmetric tridiagonal matrix.
110  *
111  * It has been modified to produce the product Q' * Z, where Z is an input
112  * vector and Q is the orthogonal matrix diagonalizing the input matrix.
113  * The changes consist (essentialy) of applying the orthogonal
114  * transformations directly to Z as they are generated.
115  *
116  * Original C++ version by John Burkardt (GPL).
117  *
118  * @see Sylvan Elhay, Jaroslav Kautsky,
119  * Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
120  * Interpolatory Quadrature,
121  * ACM Transactions on Mathematical Software,
122  * Volume 13, Number 4, December 1987, pages 399-415.
123  *
124  * @see Roger Martin, James Wilkinson,
125  * The Implicit QL Algorithm,
126  * Numerische Mathematik,
127  * Volume 12, Number 5, December 1968, pages 377-383.
128  *
129  * @param[in] n The order of the matrix.
130  * @param[in,out] diag The diagonal entries [0, n-1] of the matrix.
131  * On output, the information in diag is overwritten.
132  * @param[in,out] subd The subdiagonal entries of the matrix,
133  * in entries [0, n-2]. On output,
134  * the information in subd is overwritten.
135  * @param[in,out] rhs On input, a vector. On output, the value of Q' * Z,
136  * where Q is the matrix that diagonalizes the
137  * input symmetric tridiagonal matrix.
138  */
139  template<typename _RandAccIter, typename _RandAccIterRHS>
140  int
141  _S_tridiag_symm(std::size_t __n, _RandAccIter& __diag, _RandAccIter& __subd,
142  _RandAccIterRHS& __rhs)
143  {
144  using _Tp = std::decay_t<decltype(__subd[0])>;
145 
146  const int __max_iter = 50;
147  int __upper = 0;
148 
149  const auto __prec = std::numeric_limits<_Tp>::epsilon();
150 
151  if (__n == 1)
152  return 0;
153 
154  __subd[__n - 1] = _Tp{0};
155 
156  for (int __lower = 1; __lower <= int(__n); ++__lower)
157  {
158  int __j = 0;
159  while (true)
160  {
161  for (__upper = __lower; __upper <= int(__n); ++__upper)
162  {
163  if (__upper == int(__n))
164  break;
165  if (std::abs(__subd[__upper - 1])
166  <= __prec * (std::abs(__diag[__upper - 1])
167  + std::abs(__diag[__upper])))
168  break;
169  }
170  auto __p = __diag[__lower - 1];
171  if (__upper == __lower)
172  break;
173  if (__max_iter <= __j)
174  std::__throw_runtime_error("_S_tridiag_symm:"
175  " Iteration limit exceeded");
176  ++__j;
177  auto __g = (__diag[__lower] - __p)
178  / (_Tp{2} * __subd[__lower - 1]);
179  auto __r = std::sqrt(__g * __g + _Tp{1});
180  __g = __diag[__upper - 1] - __p
181  + __subd[__lower - 1]
182  / (__g + std::copysign(std::abs(__r), __g));
183  auto __s = _Tp{1};
184  auto __c = _Tp{1};
185  __p = _Tp{0};
186 
187  const auto __mml = __upper - __lower;
188  for (int __ii = 1; __ii <= __mml; __ii++)
189  {
190  const auto __i = __upper - __ii;
191  auto __f = __s * __subd[__i - 1];
192  auto __b = __c * __subd[__i - 1];
193 
194  if (std::abs(__g) <= std::abs(__f))
195  {
196  __c = __g / __f;
197  __r = std::sqrt(__c * __c + _Tp{1});
198  __subd[__i] = __f * __r;
199  __s = _Tp{1} / __r;
200  __c *= __s;
201  }
202  else
203  {
204  __s = __f / __g;
205  __r = std::sqrt(__s * __s + _Tp{1});
206  __subd[__i] = __g * __r;
207  __c = _Tp{1} / __r;
208  __s *= __c;
209  }
210 
211  __g = __diag[__i] - __p;
212  __r = (__diag[__i - 1] - __g) * __s + _Tp{2} * __c * __b;
213  __p = __s * __r;
214  __diag[__i] = __g + __p;
215  __g = __c * __r - __b;
216  __f = __rhs[__i];
217  __rhs[__i] = __s * __rhs[__i - 1] + __c * __f;
218  __rhs[__i - 1] = __c * __rhs[__i - 1] - __s * __f;
219  }
220  __diag[__lower - 1] = __diag[__lower - 1] - __p;
221  __subd[__lower - 1] = __g;
222  __subd[__upper - 1] = _Tp{0};
223  }
224  }
225 
226  // Sorting.
227  for (int __ii = 2; __ii <= __upper; ++__ii)
228  {
229  int __i = __ii - 1;
230  int __k = __i;
231  auto __p = __diag[__i - 1];
232 
233  for (int __j = __ii; __j <= int(__n); ++__j)
234  if (__diag[__j - 1] < __p)
235  {
236  __k = __j;
237  __p = __diag[__j - 1];
238  }
239 
240  if (__k != __i)
241  {
242  __diag[__k - 1] = __diag[__i - 1];
243  __diag[__i - 1] = __p;
244  std::swap(__rhs[__i - 1], __rhs[__k - 1]);
245  }
246  }
247 
248  return 0;
249  }
