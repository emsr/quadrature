20 /** \file jacobi.h
21  \brief Header file for jacobi library
22 
23  The jacobi library implements Jacobi polynomials and Gauss-Jacobi quadrature related functions.
24  This library was developed to be used in high order finite element methods (Spectral Element/hp
25  methods. The interfaces were inspired by the software Polylib (http://www.nektar.info).
26 
27  The main sources used to develop this library where:
28 
29  (1) Abramowitz, Milton and Stegun, Irene (editors); "Handbook of Mathematical functions",

32  (2) Karniadakis, George Em and Sherwin, Spencer; "Spectral/hp Element Methods for Computational

41  /**
42  * \brief Enumeration gor differing types of Gauss quadrature.
43  * The gauss_quad_type is used to determine the boundary condition
44  * modifications applied to orthogonal polynomials for quadrature
45  * rules.
46  */
47  enum gauss_quad_type
48  {
49  Gauss, ///< Gauss quadrature
50  Gauss_Lobatto, ///< Gauss-Lobatto quadrature
51  Gauss_Radau_lower, ///< Gauss-Radau quadrature including the node -1
52  Gauss_Radau_upper ///< Gauss-Radau quadrature including the node +1
53  };
