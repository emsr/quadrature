39  /**
40  * Adaptively integrate a function with known singular/discontinuous points.
41  *
42  * @tparam _FuncTp A function type that takes a single real scalar
43  * argument and returns a real scalar.
44  * @tparam _Tp A real type for the limits of integration and the step.
45  * @tparam _Integrator A non-adaptive integrator that is able to return
46  * an error estimate in addition to the result.
47  *
48  * @param[in] __workspace The workspace that manages adaptive quadrature
49  * @param[in] __func The single-variable function to be integrated
50  * @param[in] __pts The sorted array of points including the integration
51  * limits and intermediate discontinuities/singularities
52  * @param[in] __max_abs_err The limit on absolute error
53  * @param[in] __max_rel_err The limit on relative error
54  * @param[in] __quad The quadrature stepper taking a function object
55  * and two integration limits
56  */
57  template<typename _Tp, typename _FuncTp,
58  typename _Integrator = gauss_kronrod_integral<_Tp>>
59  auto
60  qagp_integrate(integration_workspace<_Tp,
61  std::invoke_result_t<_FuncTp, _Tp>>& __workspace,
62  _FuncTp __func,
63  std::vector<_Tp> __pts,
64  _Tp __max_abs_err, _Tp __max_rel_err,
65  _Integrator __quad = gauss_kronrod_integral<_Tp>(Kronrod_21))
66  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
67  {
68  const auto _S_max = std::numeric_limits<_Tp>::max();
69  const auto __max_iter = __workspace.capacity();
70  const auto __n_ivals = __pts.size() - 1;
71  // Try to adjust tests for varing precision.
72  const auto _S_rel_err = std::pow(_Tp{10},
73  -std::numeric_limits<_Tp>::digits / _Tp{10});
74 
75  bool __extrapolate = false;
76  bool __allow_extrapolation = true;
77 
78  if (!valid_tolerances(__max_abs_err, __max_rel_err))
79  {
80  std::ostringstream __msg;
81  __msg << "qagp_integrate: Tolerance cannot be achieved with given "
82  "absolute (" << __max_abs_err << ") and relative ("
83  << __max_rel_err << ") error limits.";
84  std::__throw_runtime_error(__msg.str().c_str());
85  }
86 
87  if (__pts.size() > __workspace.capacity())
88  std::__throw_runtime_error("qagp_integrate: "
89  "number of points exceeds size of workspace");
90 
91  // Check that the integration range and break points
92  // are in ascending order.
93 
94  if (!std::is_sorted(std::begin(__pts), std::end(__pts)))
95  std::__throw_runtime_error("qagp_integrate: "
96  "points are not in an ascending sequence");
97 
98  __workspace.clear();
99 
100  // Perform the first integration.
101  auto __result0 = _Tp{0};
102  auto __abserr0 = _Tp{0};
103  auto __resabs0 = _Tp{0};
104  for (std::size_t __i = 0; __i < __n_ivals; ++__i)
105  {
106  const auto __lower = __pts[__i];
107  const auto __upper = __pts[__i + 1];
108 
109  auto [__area0, __error0, __resabs0, __resasc0]
110  = __quad(__func, __lower, __upper);
111 
112  __result0 += __area0;
113  __abserr0 += __error0;
114  __resabs0 += __resabs0;
115  std::size_t __level = (__error0 == __resasc0 && __error0 != _Tp{0})
116  ? 1 : 0;
117  __workspace.append(__lower, __upper, __area0, __error0, __level);
118  }
119 
120  // Compute the initial error estimate.
121  auto __errsum = _Tp{0};
122  for (std::size_t __i = 0; __i < __n_ivals; ++__i)
123  {
124  if (__workspace.depth(__i) == 1)
125  {
126  __workspace.set_abs_error(__i, __abserr0);
127  __workspace.set_depth(__i, 0);
128  }
129  __errsum += __workspace.abs_error(__i);
130  }
131 
132  // We must re-sort because the errors were reassigned.
133  __workspace.sort_error();
134 
135  auto __tolerance = std::max(__max_abs_err,
136  __max_rel_err * std::abs(__result0));
137 
138  // Compute roundoff tolerance.
139  const auto __round_off = _Tp{10} * __tolerance * __resabs0;
140 
141  if (__abserr0 <= __round_off && __abserr0 > __tolerance)
142  __throw_integration_error("qagp_integrate: "
143  "Cannot reach tolerance because "
144  "of roundoff error on first attempt",
145  ROUNDOFF_ERROR, __result0, __abserr0);
146  else if (__abserr0 <= __tolerance)
147  return {__result0, __abserr0};
148  else if (__max_iter == 1)
149  __throw_integration_error("qagp_integrate: "
150  "A maximum of one iteration was insufficient",
151  MAX_ITER_ERROR, __result0, __abserr0);
152 
153  extrapolation_table<_Tp> __table;
154  __table.append(__result0);
155 
156  auto __res_ext = __result0;
157  auto __err_ext = _S_max;
158 
159  auto __area = __result0;
160  auto __iteration = __n_ivals - 1;
161  auto __ktmin = 0u;
162  auto __ertest = __tolerance;
163  auto __error_over_large_intervals = __errsum;
164  auto __reseps = _Tp{0}, __abseps = _Tp{0}, __correc = _Tp{0};
165  int __error_type = NO_ERROR, __error_type2 = NO_ERROR;
166  int __roundoff_type1 = 0, __roundoff_type2 = 0, __roundoff_type3 = 0;
167  do
168  {
169  // Bisect the subinterval with the largest error estimate.
170  const auto& __curr = __workspace.retrieve();
171 
172  const auto __current_depth = __workspace.curr_depth() + 1;
173 
174  const auto __a1 = __curr.__lower_lim;
175  const auto __mid = (__curr.__lower_lim + __curr.__upper_lim) / _Tp{2};
176  const auto __a2 = __mid;
177  const auto __b2 = __curr.__upper_lim;
178 
179  ++__iteration;
180 
181  auto [__area1, __error1, __resabs1, __resasc1]
182  = __quad(__func, __a1, __mid);
183 
184  auto [__area2, __error2, __resabs2, __resasc2]
185  = __quad(__func, __a2, __b2);
186 
187  const auto __area12 = __area1 + __area2;
188  const auto __error12 = __error1 + __error2;
189  const auto __last_e_i = __curr.__abs_error;
190  const auto __delta = __area12 - __curr.__result;
191 
192  // Improve previous approximations to the integral and test for
193  // accuracy.
194 
195  __area += __delta;
196  __errsum += __error12 - __curr.__abs_error;
197 
198  __tolerance = std::max(__max_abs_err,
199  __max_rel_err * std::abs(__area));
200 
201  if (__resasc1 != __error1 && __resasc2 != __error2)
202  {
203  if (std::abs(__delta) <= _S_rel_err * std::abs(__area12)
204  && __error12 >= _Tp{0.99} * __curr.__abs_error)
205  {
206  if (!__extrapolate)
207  ++__roundoff_type1;
208  else
209  ++__roundoff_type2;
210  }
211  if (__iteration > 10 && __error12 > __curr.__abs_error)
212  ++__roundoff_type3;
213  }
214 
215  // Test for roundoff and eventually set error flag.
216 
217  if (__roundoff_type1 + __roundoff_type2 >= 10
218  || __roundoff_type3 >= 20)
219  __error_type = ROUNDOFF_ERROR;
220 
221  if (__roundoff_type2 >= 5)
222  __error_type2 = MAX_ITER_ERROR;
223 
224  // Set error flag in the case of bad integrand behaviour at
225  // a point of the integration range.
226 
227  if (__workspace.subinterval_too_small(__a1, __a2, __b2))
228  __error_type = EXTRAP_ROUNDOFF_ERROR;
229 
230  // Split the current interval in two.
231  __workspace.split(__mid, __area1, __error1, __area2, __error2);
232 
233  if (__errsum <= __tolerance)
234  {
235  const auto __result = __workspace.total_integral();
236  __check_error(__func__, __error_type, __result, __errsum);
237  return {__result, __errsum};
238  }
239 
240  if (__error_type != NO_ERROR)
241  break;
242 
243  if (__iteration >= __max_iter - 1)
244  {
245  __error_type = MAX_ITER_ERROR;
246  break;
247  }
248 
249  if (!__allow_extrapolation)
250  continue;
251 
252  __error_over_large_intervals -= __last_e_i;
253 
254  if (__current_depth < __workspace.max_depth())
255  __error_over_large_intervals += __error12;
256 
257  if (!__extrapolate)
258  {
259  // Test whether the interval to be bisected next is the
260  // smallest interval.
261  if (__workspace.large_interval())
262  continue;
263  else
264  {
265  __extrapolate = true;
266  __workspace.increment_curr_index();
267  }
268  }
269 
270  // The smallest interval has the largest error. Before
271  // bisecting decrease the sum of the errors over the larger
272  // intervals (error_over_large_intervals) and perform
273  // extrapolation.
274 
275  if (__error_type2 == NO_ERROR
276  && __error_over_large_intervals > __ertest)
277  if (__workspace.increment_curr_index())
278  continue;
279 
280  // Perform extrapolation.
281  __table.append(__area);
282 
283  if (__table.get_nn() < 3)
284  {
285  __workspace.reset_curr_index();
286  __extrapolate = false;
287  __error_over_large_intervals = __errsum;
288  continue;
289  }
290 
291  std::tie(__reseps, __abseps) = __table.qelg();
292 
293  ++__ktmin;
294  if (__ktmin > 5 && __err_ext < 0.001 * __errsum)
295  __error_type = DIVERGENCE_ERROR;
296 
297  if (__abseps < __err_ext)
298  {
299  __ktmin = 0;
300  __err_ext = __abseps;
301  __res_ext = __reseps;
302  __correc = __error_over_large_intervals;
303  __ertest = std::max(__max_abs_err,
304  __max_rel_err * std::abs(__reseps));
305  if (__err_ext <= __ertest)
306  break;
307  }
308 
309  // Prepare bisection of the smallest interval.
310  if (__table.get_nn() == 1)
311  __allow_extrapolation = false;
312 
313  if (__error_type == DIVERGENCE_ERROR)
314  break;
315 
316  // Work on interval with largest error.
317  __workspace.reset_curr_index();
318  __extrapolate = false;
319  __error_over_large_intervals = __errsum;
320  }
321  while (__iteration < __max_iter);
322 
323  auto __result = __res_ext;
324  auto __abserr = __err_ext;
325 
326  if (__err_ext == _S_max)
327  {
328  const auto __result = __workspace.total_integral();
329  __check_error(__func__, __error_type, __result, __errsum);
330  return {__result, __errsum};
331  }
332 
333  if (__error_type != NO_ERROR || __error_type2 != NO_ERROR)
334  {
335  if (__error_type2 != NO_ERROR)
336  __err_ext += __correc;
337 
338  if (__error_type == NO_ERROR)
339  __error_type = SINGULAR_ERROR;
340 
341  if (__result != _Tp{0} && __area != _Tp{0})
342  {
343  if (__err_ext / std::abs(__res_ext) > __errsum / std::abs(__area))
344  {
345  const auto __result = __workspace.total_integral();
346  __check_error(__func__, __error_type, __result, __errsum);
347  return {__result, __errsum};
348  }
349  }
350  else if (__err_ext > __errsum)
351  {
352  const auto __result = __workspace.total_integral();
353  __check_error(__func__, __error_type, __result, __errsum);
354  return {__result, __errsum};
355  }
356  else if (__area == _Tp{0})
357  {
358  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
359  std::__throw_runtime_error("qagp_integrate: Unknown error.");
360  }
361  }
362 
363  // Test on divergence.
364  auto __positive_integrand = __test_positivity(__result0, __resabs0);
365  auto __max_area = std::max(std::abs(__res_ext), std::abs(__area));
366  if (!__positive_integrand && __max_area < _Tp{0.01} * __resabs0)
367  {
368  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
369  std::__throw_runtime_error("qagp_integrate: Unknown error.");
370  }
371 
372  auto __ratio = __res_ext / __area;
373  if (__ratio < _Tp{0.01} || __ratio > _Tp{100}
374  || __errsum > std::abs(__area))
375  __error_type = UNKNOWN_ERROR;
376 
377  if (__error_type == NO_ERROR)
378  return {__result, __abserr};
379 
380  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
381  __throw_integration_error("qagp_integrate: Unknown error.",
382  UNKNOWN_ERROR, __result, __abserr);
383  }
