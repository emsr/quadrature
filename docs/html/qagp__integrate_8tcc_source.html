38  /**
39  * Adaptively integrate a function with known singular/discontinuous points.
40  *
41  * @tparam FuncTp A function type that takes a single real scalar
42  * argument and returns a real scalar.
43  * @tparam Tp A real type for the limits of integration and the step.
44  * @tparam Integrator A non-adaptive integrator that is able to return
45  * an error estimate in addition to the result.
46  *
47  * @param[in] workspace The workspace that manages adaptive quadrature
48  * @param[in] func The single-variable function to be integrated
49  * @param[in] pts The sorted array of points including the integration
50  * limits and intermediate discontinuities/singularities
51  * @param[in] max_abs_err The limit on absolute error
52  * @param[in] max_rel_err The limit on relative error
53  * @param[in] quad The quadrature stepper taking a function object
54  * and two integration limits
55  */
56  template<typename Tp, typename FuncTp,
57  typename Integrator = gauss_kronrod_integral<Tp>>
58  auto
59  qagp_integrate(integration_workspace<Tp,
60  std::invoke_result_t<FuncTp, Tp>>& workspace,
61  FuncTp func,
62  std::vector<Tp> pts,
63  Tp max_abs_err, Tp max_rel_err,
64  Integrator quad = gauss_kronrod_integral<Tp>(Kronrod_21))
65  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
66  {
67  using AreaTp = std::invoke_result_t<FuncTp, Tp>;
68  using AbsAreaTp = decltype(std::abs(AreaTp{}));
69 
70  const auto s_max = std::numeric_limits<Tp>::max();
71  const auto max_iter = workspace.capacity();
72  const auto n_ivals = pts.size() - 1;
73  // Try to adjust tests for varing precision.
74  const auto s_rel_err = std::pow(Tp{10},
75  -std::numeric_limits<Tp>::digits / Tp{10});
76 
77  bool extrapolate = false;
78  bool allow_extrapolation = true;
79 
80  if (!valid_tolerances(max_abs_err, max_rel_err))
81  {
82  std::ostringstream msg;
83  msg << "qagp_integrate: Tolerance cannot be achieved with given "
84  "absolute (" << max_abs_err << ") and relative ("
85  << max_rel_err << ") error limits.";
86  throw std::runtime_error(msg.str().c_str());
87  }
88 
89  if (pts.size() > workspace.capacity())
90  throw std::runtime_error("qagp_integrate: "
91  "number of points exceeds size of workspace");
92 
93  // Check that the integration range and break points
94  // are in ascending order.
95 
96  if (!std::is_sorted(std::begin(pts), std::end(pts)))
97  throw std::runtime_error("qagp_integrate: "
98  "points are not in an ascending sequence");
99 
100  workspace.clear();
101 
102  // Perform the first integration.
103  auto result0 = Tp{0};
104  auto abserr0 = Tp{0};
105  auto resabs0 = Tp{0};
106  for (std::size_t i = 0; i < n_ivals; ++i)
107  {
108  const auto lower = pts[i];
109  const auto upper = pts[i + 1];
110 
111  auto [area0, error0, resabs0, resasc0]
112  = quad(func, lower, upper);
113 
114  result0 += area0;
115  abserr0 += error0;
116  resabs0 += resabs0;
117  std::size_t level = (error0 == resasc0 && error0 != Tp{0})
118  ? 1 : 0;
119  workspace.append(lower, upper, area0, error0, level);
120  }
121 
122  // Compute the initial error estimate.
123  auto errsum = Tp{0};
124  for (std::size_t i = 0; i < n_ivals; ++i)
125  {
126  if (workspace.depth(i) == 1)
127  {
128  workspace.set_abs_error(i, abserr0);
129  workspace.set_depth(i, 0);
130  }
131  errsum += workspace.abs_error(i);
132  }
133 
134  // We must re-sort because the errors were reassigned.
135  workspace.sort_error();
136 
137  auto tolerance = std::max(max_abs_err,
138  max_rel_err * std::abs(result0));
139 
140  // Compute roundoff tolerance.
141  const auto round_off = Tp{10} * tolerance * resabs0;
142 
143  if (abserr0 <= round_off && abserr0 > tolerance)
144  throw integration_error("qagp_integrate: "
145  "Cannot reach tolerance because "
146  "of roundoff error on first attempt",
147  ROUNDOFF_ERROR, result0, abserr0);
148  else if (abserr0 <= tolerance)
149  return {result0, abserr0};
150  else if (max_iter == 1)
151  throw integration_error("qagp_integrate: "
152  "A maximum of one iteration was insufficient",
153  MAX_ITER_ERROR, result0, abserr0);
154 
155  extrapolation_table<AreaTp, AbsAreaTp> table;
156  table.append(result0);
157 
158  auto res_ext = result0;
159  auto err_ext = s_max;
160 
161  auto area = result0;
162  auto iteration = n_ivals - 1;
163  auto ktmin = 0u;
164  auto ertest = tolerance;
165  auto error_over_large_intervals = errsum;
166  auto reseps = Tp{0}, abseps = Tp{0}, correc = Tp{0};
167  int error_type = NO_ERROR, error_type2 = NO_ERROR;
168  int roundoff_type1 = 0, roundoff_type2 = 0, roundoff_type3 = 0;
169  do
170  {
171  // Bisect the subinterval with the largest error estimate.
172  const auto& curr = workspace.retrieve();
173 
174  const auto current_depth = workspace.curr_depth() + 1;
175 
176  const auto a1 = curr.lower_lim;
177  const auto mid = (curr.lower_lim + curr.upper_lim) / Tp{2};
178  const auto a2 = mid;
179  const auto b2 = curr.upper_lim;
180 
181  ++iteration;
182 
183  auto [area1, error1, resabs1, resasc1]
184  = quad(func, a1, mid);
185 
186  auto [area2, error2, resabs2, resasc2]
187  = quad(func, a2, b2);
188 
189  const auto area12 = area1 + area2;
190  const auto error12 = error1 + error2;
191  const auto last_e_i = curr.abs_error;
192  const auto delta = area12 - curr.result;
193 
194  // Improve previous approximations to the integral and test for
195  // accuracy.
196 
197  area += delta;
198  errsum += error12 - curr.abs_error;
199 
200  tolerance = std::max(max_abs_err,
201  max_rel_err * std::abs(area));
202 
203  if (resasc1 != error1 && resasc2 != error2)
204  {
205  if (std::abs(delta) <= s_rel_err * std::abs(area12)
206  && error12 >= Tp{0.99} * curr.abs_error)
207  {
208  if (!extrapolate)
209  ++roundoff_type1;
210  else
211  ++roundoff_type2;
212  }
213  if (iteration > 10 && error12 > curr.abs_error)
214  ++roundoff_type3;
215  }
216 
217  // Test for roundoff and eventually set error flag.
218 
219  if (roundoff_type1 + roundoff_type2 >= 10
220  || roundoff_type3 >= 20)
221  error_type = ROUNDOFF_ERROR;
222 
223  if (roundoff_type2 >= 5)
224  error_type2 = MAX_ITER_ERROR;
225 
226  // Set error flag in the case of bad integrand behaviour at
227  // a point of the integration range.
228 
229  if (workspace.subinterval_too_small(a1, a2, b2))
230  error_type = EXTRAP_ROUNDOFF_ERROR;
231 
232  // Split the current interval in two.
233  workspace.split(mid, area1, error1, area2, error2);
234 
235  if (errsum <= tolerance)
236  {
237  const auto result = workspace.total_integral();
238  check_error(__func__, error_type, result, errsum);
239  return {result, errsum};
240  }
241 
242  if (error_type != NO_ERROR)
243  break;
244 
245  if (iteration >= max_iter - 1)
246  {
247  error_type = MAX_ITER_ERROR;
248  break;
249  }
250 
251  if (!allow_extrapolation)
252  continue;
253 
254  error_over_large_intervals -= last_e_i;
255 
256  if (current_depth < workspace.max_depth())
257  error_over_large_intervals += error12;
258 
259  if (!extrapolate)
260  {
261  // Test whether the interval to be bisected next is the
262  // smallest interval.
263  if (workspace.large_interval())
264  continue;
265  else
266  {
267  extrapolate = true;
268  workspace.increment_curr_index();
269  }
270  }
271 
272  // The smallest interval has the largest error. Before
273  // bisecting decrease the sum of the errors over the larger
274  // intervals (error_over_large_intervals) and perform
275  // extrapolation.
276 
277  if (error_type2 == NO_ERROR
278  && error_over_large_intervals > ertest)
279  if (workspace.increment_curr_index())
280  continue;
281 
282  // Perform extrapolation.
283  table.append(area);
284 
285  if (table.get_nn() < 3)
286  {
287  workspace.reset_curr_index();
288  extrapolate = false;
289  error_over_large_intervals = errsum;
290  continue;
291  }
292 
293  std::tie(reseps, abseps) = table.qelg();
294 
295  ++ktmin;
296  if (ktmin > 5 && err_ext < 0.001 * errsum)
297  error_type = DIVERGENCE_ERROR;
298 
299  if (abseps < err_ext)
300  {
301  ktmin = 0;
302  err_ext = abseps;
303  res_ext = reseps;
304  correc = error_over_large_intervals;
305  ertest = std::max(max_abs_err,
306  max_rel_err * std::abs(reseps));
307  if (err_ext <= ertest)
308  break;
309  }
310 
311  // Prepare bisection of the smallest interval.
312  if (table.get_nn() == 1)
313  allow_extrapolation = false;
314 
315  if (error_type == DIVERGENCE_ERROR)
316  break;
317 
318  // Work on interval with largest error.
319  workspace.reset_curr_index();
320  extrapolate = false;
321  error_over_large_intervals = errsum;
322  }
323  while (iteration < max_iter);
324 
325  auto result = res_ext;
326  auto abserr = err_ext;
327 
328  if (err_ext == s_max)
329  {
330  const auto result = workspace.total_integral();
331  check_error(__func__, error_type, result, errsum);
332  return {result, errsum};
333  }
334 
335  if (error_type != NO_ERROR || error_type2 != NO_ERROR)
336  {
337  if (error_type2 != NO_ERROR)
338  err_ext += correc;
339 
340  if (error_type == NO_ERROR)
341  error_type = SINGULAR_ERROR;
342 
343  if (result != Tp{0} && area != Tp{0})
344  {
345  if (err_ext / std::abs(res_ext) > errsum / std::abs(area))
346  {
347  const auto result = workspace.total_integral();
348  check_error(__func__, error_type, result, errsum);
349  return {result, errsum};
350  }
351  }
352  else if (err_ext > errsum)
353  {
354  const auto result = workspace.total_integral();
355  check_error(__func__, error_type, result, errsum);
356  return {result, errsum};
357  }
358  else if (area == Tp{0})
359  {
360  check_error(__func__, error_type, result, abserr);
361  throw std::runtime_error("qagp_integrate: Unknown error.");
362  }
363  }
364 
365  // Test on divergence.
366  auto positive_integrand = test_positivity(result0, resabs0);
367  auto max_area = std::max(std::abs(res_ext), std::abs(area));
368  if (!positive_integrand && max_area < Tp{0.01} * resabs0)
369  {
370  check_error(__func__, error_type, result, abserr);
371  throw std::runtime_error("qagp_integrate: Unknown error.");
372  }
373 
374  auto ratio = res_ext / area;
375  if (ratio < Tp{0.01} || ratio > Tp{100}
376  || errsum > std::abs(area))
377  error_type = UNKNOWN_ERROR;
378 
379  if (error_type == NO_ERROR)
380  return {result, abserr};
381 
382  check_error(__func__, error_type, result, abserr);
383  throw integration_error("qagp_integrate: Unknown error.",
384  UNKNOWN_ERROR, result, abserr);
385  }
