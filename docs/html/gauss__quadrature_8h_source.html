249  template<typename Tp, typename FuncTp>
250  fixed_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
251  integrate_fixed_gauss_legendre(int n,
252  FuncTp func, Tp lower, Tp upper)
253  {
254  using integ_t = fixed_integral_t<Tp,
255  std::invoke_result_t<FuncTp, Tp>>;
256  using area_t = typename integ_t::AreaTp;
257 
258  if (std::isnan(lower) || std::isnan(upper))
259  {
260  const auto s_NaN = std::numeric_limits<Tp>::quiet_NaN();
261  return {area_t{} * s_NaN};
262  }
263  else if (lower == upper)
264  return {area_t{}};
265  else
266  {
267  fixed_gauss_legendre_integral<Tp> integ(n);
268  return { integ(func, lower, upper) };
269  }
270  }
