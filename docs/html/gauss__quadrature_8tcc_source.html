32  /**
33  * Build the absiscae and weights of a Gauss quadrature rule from
34  * a symmetric tridiagonal Jacobi matrix using the Golub-Welsch technique.
35  *
36  * Sylvan Elhay, Jaroslav Kautsky,
37  * Algorithm 655: IQPACK, Fortran Subroutines for the Weights of
38  * Interpolatory Quadrature,
39  * ACM Transactions on Mathematical Software,
40  * Volume 13, Number 4, December 1987, pages 399-415.
41  *
42  * @param[in] n The number of knots.
43  * @param[in] diag The diagonal of the Jacobi matrix in [0..n-1].
44  * @param[in,out] subd The subdiagonal of the Jacobi matrix in [1 ... n-1].
45  * On output subd is overwritten.
46  * @param[in] moment0 The zero-th moment of the weight function.
47  *
48  * @param[out] pt[n] The points of the integration rule.
49  * @param[out] wt[n] The weights of the integration rule.
50  */
51  template<typename _Tp, typename _InIter, typename _OutIter>
52  void
53  __golub_welsch(_Tp __moment0, int __n, _InIter& __diag, _InIter& __subd,
54  _OutIter& __pt, _OutIter& __wt)
55  {
56  // Bail if the zero-th moment is not positive.
57  if (__moment0 <= _Tp{0})
58  std::__throw_domain_error("__golub_welsch: moment0 <= 0");
59 
60  // Set up vectors for matrix diagonalization.
61  for (int i = 0; i < __n; ++i)
62  __pt[i] = __diag[i];
63 
64  __wt[0] = std::sqrt(__moment0);
65  for (int i = 1; i < __n; ++i)
66  __wt[i] = _Tp{0};
67 
68  // Diagonalize the Jacobi matrix.
69  _S_tridiag_symm(std::size_t(__n), __pt, __subd, __wt);
70 
71  for (int i = 0; i < __n; ++i)
72  __wt[i] *= __wt[i];
73 
74  return;
75  }

80  /**
81  * Construct a Gauss-Legendre rule of order @c n.
82  *
83  * Weight function: @f$ 1 @f$
84  * Constraints: @f$ b > a @f$
85  */
86  template<typename _Tp>
87  fixed_gauss_legendre_integral<_Tp>::
88  fixed_gauss_legendre_integral(int __n)
89  : order(__n),
90  point(__n),
91  weight(__n)
92  {
93  const auto __mu_0 = _Tp{2};
94 
95  std::vector<_Tp> __diag(this->order, _Tp{0});
96  std::vector<_Tp> __subd(this->order);
97 
98  for (int __i = 1; __i <= this->order; ++__i)
99  __subd[__i - 1] = _Tp(__i) / std::sqrt(_Tp(4 * __i * __i - 1));
100 
101  __detail::__golub_welsch(__mu_0, this->order, __diag, __subd,
102  this->point, this->weight);
103  }

105  /**
106  * Evaluate the integral of a function @c func from limits @c a to @c b
107  * using the Gauss-Legendre rule.
108  *
109  * Interval: @f$ (a, b) @f$
110  */
111  template<typename _Tp>
112  template<typename _FuncTp>
113  decltype(std::invoke_result_t<_FuncTp, _Tp>{} * _Tp{})
114  fixed_gauss_legendre_integral<_Tp>::
115  operator()(_FuncTp __func, _Tp __a, _Tp __b) const
116  {
117  using _RetTp = std::invoke_result_t<_FuncTp, _Tp>;
118  using _AreaTp = decltype(_RetTp{} * _Tp{});
119 
120  auto __sign = _Tp{1};
121  if (__b < __a)
122  {
123  __sign = _Tp{-1};
124  std::swap(__a, __b);
125  }
126 
127  const auto __shift = (__b + __a) / _Tp{2};
128  const auto __slope = (__b - __a) / _Tp{2};
129  const auto __fact = __slope;
130 
131  auto __sum = _AreaTp{0};
132  for (int __i = 0; __i < this->order; ++__i)
133  __sum += this->weight[__i]
134  * __func(__shift + __slope * this->point[__i]);
135 
136  return __sign * __fact * __sum;
137  }

452  /**
453  * Construct a Gauss-Jacobi rule of order @c n.
454  *
455  * Interval: @f$ (-1, +1) @f$
456  * Weight function: @f$ (1-x)^\alpha (1+x)^\beta @f$
457  * Constraints: @f$ \alpha, \beta > -1 @f$
458  */
459  template<typename _Tp>
460  fixed_gauss_jacobi_integral<_Tp>::
461  fixed_gauss_jacobi_integral(int __n, _Tp __alf, _Tp __bet)
462  : order(__n),
463  alpha(__alf),
464  beta(__bet),
465  point(__n),
466  weight(__n)
467  {
468  const auto __ab = this->alpha + this->beta;
469  auto __abp2i = __ab + _Tp{2};
470  const auto __mu_0 = std::pow(_Tp{2}, __ab + _Tp{1})
471  * std::tgamma(this->alpha + _Tp{1})
472  * std::tgamma(this->beta + _Tp{1})
473  / std::tgamma(__abp2i);
474 
475  std::vector<_Tp> __diag(this->order);
476  std::vector<_Tp> __subd(this->order);
477 
478  __diag[0] = (this->beta - this->alpha) / __abp2i;
479  __subd[0] = _Tp{2} * std::sqrt((this->alpha + _Tp{1})
480  * (this->beta + _Tp{1})
481  / (__abp2i + _Tp{1})) / __abp2i;
482  const auto a2mb2 = (this->beta - this->alpha)
483  * (this->beta + this->alpha);
484  for (int __i = 1; __i < this->order; ++__i)
485  {
486  const auto abp2ip2 = __abp2i + _Tp{2};
487  __diag[__i] = a2mb2 / __abp2i / abp2ip2;
488  const auto __ip1 = _Tp(__i + 1);
489  __subd[__i] = std::sqrt(_Tp(4 * __ip1) * (this->alpha + __ip1)
490  * (this->beta + __ip1) * (__ab + __ip1)
491  / (abp2ip2 * abp2ip2 - _Tp{1})) / abp2ip2;
492  __abp2i += _Tp{2};
493  }
494 
495  __detail::__golub_welsch(__mu_0, this->order, __diag, __subd,
496  this->point, this->weight);
497  }
