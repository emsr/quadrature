40  /**
41  * Integrates a function from finite a to finite b using an adaptive
42  * quadrature rule. The integration domain is recursively subdivided
43  * prioritized by the segment with the greatest absolute error
44  * estimate. Subdivision continues until a prescribed error target is reached
45  * or until a maximum number of divisions is performed.
46  *
47  * Once either the absolute or relative error limit is reached,
48  * qag_integrate() returns
49  *
50  * @tparam _FuncTp A function type that takes a single real scalar
51  * argument and returns a real scalar.
52  * @tparam _Tp A real type for the limits of integration and the step.
53  * @tparam _Integrator A non-adaptive integrator that is able to return
54  * an error estimate in addition to the result.
55  *
56  * @param[in] __workspace The workspace that manages adaptive quadrature
57  * @param[in] __func The single-variable function to be integrated
58  * @param[in] __lower The lower limit of integration
59  * @param[in] __upper The upper limit of integration
60  * @param[in] __max_abs_err The limit on absolute error
61  * @param[in] __max_rel_err The limit on relative error
62  * @param[in] __quad The quadrature stepper taking a function object
63  * and two integration limits
64  *
65  * @return A tuple with the first value being the integration result,
66  * and the second value being the estimated error.
67  */
68  template<typename _Tp, typename _FuncTp,
69  typename _Integrator = gauss_kronrod_integral<_Tp>>
70  auto
71  qag_integrate(integration_workspace<_Tp,
72  std::invoke_result_t<_FuncTp, _Tp>>& __workspace,
73  _FuncTp __func,
74  _Tp __lower, _Tp __upper,
75  _Tp __max_abs_err, _Tp __max_rel_err,
76  _Integrator __quad = gauss_kronrod_integral<_Tp>(Kronrod_21))
77  -> adaptive_integral_t<_Tp, std::invoke_result_t<_FuncTp, _Tp>>
78  {
79  const auto __max_iter = __workspace.capacity();
80  // Try to adjust tests for varing precision.
81  const auto _S_rel_err = std::pow(_Tp{10},
82  -std::numeric_limits<_Tp>::digits / _Tp{10});
83 
84  if (!valid_tolerances(__max_abs_err, __max_rel_err))
85  {
86  std::ostringstream __msg;
87  __msg << "qag_integrate: Tolerance cannot be achieved with given "
88  "absolute (" << __max_abs_err << ") and relative ("
89  << __max_rel_err << ") error limits.";
90  std::__throw_runtime_error(__msg.str().c_str());
91  }
92 
93  auto [__result0, __abserr0, __resabs0, __resasc0]
94  = __quad(__func, __lower, __upper);
95 
96  auto __tolerance = std::max(__max_abs_err,
97  __max_rel_err * std::abs(__result0));
98 
99  // Compute roundoff tolerance.
100  const auto __round_off = _Tp{10} * __tolerance * __resabs0;
101 
102  if (__abserr0 <= __round_off && __abserr0 > __tolerance)
103  __throw_integration_error("qag_integrate: "
104  "Cannot reach tolerance because "
105  "of roundoff error on first attempt",
106  ROUNDOFF_ERROR, __result0, __abserr0);
107  else if ((__abserr0 <= __tolerance && __abserr0 != __resasc0)
108  || __abserr0 == _Tp{0})
109  return {__result0, __abserr0};
110  else if (__max_iter == 1)
111  __throw_integration_error("qag_integrate: "
112  "A maximum of one iteration was insufficient",
113  MAX_ITER_ERROR, __result0, __abserr0);
114 
115  __workspace.clear();
116  __workspace.append(__lower, __upper, __result0, __abserr0);
117 
118  auto __area = __result0;
119  auto __errsum = __abserr0;
120  int __error_type = NO_ERROR;
121  std::size_t __iteration = 1;
122 
123  int __roundoff_type1 = 0, __roundoff_type2 = 0;
124  do
125  {
126  // Bisect the subinterval with the largest error estimate
127  const auto& __curr = __workspace.retrieve();
128 
129  const auto __a1 = __curr.__lower_lim;
130  const auto __mid = (__curr.__lower_lim + __curr.__upper_lim) / _Tp{2};
131  const auto __a2 = __mid;
132  const auto __b2 = __curr.__upper_lim;
133 
134  auto [__area1, __error1, __resabs1, __resasc1]
135  = __quad(__func, __a1, __mid);
136 
137  auto [__area2, __error2, __resabs2, __resasc2]
138  = __quad(__func, __a2, __b2);
139 
140  const auto __area12 = __area1 + __area2;
141  const auto __error12 = __error1 + __error2;
142  const auto __delta = __area12 - __curr.__result;
143 
144  __area += __delta;
145  __errsum += __error12 - __curr.__abs_error;
146 
147  __tolerance = std::max(__max_abs_err,
148  __max_rel_err * std::abs(__area));
149 
150  if (__resasc1 != __error1 && __resasc2 != __error2)
151  {
152  if (std::abs(__delta) <= _S_rel_err * std::abs(__area12)
153  && __error12 >= _Tp{0.99} * __curr.__abs_error)
154  ++__roundoff_type1;
155  if (__iteration >= 10 && __error12 > __curr.__abs_error)
156  ++__roundoff_type2;
157  }
158 
159  if (__errsum > __tolerance)
160  {
161  if (__roundoff_type1 >= 6 || __roundoff_type2 >= 20)
162  __error_type = ROUNDOFF_ERROR;
163 
164  // Set error flag in the case of bad integrand behaviour at
165  // a point of the integration range.
166  if (__workspace.subinterval_too_small(__a1, __a2, __b2))
167  __error_type = SINGULAR_ERROR;
168  }
169 
170  __workspace.split(__mid, __area1, __error1, __area2, __error2);
171 
172  ++__iteration;
173  }
174  while (__iteration < __max_iter
175  && !__error_type
176  && __errsum > __tolerance);
177 
178  auto __result = __workspace.total_integral();
179  auto __abserr = __errsum;
180 
181  if (__errsum <= __tolerance)
182  return {__result, __abserr};
183 
184  if (__error_type == NO_ERROR && __iteration >= __max_iter)
185  __error_type = MAX_ITER_ERROR;
186 
187  __check_error<_Tp>(__func__, __error_type, __result, __abserr);
188  __throw_integration_error("qag_integrate: Unknown error.",
189  UNKNOWN_ERROR, __result, __abserr);
190  }
