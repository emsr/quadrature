39  /**
40  * Integrates a function from finite a to finite b using an adaptive
41  * quadrature rule. The integration domain is recursively subdivided
42  * prioritized by the segment with the greatest absolute error
43  * estimate. Subdivision continues until a prescribed error target is reached
44  * or until a maximum number of divisions is performed.
45  *
46  * Once either the absolute or relative error limit is reached,
47  * qag_integrate() returns
48  *
49  * @tparam FuncTp A function type that takes a single real scalar
50  * argument and returns a real scalar.
51  * @tparam Tp A real type for the limits of integration and the step.
52  * @tparam Integrator A non-adaptive integrator that is able to return
53  * an error estimate in addition to the result.
54  *
55  * @param[in] workspace The workspace that manages adaptive quadrature
56  * @param[in] func The single-variable function to be integrated
57  * @param[in] lower The lower limit of integration
58  * @param[in] upper The upper limit of integration
59  * @param[in] max_abs_err The limit on absolute error
60  * @param[in] max_rel_err The limit on relative error
61  * @param[in] quad The quadrature stepper taking a function object
62  * and two integration limits
63  *
64  * @return A tuple with the first value being the integration result,
65  * and the second value being the estimated error.
66  */
67  template<typename Tp, typename FuncTp,
68  typename Integrator = gauss_kronrod_integral<Tp>>
69  auto
70  qag_integrate(integration_workspace<Tp,
71  std::invoke_result_t<FuncTp, Tp>>& workspace,
72  FuncTp func,
73  Tp lower, Tp upper,
74  Tp max_abs_err, Tp max_rel_err,
75  Integrator quad = gauss_kronrod_integral<Tp>(Kronrod_21))
76  -> adaptive_integral_t<Tp, std::invoke_result_t<FuncTp, Tp>>
77  {
78  const auto max_iter = workspace.capacity();
79  // Try to adjust tests for varing precision.
80  const auto s_rel_err = std::pow(Tp{10},
81  -std::numeric_limits<Tp>::digits / Tp{10});
82 
83  if (!valid_tolerances(max_abs_err, max_rel_err))
84  {
85  std::ostringstream msg;
86  msg << "qag_integrate: Tolerance cannot be achieved with given "
87  "absolute (" << max_abs_err << ") and relative ("
88  << max_rel_err << ") error limits.";
89  throw std::runtime_error(msg.str().c_str());
90  }
91 
92  auto [result0, abserr0, resabs0, resasc0]
93  = quad(func, lower, upper);
94 
95  auto tolerance = std::max(max_abs_err, max_rel_err * std::abs(result0));
96 
97  // Compute roundoff tolerance.
98  const auto round_off = Tp{10} * tolerance * resabs0;
99 
100  if (abserr0 <= round_off && abserr0 > tolerance)
101  throw integration_error("qag_integrate: "
102  "Cannot reach tolerance because "
103  "of roundoff error on first attempt",
104  ROUNDOFF_ERROR, result0, abserr0);
105  else if ((abserr0 <= tolerance && abserr0 != resasc0)
106  || abserr0 == Tp{0})
107  return {result0, abserr0};
108  else if (max_iter == 1)
109  throw integration_error("qag_integrate: "
110  "A maximum of one iteration was insufficient",
111  MAX_ITER_ERROR, result0, abserr0);
112 
113  workspace.clear();
114  workspace.append(lower, upper, result0, abserr0);
115 
116  auto area = result0;
117  auto errsum = abserr0;
118  int error_type = NO_ERROR;
119  std::size_t iteration = 1;
120 
121  int roundoff_type1 = 0, roundoff_type2 = 0;
122  do
123  {
124  // Bisect the subinterval with the largest error estimate
125  const auto& curr = workspace.retrieve();
126 
127  const auto a1 = curr.lower_lim;
128  const auto mid = (curr.lower_lim + curr.upper_lim) / Tp{2};
129  const auto a2 = mid;
130  const auto b2 = curr.upper_lim;
131 
132  auto [area1, error1, resabs1, resasc1]
133  = quad(func, a1, mid);
134 
135  auto [area2, error2, resabs2, resasc2]
136  = quad(func, a2, b2);
137 
138  const auto area12 = area1 + area2;
139  const auto error12 = error1 + error2;
140  const auto delta = area12 - curr.result;
141 
142  area += delta;
143  errsum += error12 - curr.abs_error;
144 
145  tolerance = std::max(max_abs_err,
146  max_rel_err * std::abs(area));
147 
148  if (resasc1 != error1 && resasc2 != error2)
149  {
150  if (std::abs(delta) <= s_rel_err * std::abs(area12)
151  && error12 >= Tp{0.99} * curr.abs_error)
152  ++roundoff_type1;
153  if (iteration >= 10 && error12 > curr.abs_error)
154  ++roundoff_type2;
155  }
156 
157  if (errsum > tolerance)
158  {
159  if (roundoff_type1 >= 6 || roundoff_type2 >= 20)
160  error_type = ROUNDOFF_ERROR;
161 
162  // Set error flag in the case of bad integrand behaviour at
163  // a point of the integration range.
164  if (workspace.subinterval_too_small(a1, a2, b2))
165  error_type = SINGULAR_ERROR;
166  }
167 
168  workspace.split(mid, area1, error1, area2, error2);
169 
170  ++iteration;
171  }
172  while (iteration < max_iter
173  && !error_type
174  && errsum > tolerance);
175 
176  auto result = workspace.total_integral();
177  auto abserr = errsum;
178 
179  if (errsum <= tolerance)
180  return {result, abserr};
181 
182  if (error_type == NO_ERROR && iteration >= max_iter)
183  error_type = MAX_ITER_ERROR;
184 
185  check_error(__func__, error_type, result, abserr);
186  throw integration_error("qag_integrate: Unknown error.",
187  UNKNOWN_ERROR, result, abserr);
188  }
