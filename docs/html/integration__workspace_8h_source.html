302  static bool
303  subinterval_too_small(_Tp __a1, _Tp __a2, _Tp __b2)
304  {
305  const auto _S_eps = _Tp{100} * std::numeric_limits<_Tp>::epsilon();
306  const auto _S_min = _Tp{1000} * std::numeric_limits<_Tp>::min();
307 
308  const auto __tmp = (_Tp{1} + _S_eps) * (std::abs(__a2) + _S_min);
309 
310  return std::abs(__a1) <= __tmp
